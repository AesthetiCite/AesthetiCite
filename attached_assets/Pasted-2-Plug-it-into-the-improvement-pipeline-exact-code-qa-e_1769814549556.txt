# app/routers/qa.py

from fastapi import APIRouter
from app.core.improve_pack import (
    improved_answer,
    RetrievalConfig,
    AnswerConfig
)
from app.core.retrieve_wrapper import retrieve_chunks

router = APIRouter()

@router.post("/qa")
async def qa_endpoint(req: QARequest):
    return improved_answer(
        query=req.query,
        filters=req.filters or {"domain": "aesthetic_medicine"},
        retrieve_chunks=retrieve_chunks,
        retrieval_cfg=RetrievalConfig(
            wide_k=60,      # IMPORTANT: fixes your benchmark
            final_k=18,
            require_pi_or_guideline_for_high_stakes=True,
            min_unique_sources_high_stakes=3,
        ),
        answer_cfg=AnswerConfig(
            compact=True,
            single_call_json=True
        ),
        enable_cache=True
    )

@router.post("/emergency")
async def emergency_endpoint(req: QARequest):
    return improved_answer(
        query=req.query,
        filters={"domain": "aesthetic_medicine", "bias": "complications"},
        retrieve_chunks=retrieve_chunks,
        retrieval_cfg=RetrievalConfig(
            wide_k=70,      # emergencies benefit from a bit more context
            final_k=22,
            require_pi_or_guideline_for_high_stakes=True,
            min_unique_sources_high_stakes=4,
        ),
        answer_cfg=AnswerConfig(
            compact=True,
            single_call_json=True
        ),
        enable_cache=True
    )
