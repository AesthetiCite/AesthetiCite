CREATE_TABLES_SQL = r"""
CREATE TABLE IF NOT EXISTS conversations (
  id            TEXT PRIMARY KEY,
  created_at    TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE IF NOT EXISTS messages (
  id             BIGSERIAL PRIMARY KEY,
  conversation_id TEXT NOT NULL REFERENCES conversations(id) ON DELETE CASCADE,
  role           TEXT NOT NULL CHECK (role IN ('user','assistant')),
  content        TEXT NOT NULL,
  created_at     TIMESTAMPTZ DEFAULT now()
);

CREATE INDEX IF NOT EXISTS idx_messages_conv_created ON messages(conversation_id, created_at DESC);
"""

@app.on_event("startup")
async def startup():
    global POOL
    POOL = await asyncpg.create_pool(
        dsn=DATABASE_URL,
        min_size=2,
        max_size=20,
        max_inactive_connection_lifetime=300,
        command_timeout=30,
    )
    async with POOL.acquire() as con:
        await con.execute(CREATE_TABLES_SQL)

@app.on_event("shutdown")
async def shutdown():
    global POOL
    if POOL:
        await POOL.close()
        POOL = None

def sse(event: str, data: Any) -> str:
    return f"event: {event}\ndata: {json.dumps(data, ensure_ascii=False)}\n\n"

# Evidence typing + authority header
# -----------------------------
def infer_evidence_type(document_type: Optional[str], journal_or_org: Optional[str]) -> str:
    t = (document_type or "").lower()
    j = (journal_or_org or "").lower()
    if "guideline" in t or "consensus" in t or "recommendation" in t:
        return "Guideline/Consensus"
    if "systematic" in t or "meta" in t:
        return "Systematic Review"
    if "random" in t or "rct" in t or "trial" in t:
        return "Randomized Trial"
    if "cohort" in t or "case-control" in t or "observational" in t:
        return "Observational Study"
    if "case report" in t or "case series" in t:
        return "Case Report/Series"
    if "review" in t:
        return "Narrative Review"
    if any(x in j for x in ["nejm", "jama", "lancet", "bmj"]):
        return "Journal Article"
    return "Other"

def evidence_rank(evidence_type: str) -> int:
    order = {
        "Guideline/Consensus": 1,
        "Systematic Review": 2,
        "Randomized Trial": 3,
        "Observational Study": 4,
        "Case Report/Series": 5,
        "Narrative Review": 6,
        "Journal Article": 7,
        "Other": 8,
    }
    return order.get(evidence_type, 9)

def compute_aci_lightweight(sources: List[Dict[str, Any]]) -> Dict[str, Any]:
    """
    Lightweight proxy ACI for display (since your real compute_aci() exists elsewhere).
    If you already have compute_aci(chunks), replace this function with yours.
    """
    typed = []
    for s in sources:
        et = infer_evidence_type(s.get("document_type"), s.get("organization_or_journal") or s.get("journal"))
        typed.append((evidence_rank(et), et))
    typed.sort(key=lambda x: x[0])
    highest = typed[0][1] if typed else "N/A"
    supporting = len(sources)
    # Simple confidence: more sources + better highest evidence => higher
    base = 6.0
    if highest in ("Guideline/Consensus", "Systematic Review"):
        base += 2.0
    elif highest == "Randomized Trial":
        base += 1.2
    base += min(2.0, supporting / 20.0)  # up to +2
    conf = max(0.0, min(10.0, base))
    gaps = []
    if supporting < 5:
        gaps.append("Limited number of directly relevant sources retrieved")
    return {
        "overall_confidence_0_10": conf,
        "highest_level": highest,
        "supporting_count": supporting,
        "gaps": gaps,
    }

def build_authority_header(question: str, sources: List[Dict[str, Any]], aci: Dict[str, Any]) -> str:
    conf = aci.get("overall_confidence_0_10")
    highest = aci.get("highest_level", "N/A")
    n_support = aci.get("supporting_count", len(sources))
    gaps = aci.get("gaps", []) or []
    conf_str = f"{conf:.1f}/10" if isinstance(conf, (int, float)) else "N/A"

    # evidence mix
    mix = []
    for s in sources[:10]:
        et = infer_evidence_type(s.get("document_type"), s.get("organization_or_journal") or s.get("journal"))
        if et not in mix:
            mix.append(et)
        if len(mix) >= 3:
            break

    gaps_text = ""
    if gaps:
        gaps_text = "\n- Evidence gaps: " + "; ".join(gaps[:3])

    safety = (
        "Safety considerations (aesthetic practice):\n"
        "- Where vascular risk exists, use conservative technique and have an occlusion response protocol.\n"
        "- Document contraindications and informed consent; reassess after conservative dosing/volumes.\n"
    )

    return (
        "Evidence Strength Summary\n"
        f"- Overall confidence: {conf_str}\n"
        f"- Highest evidence level found: {highest}\n"
        f"- Supporting sources used: {n_support}\n"
        f"- Evidence mix: {', '.join(mix) if mix else 'N/A'}"
        f"{gaps_text}\n\n"
        f"{safety}\n"
        "What is well-supported vs uncertain is stated explicitly below.\n\n"
    )

def skeleton(sources: List[Dict[str, Any]]) -> str:
    top = sources[:5]
    cites = " ".join([f"[{s.get('sid','S?')}]" for s in top])
    return (
        "Clinical summary (loading grounded answer…)\n"
        "- Evidence retrieved; generating a grounded answer with inline citations.\n"
        "- Claims will only be stated when supported by provided sources.\n\n"
        f"Top sources: {cites}\n\n"
    )

# -----------------------------
# Conversation memory
# -----------------------------
async def ensure_conversation(conversation_id: str) -> None:
    assert POOL is not None
    async with POOL.acquire() as con:
        await con.execute("INSERT INTO conversations(id) VALUES($1) ON CONFLICT DO NOTHING", conversation_id)

async def add_message(conversation_id: str, role: str, content: str) -> None:
    assert POOL is not None
    async with POOL.acquire() as con:
        await con.execute(
            "INSERT INTO messages(conversation_id, role, content) VALUES($1,$2,$3)",
            conversation_id, role, content
        )

async def fetch_recent_messages(conversation_id: str, limit: int = MEMORY_MAX_TURNS) -> List[Dict[str, Any]]:
    assert POOL is not None
    async with POOL.acquire() as con:
        rows = await con.fetch(
            "SELECT role, content, created_at FROM messages WHERE conversation_id=$1 ORDER BY created_at DESC LIMIT $2",
            conversation_id, limit
        )
    msgs = [{"role": r["role"], "content": r["content"]} for r in reversed(rows)]
    return msgs

def compress_context(messages: List[Dict[str, Any]]) -> str:
    """
    Cheap deterministic compression: take last turns, trim to MEMORY_MAX_CHARS.
    """
    parts = []
    for m in messages[-MEMORY_MAX_TURNS:]:
        role = "User" if m["role"] == "user" else "Assistant"
        txt = m["content"].strip()
        if len(txt) > 700:
            txt = txt[:700] + "…"
        parts.append(f"{role}: {txt}")
    ctx = "\n".join(parts)
    if len(ctx) > MEMORY_MAX_CHARS:
        ctx = ctx[-MEMORY_MAX_CHARS:]
    return ctx

# -----------------------------
# Single-call prompt (authority + followups + multi-cite)
# -----------------------------
def build_single_call_prompt(question: str, conversation_context: str, chunks: List[Dict[str, Any]]) -> str:
    compact = []
    for c in chunks[:MAX_SOURCES_TO_LLM]:
        compact.append({
            "sid": c["sid"],
            "t": (c.get("title") or "")[:160],
            "y": c.get("year"),
            "id": c.get("doi") or c.get("url"),
            "x": (c.get("text") or "")[:CHUNK_CHAR_CAP],
        })

    return f"""
You are AesthetiCite (evidence-grounded clinical assistant for aesthetic medicine).

Conversation context (most recent turns):
{conversation_context}

Hard rules:
1) EVERY factual claim MUST include inline citations like [S1] [S3]. You may cite multiple sources per claim.
2) If sources do not support a claim, DO NOT state it. Write "Evidence insufficient".
3) Do NOT use outside knowledge for factual claims.
4) Prefer fewer, stronger, cited points over many weak points.
5) Make uncertainty explicit.

Write exactly in this structure:
Clinical Summary (2–3 lines)

Key Evidence-Based Points
- Bullet (end with citations like [S1][S3])
- Bullet

Safety Considerations (if applicable)

Limitations / Uncertainty

Suggested Follow-up Questions
- Question 1
- Question 2
- Question 3

User question: {question}

Sources (JSON):
{json.dumps(compact, ensure_ascii=False)}
""".strip()

# -----------------------------
# Proxy streaming call
# -----------------------------
async def llm_stream(prompt: str) -> AsyncGenerator[str, None]:
    payload = {
        "model": COMPOSE_MODEL,
        "messages": [
            {"role": "system", "content": "Be concise. Follow citation rules exactly. No invented facts."},
            {"role": "user", "content": prompt},
        ],
        "temperature": COMPOSE_TEMP,
        "max_tokens": COMPOSE_MAX_TOKENS,
        "stream": True,
    }
    async with httpx.AsyncClient(timeout=None) as client:
        async with client.stream("POST", AI_PROXY_URL, json=payload) as resp:
            resp.raise_for_status()
            async for line in resp.aiter_lines():
                if not line or not line.startswith("data: "):
                    continue
                chunk = line[6:].strip()
                if chunk == "[DONE]":
                    break
                try:
                    obj = json.loads(chunk)
                    delta = obj["choices"][0].get("delta", {})
                    tok = delta.get("content")
                    if tok:
                        yield tok
                except Exception:
                    continue

# -----------------------------
# Main streaming endpoint (Authority + Memory + Linking-ready sources)
# -----------------------------
@app.post("/ask/stream")
async def ask_stream(body: AskBody, request: Request):
    q = body.q.strip()
    if not q:
        return StreamingResponse(iter([sse("error", {"message": "Empty query"})]),
                                 media_type="text/event-stream")

    conversation_id = body.conversation_id.strip()
    if not conversation_id:
        raise HTTPException(status_code=400, detail="conversation_id required")

    async def gen() -> AsyncGenerator[str, None]:
        t0 = time.perf_counter()

        # Ensure conv + store user msg
        await ensure_conversation(conversation_id)
        await add_message(conversation_id, "user", q)

        # Pull recent context (multi-turn memory)
        recent = await fetch_recent_messages(conversation_id, limit=MEMORY_MAX_TURNS)
        ctx = compress_context(recent)

        # Retrieval
        yield sse("status", {"phase": "retrieval", "message": "Searching evidence..."})
        t_r0 = time.perf_counter()
        qvec = await embed_query(q)
        r = await retrieve_chunks_unified(q, qvec, body.k)
        chunks = r["chunks"]
        sources = r["sources"]
        t_r1 = time.perf_counter()

        # Add evidence_type tags + link fields for frontend hover cards
        enriched_sources = []
        for s in sources:
            et = infer_evidence_type(s.get("document_type"), s.get("organization_or_journal") or s.get("journal"))
            s2 = dict(s)
            s2["evidence_type"] = et
            s2["evidence_rank"] = evidence_rank(et)
            enriched_sources.append(s2)

        # Authority header (visible trust)
        aci = compute_aci_lightweight(enriched_sources)  # replace with your real compute_aci if you have it
        header = build_authority_header(q, enriched_sources, aci)

        # Immediate UX
        yield sse("sources", {"sources": enriched_sources, "ms": int((t_r1 - t0) * 1000)})
        yield sse("replace", {"channel": "final", "message": "Answer (streaming)…"})
        yield sse("token", {"t": header, "channel": "final"})
        yield sse("token", {"t": skeleton(enriched_sources), "channel": "final"})

        if await request.is_disconnected():
            return

        # Single-call grounded stream WITH multi-turn context and follow-up questions
        prompt = build_single_call_prompt(q, ctx, chunks)

        yield sse("status", {"phase": "answer", "message": "Generating grounded answer..."})
        t_c0 = time.perf_counter()
        first_tok_ms: Optional[int] = None

        # Stream answer, buffer to store in conversation history at end
        buf: List[str] = []
        async for tok in llm_stream(prompt):
            if first_tok_ms is None:
                first_tok_ms = int((time.perf_counter() - t_c0) * 1000)
            buf.append(tok)
            yield sse("token", {"t": tok, "channel": "final"})
            if await request.is_disconnected():
                return

        t1 = time.perf_counter()
        answer_text = "".join(buf).strip()

        # Store assistant message (multi-turn memory)
        await add_message(conversation_id, "assistant", answer_text)

        yield sse("meta", {
            "conversation_id": conversation_id,
            "retrieval_ms": int((t_r1 - t_r0) * 1000),
            "compose_first_token_ms": first_tok_ms,
            "compose_total_ms": int((t1 - t_c0) * 1000),
            "total_ms": int((t1 - t0) * 1000),
            "model": COMPOSE_MODEL,
            "caps": {
                "k": body.k,
                "chunk_char_cap": CHUNK_CHAR_CAP,
                "max_sources_to_llm": MAX_SOURCES_TO_LLM,
                "compose_max_tokens": COMPOSE_MAX_TOKENS,
                "compose_temp": COMPOSE_TEMP,
                "memory_max_turns": MEMORY_MAX_TURNS,
                "memory_max_chars": MEMORY_MAX_CHARS,
            }
        })
        yield sse("done", {"ok": True})

    return StreamingResponse(gen(), media_type="text/event-stream")
