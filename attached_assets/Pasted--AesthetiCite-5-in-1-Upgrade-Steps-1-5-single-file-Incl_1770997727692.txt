def tag_chunk(chunk: Chunk) -> Chunk:
    t = (chunk.text + " " + chunk.title).lower()
    tags = set(chunk.tags or [])

    if any(x in t for x in INJECTABLE_TERMS): tags.add("injectables")
    if any(x in t for x in DEVICE_TERMS): tags.add("devices")
    if any(x in t for x in COMPLICATION_TERMS): tags.add("complications")
    if any(x in t for x in REG_TERMS): tags.add("regulatory")

    # best-effort evidence_type inference if missing
    if not chunk.evidence_type:
        if "guideline" in t or "consensus" in t: chunk.evidence_type = "guideline"
        elif "systematic review" in t: chunk.evidence_type = "systematic-review"
        elif "meta-analysis" in t: chunk.evidence_type = "meta-analysis"
        elif "randomized" in t or "randomised" in t: chunk.evidence_type = "RCT"
        elif "case series" in t: chunk.evidence_type = "case-series"
        elif "case report" in t: chunk.evidence_type = "case-report"
        elif "review" in t: chunk.evidence_type = "review"

    chunk.tags = sorted(tags)
    return chunk

def classify_query(query: str) -> Dict[str, Any]:
    q = query.lower()
    is_injectable = any(t in q for t in INJECTABLE_TERMS)
    is_device = any(t in q for t in DEVICE_TERMS)
    needs_reg = any(t in q for t in REG_TERMS) or is_injectable or is_device
    risk_level = "high" if any(t in q for t in HIGH_RISK_ZONE_TERMS) else ("medium" if is_injectable else "low")
    return {"is_injectable": is_injectable, "is_device": is_device, "needs_regulatory": needs_reg, "risk_level": risk_level}

# This is intentionally simple. Swap it with your hybrid retriever later.
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity

class AestheticRetriever:
    def __init__(self, chunks: List[Chunk]):
        self.chunks = chunks
        self.vectorizer = TfidfVectorizer(stop_words="english", max_features=5000)
        self.matrix = self.vectorizer.fit_transform([self._doc_text(c) for c in chunks])

    def _doc_text(self, c: Chunk) -> str:
        return f"{c.title}\n{c.text}"

    def search(self, query: str, k: int = 8, filters: Optional[Dict[str, Any]] = None) -> List[Chunk]:
        filters = filters or {}
        qv = self.vectorizer.transform([query])
        sims = cosine_similarity(qv, self.matrix)[0]

        meta = classify_query(query)

        scored: List[Tuple[float, Chunk]] = []
        for i, c in enumerate(self.chunks):
            s = float(sims[i])

            # Aesthetic boosts
            if meta["is_injectable"] and c.tags and "injectables" in c.tags: s *= 1.25
            if meta["is_device"] and c.tags and "devices" in c.tags: s *= 1.20
            if meta["needs_regulatory"] and c.tags and "regulatory" in c.tags: s *= 1.10
            if meta["is_injectable"] and c.tags and "complications" in c.tags: s *= 1.20

            # Evidence-type boost
            et = c.evidence_type
            s *= (0.9 + 0.2 * EVIDENCE_WEIGHTS.get(et, 0.4))  # gentle

            # Filters (example)
            prefer = set(filters.get("prefer", []))
            if prefer and et and et not in prefer:
                s *= 0.92  # slight penalty

            scored.append((s, c))

        scored.sort(key=lambda x: x[0], reverse=True)
        out = []
        for s, c in scored[:k]:
            cc = Chunk(**asdict(c))
            cc.score = round(s, 4)
            out.append(cc)
        return out


retriever = AestheticRetriever(DEMO_CORPUS)

def _year_modifier(year: Optional[int], now_year: int = 2026) -> float:
    if not year:
        return 0.90
    age = max(0, now_year - year)
    if age <= 2: return 1.00
    if age <= 5: return 0.95
    if age <= 10: return 0.90
    return 0.85

def _consensus_modifier(n_supporting_sources: int) -> float:
    if n_supporting_sources >= 4: return 1.00
    if n_supporting_sources == 3: return 0.95
    if n_supporting_sources == 2: return 0.90
    if n_supporting_sources == 1: return 0.80
    return 0.60

def compute_aci(evidence_items: List[EvidenceItem], risk_level: str) -> float:
    if not evidence_items:
        return 0.0
    base = sum(e.weight for e in evidence_items)
    base = min(base, 4.0)
    consensus = _consensus_modifier(len({e.chunk.source for e in evidence_items}))
    risk_penalty = {"low": 0.0, "medium": 0.6, "high": 1.2}.get(risk_level, 0.6)
    score = 10.0 * (1.0 - math.exp(-base)) * consensus
    score = max(0.0, min(10.0, score - risk_penalty))
    return round(score, 1)

def retrieve_evidence(plan: Dict[str, Any], query_meta: Dict[str, Any]) -> List[Chunk]:
    filters = {
        "domain": "aesthetic_medicine",
        "prefer": ["guideline", "meta-analysis", "systematic-review", "RCT"],
        "include_regulatory": query_meta["needs_regulatory"],
    }
    chunks: List[Chunk] = []
    seen = set()
    for q in (plan.get("search_queries") or [])[:10]:
        for c in retrieve_chunks(q, k=8, filters=filters) or []:
            key = (c.source, c.title, c.year or 0)
            if key in seen:
                continue
            seen.add(key)
            chunks.append(c)
    return chunks[:30]

def gate_claims(claims: List[Claim], chunks: List[Chunk], claim_maps: Dict[str, Any]) -> Tuple[List[Claim], Dict[str, List[EvidenceItem]]]:
    allowed: List[Claim] = []
    evidence_by_claim: Dict[str, List[EvidenceItem]] = {}

    for cl in claims:
        supports = []
        for cm in claim_maps.get("claim_maps", []):
            if (cm.get("claim") or "").strip() == cl.claim.strip():
                supports = cm.get("support", []) or []
                break

        items: List[EvidenceItem] = []
        for s in supports:
            excerpt = (s.get("excerpt") or "").strip()
            if not excerpt:
                continue

            hint = (s.get("source_hint") or "").strip()
            matched = None
            for c in chunks:
                if (c.source and c.source in hint) or (c.title and c.title in hint):
                    matched = c
                    break
            if not matched:
                continue

            et = (s.get("evidence_type") or matched.evidence_type)
            w = EVIDENCE_WEIGHTS.get(et, EVIDENCE_WEIGHTS[None]) * _year_modifier(matched.year)
            items.append(EvidenceItem(chunk=matched, excerpt=excerpt, supports=[cl.claim], weight=w))

        if items:
            allowed.append(cl)
            evidence_by_claim[cl.claim] = items

    return allowed, evidence_by_claim

def build_answer(query: str, allowed_claims: List[Claim], evidence_by_claim: Dict[str, List[EvidenceItem]], meta: Dict[str, Any], protocol: Optional[Dict[str, Any]]) -> Dict[str, Any]:
    # Build ref list
    refs = []
    ref_ids = {}
    idx = 1
    for cl in allowed_claims:
        for e in evidence_by_claim.get(cl.claim, []):
            key = (e.chunk.source, e.chunk.title, e.chunk.year or 0)
            if key in ref_ids:
                continue
            ref_ids[key] = idx
            refs.append({
                "id": idx,
                "title": e.chunk.title,
                "year": e.chunk.year,
                "evidence_type": e.chunk.evidence_type,
                "source": e.chunk.source,
                "doi": e.chunk.doi,
                "journal": e.chunk.journal,
                "retrieval_score": e.chunk.score,
                "tags": e.chunk.tags,
            })
            idx += 1

    all_evidence = [ei for lst in evidence_by_claim.values() for ei in lst]
    aci = compute_aci(all_evidence, risk_level=meta["risk_level"])

    # Tool suggestions (Step 3) — inline helpers
    tool_calls = []
    if meta["is_injectable"]:
        tool_calls.append({"name": "hyaluronidase_helper", "args": {"region": "auto_from_query", "query": query}})
        tool_calls.append({"name": "botox_dilution_helper", "args": {"vial_units": 100, "diluent_ml": 2.5}})
    if meta["is_device"]:
        tool_calls.append({"name": "fitzpatrick_laser_adjustment", "args": {"fitzpatrick": "III"}})

    tool_results = []
    for tc in tool_calls:
        tool_results.append(execute_tool(tc["name"], tc["args"]))

    # Writer pack
    pack_lines = []
    for cl in allowed_claims:
        eis = evidence_by_claim.get(cl.claim, [])
        mapped = []
        for ei in eis:
            rid = ref_ids.get((ei.chunk.source, ei.chunk.title, ei.chunk.year or 0))
            mapped.append(f"- Excerpt ({rid}): {ei.excerpt}")
        pack_lines.append(
            f"CLAIM: {cl.claim}\nRISK: {cl.risk_level}\n" + "\n".join(mapped) + "\n"
        )

    protocol_txt = ""
    if protocol:
        protocol_txt = (
            f"\nPROTOCOL_BLOCK_TITLE: {protocol['title']}\n"
            f"RED_FLAGS: {', '.join(protocol['red_flags'])}\n"
            f"IMMEDIATE_ACTIONS: {', '.join(protocol['immediate_actions'])}\n"
            f"NOTE: {protocol['note']}\n"
        )

    user = (
        f"USER QUESTION:\n{query}\n\n"
        f"OVERALL_ACI: {aci}/10\n"
        f"QUERY_META: {meta}\n"
        f"{protocol_txt}\n"
        f"EVIDENCE_PACK:\n" + "\n".join(pack_lines) + "\n"
        "REFERENCES:\n" + "\n".join([f"[{r['id']}] {r['title']} ({r.get('year','n.d.')}) {r['source']}" for r in refs])
    )

    answer_text = llm_text(system="You are AesthetiCite Writer.", user=user)

    return {
        "aci": aci,
        "answer": f"Aesthetic Confidence Index (ACI™): {aci}/10\n\n{answer_text}",
        "references": refs,
        "tools": tool_results,
        "meta": meta,
    }

            rationale=c.get("rationale", "").strip(),
            needed=c.get("needed", "").strip(),
            risk_level=(c.get("risk_level") or meta["risk_level"]).strip(),
            topic_tags=c.get("topic_tags") or [],
        )
        for c in (plan.get("claims") or [])
    ]

    chunks = retrieve_evidence(plan, meta)
    claim_maps = extract_supporting_excerpts(claims, chunks)
    allowed, evidence_by_claim = gate_claims(claims, chunks, claim_maps)

    protocol = complication_protocol_layer(query)

    if not allowed:
        return {
            "ok": True,
            "answer": (
                "Aesthetic Confidence Index (ACI™): 0/10\n\n"
                "I couldn’t find directly supporting evidence in the current corpus for the question as phrased. "
                "Try specifying the exact procedure/product/device and the outcome (efficacy, longevity, complication management)."
            ),
            "aci": 0.0,
            "references": [],
            "tools": [],
            "debug": {"meta": meta, "plan": plan, "n_chunks": len(chunks), "n_claims": len(claims), "n_allowed": 0, "latency_s": round(time.time() - t0, 2)},
        }

    built = build_answer(query, allowed, evidence_by_claim, meta, protocol)
    return {
        "ok": True,
        **built,
        "debug": {
            "plan": plan,
            "n_chunks": len(chunks),
            "n_claims": len(claims),
            "n_allowed": len(allowed),
            "latency_s": round(time.time() - t0, 2),
        }
    }

def run_benchmark() -> Dict[str, Any]:
    results = []
    n_cited_claims = 0
    n_total = 0
    n_zero_aci = 0

    for q in GOLD_QUESTIONS_30:
        out = deepconsult(q)
        n_total += 1
        if out.get("aci", 0.0) <= 0.0:
            n_zero_aci += 1

        # crude “citation presence” check in text: [1], [2]...
        cited = bool(re.search(r"\[\d+\]", out.get("answer", "")))
        if cited:
            n_cited_claims += 1

        results.append({
            "query": q,
            "aci": out.get("aci", 0.0),
            "has_citations": cited,
            "n_refs": len(out.get("references", [])),
        })

    return {
        "n": n_total,
        "avg_aci": round(sum(r["aci"] for r in results) / max(1, n_total), 2),
        "pct_with_citations": round(100.0 * n_cited_claims / max(1, n_total), 1),
        "n_zero_aci": n_zero_aci,
        "results": results,
    }

@app.post("/deepconsult/pdf/from-query")
def deepconsult_pdf(payload: QueryPayload):
    out = deepconsult(payload.query.strip())
    try:
        from reportlab.lib.pagesizes import A4
        from reportlab.pdfgen import canvas
        from reportlab.lib.units import cm
        import io

        buf = io.BytesIO()
        c = canvas.Canvas(buf, pagesize=A4)
        width, height = A4

        text = c.beginText(2 * cm, height - 2 * cm)
        text.setFont("Helvetica", 11)

        lines = (out.get("answer") or "").splitlines()
        for line in lines:
            for part in _wrap_line(line, max_chars=95):
                if text.getY() < 2 * cm:
                    c.drawText(text)
                    c.showPage()
                    text = c.beginText(2 * cm, height - 2 * cm)
                    text.setFont("Helvetica", 11)
                text.textLine(part)

        # Append references
        text.textLine("")
        text.textLine("References")
        for r in out.get("references", []):
            line = f"[{r['id']}] {r['title']} ({r.get('year','n.d.')}) — {r.get('source','')}"
            for part in _wrap_line(line, max_chars=95):
                if text.getY() < 2 * cm:
                    c.drawText(text); c.showPage()
                    text = c.beginText(2 * cm, height - 2 * cm)
                    text.setFont("Helvetica", 11)
                text.textLine(part)

        c.drawText(text)
        c.showPage()
        c.save()
        pdf_bytes = buf.getvalue()
        return Response(content=pdf_bytes, media_type="application/pdf")
    except Exception as e:
        return JSONResponse({"ok": False, "error": f"PDF generation failed: {e}", "result": out}, status_code=500)

def _wrap_line(s: str, max_chars: int = 95) -> List[str]:
    if len(s) <= max_chars:
        return [s]
    words = s.split(" ")
    lines = []
    cur = ""
    for w in words:
        if len(cur) + len(w) + 1 <= max_chars:
            cur = (cur + " " + w).strip()
        else:
            if cur:
                lines.append(cur)
            cur = w
    if cur:
        lines.append(cur)
    return lines
