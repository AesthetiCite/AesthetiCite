import React, { useMemo, useState } from "react"
import {
  AlertTriangle,
  CheckCircle2,
  Clock3,
  Eye,
  FileDown,
  ShieldAlert,
  Syringe,
  Thermometer,
  PhoneCall,
  ChevronRight,
} from "lucide-react"

type Citation = {
  id: string
  title: string
  locator?: string
  snippet?: string
}

type EmergencyResult = {
  protocol_title: string
  urgency: "Low" | "Moderate" | "High" | "Emergency"
  probable_issue: string
  summary: string
  immediate_actions: string[]
  treatment: string[]
  escalation: string[]
  monitoring: string[]
  do_not_do: string[]
  patient_message: string
  citations: Citation[]
}

const demoResult: EmergencyResult = {
  protocol_title: "Injection Chairside Safety Protocol",
  urgency: "Emergency",
  probable_issue: "Possible vascular compromise after filler injection",
  summary:
    "The entered signs are highly concerning for vascular compromise. Immediate interruption of treatment, rapid reperfusion-focused management, and escalation for visual symptoms are required.",
  immediate_actions: [
    "Stop injection immediately",
    "Keep the patient in the clinic and start focused assessment",
    "Check capillary refill, skin color, pain pattern, and livedo",
    "Massage the affected area and apply warm compress",
    "Document exact product, side, region, and time of onset",
  ],
  treatment: [
    "Prepare high-dose pulsed hyaluronidase according to clinic protocol",
    "Reassess tissue response after each treatment cycle",
    "Consider aspirin if appropriate under local protocol and patient suitability",
    "Treat this as time-critical if pain and blanching persist",
  ],
  escalation: [
    "Any visual symptom requires immediate ophthalmology / emergency escalation",
    "Escalate if livedo expands or skin turns dusky",
    "Escalate if severe pain persists despite initial management",
  ],
  monitoring: [
    "Repeat capillary refill checks every 15–30 minutes initially",
    "Track pain, blanching, livedo, and temperature change",
    "Take serial photos for follow-up documentation",
  ],
  do_not_do: [
    "Do not continue injecting",
    "Do not send the patient home without reassessment",
    "Do not delay escalation if eye symptoms are present",
  ],
  patient_message:
    "We have identified signs that need urgent treatment to protect the skin and blood flow. We are starting the emergency protocol now and will monitor you closely.",
  citations: [
    {
      id: "src_1",
      title: "Consensus guidance for vascular occlusion management",
      locator: "Page 4",
      snippet:
        "Prompt recognition and immediate treatment are critical to reduce tissue injury.",
    },
    {
      id: "src_2",
      title: "Hyaluronidase protocol review",
      locator: "Page 7",
      snippet:
        "Repeat dosing and close reassessment are preferred when clinical compromise persists.",
    },
  ],
}

function badgeClass(level: EmergencyResult["urgency"]) {
  if (level === "Emergency") return "bg-red-50 text-red-700 ring-red-200"
  if (level === "High") return "bg-orange-50 text-orange-700 ring-orange-200"
  if (level === "Moderate") return "bg-amber-50 text-amber-700 ring-amber-200"
  return "bg-emerald-50 text-emerald-700 ring-emerald-200"
}

function Section({
  title,
  icon,
  items,
  accent = "default",
}: {
  title: string
  icon: React.ReactNode
  items: string[]
  accent?: "default" | "danger"
}) {
  return (
    <div className="rounded-3xl border border-slate-200 bg-white p-5 shadow-sm">
      <div className="mb-4 flex items-center gap-3">
        <div
          className={`rounded-2xl p-2 ${
            accent === "danger" ? "bg-red-50 text-red-600" : "bg-slate-100 text-slate-700"
          }`}
        >
          {icon}
        </div>
        <h3 className="text-base font-semibold text-slate-900">{title}</h3>
      </div>
      <ul className="space-y-3">
        {items.map((item, i) => (
          <li key={i} className="flex gap-3 text-sm leading-6 text-slate-700">
            {accent === "danger" ? (
              <AlertTriangle className="mt-1 h-4 w-4 shrink-0 text-red-500" />
            ) : (
              <ChevronRight className="mt-1 h-4 w-4 shrink-0 text-slate-400" />
            )}
            <span>{item}</span>
          </li>
        ))}
      </ul>
    </div>
  )
}

export default function ChairsideEmergencyMode() {
  const [procedure, setProcedure] = useState("HA filler")
  const [product, setProduct] = useState("Juvederm Voluma")
  const [region, setRegion] = useState("Nasolabial fold")
  const [timeFromInjection, setTimeFromInjection] = useState("2 minutes")
  const [blanching, setBlanching] = useState(true)
  const [severePain, setSeverePain] = useState(true)
  const [livedo, setLivedo] = useState(true)
  const [visualSymptoms, setVisualSymptoms] = useState(false)
  const [duskySkin, setDuskySkin] = useState(false)
  const [coldSkin, setColdSkin] = useState(true)
  const [loading, setLoading] = useState(false)
  const [result, setResult] = useState<EmergencyResult | null>(demoResult)

  const triageLabel = useMemo(() => {
    if (visualSymptoms) return "VISION EMERGENCY"
    if (blanching && severePain) return "PROBABLE VASCULAR COMPROMISE"
    if (livedo || duskySkin) return "HIGH-RISK TISSUE ISCHEMIA"
    return "REQUIRES ASSESSMENT"
  }, [visualSymptoms, blanching, severePain, livedo, duskySkin])

  const reportText = useMemo(() => {
    if (!result) return ""
    return [
      "AesthetiCite Chairside Emergency Report",
      "",
      `Procedure: ${procedure}`,
      `Product: ${product}`,
      `Region: ${region}`,
      `Time from injection: ${timeFromInjection}`,
      `Blanching: ${blanching ? "Yes" : "No"}`,
      `Severe pain: ${severePain ? "Yes" : "No"}`,
      `Livedo: ${livedo ? "Yes" : "No"}`,
      `Visual symptoms: ${visualSymptoms ? "Yes" : "No"}`,
      `Dusky skin: ${duskySkin ? "Yes" : "No"}`,
      `Cold skin: ${coldSkin ? "Yes" : "No"}`,
      "",
      `Triage: ${triageLabel}`,
      `Protocol: ${result.protocol_title}`,
      `Urgency: ${result.urgency}`,
      `Probable issue: ${result.probable_issue}`,
      "",
      "Immediate actions",
      ...result.immediate_actions.map((x) => `• ${x}`),
      "",
      "Treatment",
      ...result.treatment.map((x) => `• ${x}`),
      "",
      "Escalation",
      ...result.escalation.map((x) => `• ${x}`),
      "",
      "Monitoring",
      ...result.monitoring.map((x) => `• ${x}`),
      "",
      "Do not do",
      ...result.do_not_do.map((x) => `• ${x}`),
      "",
      "Patient message",
      result.patient_message,
    ].join("\n")
  }, [
    result,
    procedure,
    product,
    region,
    timeFromInjection,
    blanching,
    severePain,
    livedo,
    visualSymptoms,
    duskySkin,
    coldSkin,
    triageLabel,
  ])

  async function runEmergencyMode() {
    setLoading(true)
    try {
      const res = await fetch("/api/safety/v2/chairside-emergency", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          procedure,
          product,
          region,
          time_from_injection: timeFromInjection,
          symptoms: {
            blanching,
            severe_pain: severePain,
            livedo,
            visual_symptoms: visualSymptoms,
            dusky_skin: duskySkin,
            cold_skin: coldSkin,
          },
        }),
      })

      const data = await res.json()
      setResult(data)
    } catch {
      setResult(demoResult)
    } finally {
      setLoading(false)
    }
  }

  function exportReport() {
    const blob = new Blob([reportText], { type: "text/plain;charset=utf-8" })
    const url = URL.createObjectURL(blob)
    const a = document.createElement("a")
    a.href = url
    a.download = "aestheticite-chairside-emergency-report.txt"
    a.click()
    URL.revokeObjectURL(url)
  }

  function Toggle({
    checked,
    onChange,
    label,
  }: {
    checked: boolean
    onChange: (v: boolean) => void
    label: string
  }) {
    return (
      <button
        type="button"
        onClick={() => onChange(!checked)}
        className={`flex items-center justify-between rounded-2xl border px-4 py-3 text-sm font-medium transition ${
          checked
            ? "border-slate-900 bg-slate-900 text-white"
            : "border-slate-200 bg-white text-slate-700 hover:bg-slate-50"
        }`}
      >
        <span>{label}</span>
        <span
          className={`ml-3 rounded-full px-2 py-0.5 text-xs ${
            checked ? "bg-white/15 text-white" : "bg-slate-100 text-slate-600"
          }`}
        >
          {checked ? "Yes" : "No"}
        </span>
      </button>
    )
  }

  return (
    <div className="min-h-screen bg-slate-50 p-4 md:p-6">
      <div className="mx-auto max-w-7xl">
        <div className="mb-6 rounded-[32px] border border-slate-200 bg-white p-6 shadow-sm">
          <div className="flex flex-col gap-5 lg:flex-row lg:items-center lg:justify-between">
            <div>
              <div className="mb-2 inline-flex items-center gap-2 rounded-full bg-red-50 px-3 py-1 text-xs font-semibold text-red-700 ring-1 ring-red-200">
                <ShieldAlert className="h-4 w-4" />
                Open during injections
              </div>
              <h1 className="text-3xl font-semibold tracking-tight text-slate-900">
                Chairside Emergency Mode
              </h1>
              <p className="mt-2 max-w-3xl text-sm leading-6 text-slate-600">
                The feature that makes doctors open AesthetiCite during injections: ultra-fast
                triage, immediate protocol, escalation guidance, and exportable documentation on
                one screen.
              </p>
            </div>

            <div className="grid grid-cols-2 gap-3 md:grid-cols-4">
              <div className="rounded-2xl border border-slate-200 p-4">
                <div className="text-xs text-slate-500">Mode</div>
                <div className="mt-1 text-sm font-semibold">Chairside</div>
              </div>
              <div className="rounded-2xl border border-slate-200 p-4">
                <div className="text-xs text-slate-500">Focus</div>
                <div className="mt-1 text-sm font-semibold">Emergency triage</div>
              </div>
              <div className="rounded-2xl border border-slate-200 p-4">
                <div className="text-xs text-slate-500">Evidence</div>
                <div className="mt-1 text-sm font-semibold">Grounded</div>
              </div>
              <div className="rounded-2xl border border-slate-200 p-4">
                <div className="text-xs text-slate-500">Use case</div>
                <div className="mt-1 text-sm font-semibold">During injection</div>
              </div>
            </div>
          </div>
        </div>

        <div className="grid grid-cols-1 gap-6 xl:grid-cols-[420px_minmax(0,1fr)]">
          <div className="space-y-6">
            <div className="rounded-[32px] border border-slate-200 bg-white p-5 shadow-sm">
              <div className="mb-4 flex items-center gap-2">
                <Syringe className="h-5 w-5 text-slate-700" />
                <h2 className="text-lg font-semibold">Fast chairside inputs</h2>
              </div>

              <div className="space-y-3">
                <input
                  className="w-full rounded-2xl border border-slate-200 px-4 py-3 text-sm outline-none placeholder:text-slate-400 focus:border-slate-400"
                  value={procedure}
                  onChange={(e) => setProcedure(e.target.value)}
                  placeholder="Procedure"
                />
                <input
                  className="w-full rounded-2xl border border-slate-200 px-4 py-3 text-sm outline-none placeholder:text-slate-400 focus:border-slate-400"
                  value={product}
                  onChange={(e) => setProduct(e.target.value)}
                  placeholder="Product"
                />
                <input
                  className="w-full rounded-2xl border border-slate-200 px-4 py-3 text-sm outline-none placeholder:text-slate-400 focus:border-slate-400"
                  value={region}
                  onChange={(e) => setRegion(e.target.value)}
                  placeholder="Region"
                />
                <div className="relative">
                  <Clock3 className="pointer-events-none absolute left-4 top-3.5 h-4 w-4 text-slate-400" />
                  <input
                    className="w-full rounded-2xl border border-slate-200 py-3 pl-11 pr-4 text-sm outline-none placeholder:text-slate-400 focus:border-slate-400"
                    value={timeFromInjection}
                    onChange={(e) => setTimeFromInjection(e.target.value)}
                    placeholder="Time from injection"
                  />
                </div>
              </div>

              <div className="mt-5 grid grid-cols-1 gap-3">
                <Toggle checked={blanching} onChange={setBlanching} label="Blanching" />
                <Toggle checked={severePain} onChange={setSeverePain} label="Severe pain" />
                <Toggle checked={livedo} onChange={setLivedo} label="Livedo" />
                <Toggle checked={visualSymptoms} onChange={setVisualSymptoms} label="Visual symptoms" />
                <Toggle checked={duskySkin} onChange={setDuskySkin} label="Dusky skin" />
                <Toggle checked={coldSkin} onChange={setColdSkin} label="Cold skin" />
              </div>

              <div className="mt-5 flex flex-wrap gap-3">
                <button
                  onClick={runEmergencyMode}
                  className="inline-flex items-center gap-2 rounded-2xl bg-slate-900 px-5 py-3 text-sm font-semibold text-white transition hover:opacity-90"
                >
                  <ShieldAlert className="h-4 w-4" />
                  {loading ? "Analyzing..." : "Run emergency mode"}
                </button>

                <button
                  onClick={exportReport}
                  className="inline-flex items-center gap-2 rounded-2xl border border-slate-200 bg-white px-5 py-3 text-sm font-semibold text-slate-700 transition hover:bg-slate-50"
                >
                  <FileDown className="h-4 w-4" />
                  Export report
                </button>
              </div>
            </div>

            <div className="rounded-[32px] border border-slate-200 bg-white p-5 shadow-sm">
              <div className="mb-3 text-sm font-semibold text-slate-700">Instant triage label</div>
              <div className="rounded-3xl bg-red-50 p-4 ring-1 ring-red-200">
                <div className="text-xs font-semibold uppercase tracking-wide text-red-700">
                  Current triage
                </div>
                <div className="mt-2 text-lg font-semibold text-red-800">{triageLabel}</div>
              </div>

              {visualSymptoms && (
                <div className="mt-4 rounded-3xl bg-amber-50 p-4 ring-1 ring-amber-200">
                  <div className="flex items-center gap-2 text-amber-800">
                    <Eye className="h-4 w-4" />
                    <span className="text-sm font-semibold">Vision emergency</span>
                  </div>
                  <p className="mt-2 text-sm leading-6 text-amber-800">
                    Visual symptoms after injection require immediate escalation. Do not delay.
                  </p>
                </div>
              )}
            </div>
          </div>

          <div className="space-y-6">
            {result && (
              <>
                <div className="rounded-[32px] border border-slate-200 bg-white p-6 shadow-sm">
                  <div className="mb-5 flex flex-col gap-4 md:flex-row md:items-start md:justify-between">
                    <div>
                      <div className="mb-2 text-xs font-semibold uppercase tracking-wide text-slate-500">
                        Live protocol
                      </div>
                      <h2 className="text-2xl font-semibold text-slate-900">{result.protocol_title}</h2>
                      <div className="mt-2 text-sm font-medium text-slate-700">
                        Probable issue: {result.probable_issue}
                      </div>
                      <p className="mt-3 max-w-3xl text-sm leading-6 text-slate-600">
                        {result.summary}
                      </p>
                    </div>

                    <span
                      className={`inline-flex items-center rounded-full px-3 py-1 text-xs font-semibold ring-1 ring-inset ${badgeClass(
                        result.urgency
                      )}`}
                    >
                      {result.urgency}
                    </span>
                  </div>

                  <div className="grid grid-cols-1 gap-4 xl:grid-cols-2">
                    <Section
                      title="Immediate actions"
                      icon={<CheckCircle2 className="h-4 w-4" />}
                      items={result.immediate_actions}
                    />
                    <Section
                      title="Treatment"
                      icon={<Syringe className="h-4 w-4" />}
                      items={result.treatment}
                    />
                    <Section
                      title="Escalation"
                      icon={<PhoneCall className="h-4 w-4" />}
                      items={result.escalation}
                      accent="danger"
                    />
                    <Section
                      title="Monitoring"
                      icon={<Thermometer className="h-4 w-4" />}
                      items={result.monitoring}
                    />
                    <div className="xl:col-span-2">
                      <Section
                        title="Do not do"
                        icon={<AlertTriangle className="h-4 w-4" />}
                        items={result.do_not_do}
                        accent="danger"
                      />
                    </div>
                  </div>
                </div>

                <div className="grid grid-cols-1 gap-6 lg:grid-cols-[1fr_0.95fr]">
                  <div className="rounded-[32px] border border-slate-200 bg-white p-6 shadow-sm">
                    <div className="mb-4 flex items-center gap-2">
                      <ShieldAlert className="h-5 w-5 text-slate-700" />
                      <h3 className="text-xl font-semibold">Patient communication</h3>
                    </div>
                    <div className="rounded-3xl bg-slate-50 p-5 text-sm leading-7 text-slate-700 ring-1 ring-slate-200">
                      {result.patient_message}
                    </div>
                  </div>

                  <div className="rounded-[32px] border border-slate-200 bg-white p-6 shadow-sm">
                    <div className="mb-4 flex items-center gap-2">
                      <FileDown className="h-5 w-5 text-slate-700" />
                      <h3 className="text-xl font-semibold">Chairside report preview</h3>
                    </div>
                    <div className="rounded-3xl bg-slate-50 p-4 ring-1 ring-slate-200">
                      <pre className="max-h-[320px] overflow-auto whitespace-pre-wrap text-xs leading-6 text-slate-700">
                        {reportText}
                      </pre>
                    </div>
                  </div>
                </div>

                <div className="rounded-[32px] border border-slate-200 bg-white p-6 shadow-sm">
                  <div className="mb-4 text-xl font-semibold text-slate-900">Grounded evidence</div>
                  <div className="mb-4 rounded-2xl bg-emerald-50 px-4 py-3 text-xs font-semibold text-emerald-700 ring-1 ring-emerald-200">
                    Citations should come only from server-validated retrieved evidence.
                  </div>

                  <div className="space-y-4">
                    {result.citations.map((c) => (
                      <div key={c.id} className="rounded-3xl border border-slate-200 p-4">
                        <div className="text-sm font-semibold text-slate-900">{c.title}</div>
                        {c.locator && (
                          <div className="mt-1 text-xs font-medium text-slate-500">{c.locator}</div>
                        )}
                        {c.snippet && (
                          <p className="mt-2 text-sm leading-6 text-slate-600">{c.snippet}</p>
                        )}
                      </div>
                    ))}
                  </div>
                </div>
              </>
            )}
          </div>
        </div>
      </div>
    </div>
  )
}
