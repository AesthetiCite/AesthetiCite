CITATION_RE = re.compile(r"\[S\d+\]")

def validate_citations(answer: str) -> bool:
    """
    Enterprise-grade minimum:
    - At least one [S#] citation exists somewhere
    - Every bullet under 'Key Evidence-Based Points' includes at least one [S#]
    """
    if not CITATION_RE.search(answer or ""):
        return False

    # Validate bullets inside Key Evidence-Based Points section (if present)
    m = re.search(r"Key Evidence-Based Points(.*?)(Safety Considerations|Limitations / Uncertainty|Suggested Follow-up Questions|$)",
                  answer, flags=re.S)
    if m:
        block = m.group(1)
        bullets = re.findall(r"^\s*-\s+(.*)$", block, flags=re.M)
        for b in bullets:
            if not CITATION_RE.search(b):
                return False
    return True

def citation_density(answer: str) -> float:
    """
    Simple, explainable metric:
    density = #citations / #bullets (Key Evidence-Based Points bullets if present, else all bullets)
    """
    if not answer:
        return 0.0
    cite_n = len(CITATION_RE.findall(answer))
    # Prefer bullets in Key Evidence-Based Points
    m = re.search(r"Key Evidence-Based Points(.*?)(Safety Considerations|Limitations / Uncertainty|Suggested Follow-up Questions|$)",
                  answer, flags=re.S)
    if m:
        bullets = re.findall(r"^\s*-\s+.*$", m.group(1), flags=re.M)
    else:
        bullets = re.findall(r"^\s*-\s+.*$", answer, flags=re.M)
    b_n = len(bullets)
    return (cite_n / b_n) if b_n else 0.0

    """
    Stronger evidence first, then newer.
    evidence_rank: 1 (best) ... 8 (weakest)
    year: higher is newer
    """
    enriched_sources.sort(
        key=lambda s: (
            int(s.get("evidence_rank") or 9),
            -int(s.get("year") or 0),
        )
    )
