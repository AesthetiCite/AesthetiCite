    // Photo upload (multipart — use multer like /api/visual/upload)
    app.post("/api/monitor/cases/:id/submit", upload.single("file"), async (req, res) => {
      try {
        const formData = new FormData();
        if (req.file) {
          formData.append("file", req.file.buffer, {
            filename: req.file.originalname || "photo.jpg",
            contentType: req.file.mimetype,
          });
        }
        formData.append("clinic_id", req.body.clinic_id || "");
        formData.append("notes", req.body.notes || "");
        const response = await fetch(
          `${PYTHON_API_BASE}/api/monitor/cases/${req.params.id}/submit`,
          {
            method: "POST",
            headers: {
              ...formData.getHeaders(),
              ...(req.headers.authorization ? { Authorization: req.headers.authorization as string } : {}),
            },
            body: formData as unknown as BodyInit,
          }
        );
        const data = await response.json();
        res.status(response.status).json(data);
      } catch (error) {
        res.status(502).json({ detail: "Monitor submit unavailable" });
      }
    });

To power the session heatmap (Contentsquare feature), instrument existing
answer components by calling the /api/analytics/events endpoint:

    // When a clinician copies an answer section:
    fetch("/api/analytics/events", {
      method: "POST",
      headers: { "Content-Type": "application/json", Authorization: `Bearer ${token}` },
      body: JSON.stringify({
        clinic_id: selectedClinic.clinic_id,
        session_id: sessionId,           // uuid per page load
        event_type: "copy",
        answer_section: "immediate_actions",   // or "evidence", "dose_guidance", etc.
        query_id: queryId,
      }),
    });

Suggested instrumentation points:
  - Copy button click           → event_type: "copy"
  - Print button click          → event_type: "print"
  - Bookmark/save click         → event_type: "save"
  - Citation card expand        → event_type: "expand", answer_section: "evidence"
  - Evidence card visible 3s+   → event_type: "view", answer_section: "evidence"
  - Protocol step visible 3s+   → event_type: "view", answer_section: step name
