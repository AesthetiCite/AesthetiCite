# complication_protocol_engine.py
# FastAPI module for AesthetiCite
#
# What this adds:
# - POST /api/complications/protocol
# - returns structured emergency protocol for aesthetic complications
# - first version uses curated protocol library
# - later you can swap protocol texts / citations with RAG retrieval
#
# How to use:
# 1) save as complication_protocol_engine.py
# 2) import router into your FastAPI app:
#       from complication_protocol_engine import router as complication_router
#       app.include_router(complication_router)
#
# Example request:
# POST /api/complications/protocol
# {
#   "query": "suspected vascular occlusion after hyaluronic acid filler",
#   "context": {
#     "region": "nasolabial fold",
#     "product_type": "hyaluronic acid filler",
#     "symptoms": ["pain", "blanching", "mottling"],
#     "time_since_injection_minutes": 15
#   }
# }

from __future__ import annotations

from typing import Any, Dict, List, Literal, Optional
from fastapi import APIRouter, HTTPException
from pydantic import BaseModel, Field
import re

router = APIRouter(prefix="/api/complications", tags=["Complication Protocol Engine"])


# ---------------------------------------------------------------------
# Pydantic models
# ---------------------------------------------------------------------

SeverityLevel = Literal["low", "moderate", "high", "critical"]
UrgencyLevel = Literal["routine", "same_day", "urgent", "immediate"]


class ClinicalContext(BaseModel):
    region: Optional[str] = None
    product_type: Optional[str] = None
    symptoms: List[str] = Field(default_factory=list)
    time_since_injection_minutes: Optional[int] = None
    visual_symptoms: Optional[bool] = None
    skin_color_change: Optional[str] = None
    pain_score_10: Optional[int] = Field(default=None, ge=0, le=10)
    free_text: Optional[str] = None


class ProtocolRequest(BaseModel):
    query: str = Field(..., min_length=3)
    context: Optional[ClinicalContext] = None


class EvidenceItem(BaseModel):
    source_id: str
    title: str
    note: str


class ProtocolStep(BaseModel):
    step_number: int
    action: str
    rationale: str


class DoseGuidance(BaseModel):
    substance: str
    recommendation: str
    notes: str


class ProtocolResponse(BaseModel):
    matched_protocol: str
    severity: SeverityLevel
    urgency: UrgencyLevel
    confidence: float
    clinical_summary: str
    immediate_actions: List[ProtocolStep]
    dose_guidance: List[DoseGuidance]
    red_flags: List[str]
    escalation: List[str]
    monitoring: List[str]
    limitations: List[str]
    follow_up_questions: List[str]
    evidence: List[EvidenceItem]
    disclaimer: str

# ---------------------------------------------------------------------
# Matching and protocol selection
# ---------------------------------------------------------------------

def normalize(text: str) -> str:
    text = text.lower().strip()
    text = re.sub(r"\s+", " ", text)
    return text


def score_protocol(query: str, context: Optional[ClinicalContext], protocol: Dict[str, Any]) -> float:
    score = 0.0
    q = normalize(query)

    for kw in protocol["keywords"]:
        if kw in q:
            score += 1.0

    if context:
        joined_symptoms = " ".join(context.symptoms).lower()
        for kw in protocol["keywords"]:
            if kw in joined_symptoms:
                score += 0.5

        if context.visual_symptoms:
            if "vascular" in protocol["name"].lower() or "occlusion" in protocol["name"].lower():
                score += 2.0

        if context.product_type and "hyaluronic" in context.product_type.lower():
            if "hyaluronic acid filler" in protocol["name"].lower() or "ha filler" in protocol["name"].lower():
                score += 1.0

        if context.skin_color_change:
            scc = context.skin_color_change.lower()
            if scc in ("blanching", "mottling", "dusky", "violaceous"):
                if "vascular" in protocol["name"].lower():
                    score += 1.5
            if "blue" in scc or "bluish" in scc:
                if "tyndall" in protocol["name"].lower():
                    score += 1.5

    return score


def select_protocol(query: str, context: Optional[ClinicalContext]) -> tuple[str, Dict[str, Any], float]:
    best_key = ""
    best_protocol: Optional[Dict[str, Any]] = None
    best_score = -1.0

    for key, protocol in PROTOCOL_LIBRARY.items():
        score = score_protocol(query, context, protocol)
        if score > best_score:
            best_key = key
            best_protocol = protocol
            best_score = score

    if not best_protocol or best_score <= 0:
        raise HTTPException(
            status_code=404,
            detail=(
                "No matching complication protocol found. Add a broader protocol library or connect "
                "this endpoint to your retrieval pipeline."
            ),
        )

    confidence = min(0.99, 0.45 + 0.08 * best_score)
    return best_key, best_protocol, round(confidence, 2)


# ---------------------------------------------------------------------
# Response builder
# ---------------------------------------------------------------------

def build_response(protocol_key: str, protocol: Dict[str, Any], confidence: float, context: Optional[ClinicalContext]) -> ProtocolResponse:
    steps = [
        ProtocolStep(step_number=i + 1, action=s["action"], rationale=s["rationale"])
        for i, s in enumerate(protocol["steps"])
    ]

    dose_guidance = [
        DoseGuidance(
            substance=d["substance"],
            recommendation=d["recommendation"],
            notes=d["notes"],
        )
        for d in protocol["dose_guidance"]
    ]

    evidence = [
        EvidenceItem(
            source_id=e["source_id"],
            title=e["title"],
            note=e["note"],
        )
        for e in protocol["evidence"]
    ]

    # Context-sensitive escalation for ocular symptoms
    escalation = list(protocol["escalation"])
    red_flags = list(protocol["red_flags"])
    monitoring = list(protocol["monitoring"])

    if context and context.visual_symptoms:
        if "Immediate ophthalmology / emergency referral for any visual symptom" not in escalation:
            escalation.insert(0, "Immediate ophthalmology / emergency referral for any visual symptom")
        if "Visual disturbance, blurred vision, diplopia, vision loss" not in red_flags:
            red_flags.insert(0, "Visual disturbance, blurred vision, diplopia, vision loss")

    return ProtocolResponse(
        matched_protocol=protocol["name"],
        severity=protocol["severity"],
        urgency=protocol["urgency"],
        confidence=confidence,
        clinical_summary=protocol["summary"],
        immediate_actions=steps,
        dose_guidance=dose_guidance,
        red_flags=red_flags,
        escalation=escalation,
        monitoring=monitoring,
        limitations=protocol["limitations"],
        follow_up_questions=protocol["follow_up_questions"],
        evidence=evidence,
        disclaimer=(
            "This output is decision support, not a substitute for clinician judgment. "
            "Escalate immediately for visual symptoms, rapidly progressive ischemia, suspected infection, "
            "or diagnostic uncertainty."
        ),
    )


# ---------------------------------------------------------------------
# API endpoint
# ---------------------------------------------------------------------

@router.post("/protocol", response_model=ProtocolResponse)
def generate_complication_protocol(payload: ProtocolRequest) -> ProtocolResponse:
    protocol_key, protocol, confidence = select_protocol(payload.query, payload.context)
    return build_response(protocol_key, protocol, confidence, payload.context)


# ---------------------------------------------------------------------
# Optional helper: programmatic engine call for internal services
# ---------------------------------------------------------------------

def run_complication_protocol_engine(query: str, context: Optional[Dict[str, Any]] = None) -> Dict[str, Any]:
    ctx = ClinicalContext(**context) if context else None
    protocol_key, protocol, confidence = select_protocol(query, ctx)
    response = build_response(protocol_key, protocol, confidence, ctx)
    return response.model_dump()
