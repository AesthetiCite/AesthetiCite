def infer_evidence_type(
    publication_type: Optional[str],
    document_type: Optional[str],
    journal_or_org: Optional[str],
    title: Optional[str] = None
) -> str:
    """
    Deterministic first: publication_type (if populated).
    Fallback: document_type / journal / title heuristics.
    """
    pt = (publication_type or "").strip().lower()
    dt = (document_type or "").strip().lower()
    j  = (journal_or_org or "").strip().lower()
    ti = (title or "").strip().lower()

    # 1) publication_type (preferred)
    # common pubmed/crossref-ish signals
    if any(k in pt for k in ["guideline", "practice guideline", "consensus", "position statement", "recommendation"]):
        return "Guideline/Consensus"
    if any(k in pt for k in ["systematic review", "meta-analysis", "meta analysis"]):
        return "Systematic Review"
    if any(k in pt for k in ["randomized", "randomised", "clinical trial", "rct", "controlled trial"]):
        return "Randomized Trial"
    if any(k in pt for k in ["cohort", "case-control", "case control", "observational", "cross-sectional", "registry"]):
        return "Observational Study"
    if any(k in pt for k in ["case report", "case series"]):
        return "Case Report/Series"
    if "review" in pt:
        return "Narrative Review"

    # 2) document_type fallback
    if any(k in dt for k in ["guideline", "consensus", "position statement", "recommendation"]):
        return "Guideline/Consensus"
    if any(k in dt for k in ["systematic", "meta-analysis", "meta analysis"]):
        return "Systematic Review"
    if any(k in dt for k in ["random", "rct", "trial", "controlled trial"]):
        return "Randomized Trial"
    if any(k in dt for k in ["cohort", "case-control", "case control", "observational", "cross-sectional"]):
        return "Observational Study"
    if any(k in dt for k in ["case report", "case series"]):
        return "Case Report/Series"
    if "review" in dt:
        return "Narrative Review"

    # 3) title heuristics (useful when metadata sparse)
    if any(k in ti for k in ["consensus", "guideline", "recommendation", "position statement"]):
        return "Guideline/Consensus"
    if any(k in ti for k in ["systematic review", "meta-analysis", "meta analysis"]):
        return "Systematic Review"
    if any(k in ti for k in ["randomized", "randomised", "trial"]):
        return "Randomized Trial"
    if any(k in ti for k in ["case report", "case series"]):
        return "Case Report/Series"
    if "review" in ti:
        return "Narrative Review"

    # 4) journal heuristic (very weak fallback)
    if any(x in j for x in ["nejm", "jama", "lancet", "bmj"]):
        return "Journal Article"

    return "Other"

def evidence_rank(evidence_type: str) -> int:
    order = {
        "Guideline/Consensus": 1,
        "Systematic Review": 2,
        "Randomized Trial": 3,
        "Observational Study": 4,
        "Case Report/Series": 5,
        "Narrative Review": 6,
        "Journal Article": 7,
        "Other": 8,
    }
    return order.get(evidence_type, 9)

def source_tier_from_evidence(evidence_type: str, journal_or_org: Optional[str]) -> str:
    """
    Simple, deterministic A/B/C tier for UI.
    """
    et = evidence_type
    j = (journal_or_org or "").lower()

    if et in ("Guideline/Consensus", "Systematic Review"):
        return "A"
    if et in ("Randomized Trial", "Observational Study"):
        return "B"
    # narrative/case/other
    return "C"

# -----------------------------
def compute_aci_engine(sources: List[Dict[str, Any]]) -> Dict[str, Any]:
    """
    Deterministic ACI (0–10) using:
    - evidence rank distribution
    - supporting count
    - recency
    - penalties for low-evidence dominance
    """
    now_year = time.gmtime().tm_year
    n = len(sources)
    if n == 0:
        return {
            "overall_confidence_0_10": 0.0,
            "highest_level": "N/A",
            "supporting_count": 0,
            "gaps": ["No sources retrieved"],
            "mix": {},
        }

    def recency_score(y: Optional[int]) -> float:
        if not y or y < 1900 or y > now_year:
            return 0.55
        age = max(0, now_year - y)
        # half-life ~6 years (tunable)
        return 2 ** (-(age / 6.0))

    mix = {}
    ranks = []
    recs = []
    for s in sources:
        et = s.get("evidence_type") or "Other"
        mix[et] = mix.get(et, 0) + 1
        ranks.append(int(s.get("evidence_rank") or 9))
        y = s.get("year")
        recs.append(recency_score(y if isinstance(y, int) else None))

    ranks.sort()
    highest = sources[min(range(n), key=lambda i: int(sources[i].get("evidence_rank") or 9))].get("evidence_type") or "Other"

    hi = sum(1 for r in ranks if r == 1)
    sr = sum(1 for r in ranks if r == 2)
    rct = sum(1 for r in ranks if r == 3)
    low = sum(1 for r in ranks if r >= 5)

    strong_frac = (hi + sr + rct) / n
    low_frac = low / n
    rec = sum(recs) / len(recs) if recs else 0.55
    support_score = min(1.0, (n ** 0.5) / (20 ** 0.5))

    base = 0.0
    base += 0.55 * strong_frac
    base += 0.20 * (hi / n)
    base += 0.15 * rec
    base += 0.10 * support_score
    base -= 0.35 * max(0.0, low_frac - 0.35)

    conf = max(0.0, min(1.0, base))
    conf_0_10 = round(conf * 10.0, 1)

    gaps = []
    if n < 5:
        gaps.append("Limited number of directly relevant sources retrieved")
    if strong_frac < 0.25:
        gaps.append("Low proportion of high-level evidence (guidelines/systematic reviews/RCTs)")
    if rec < 0.45:
        gaps.append("Evidence appears older on average (limited recent high-quality data)")
    if highest in ("Case Report/Series", "Narrative Review", "Other"):
        gaps.append("Highest evidence level found is low; interpret cautiously")

    return {
        "overall_confidence_0_10": conf_0_10,
        "highest_level": highest,
        "supporting_count": n,
        "gaps": gaps,
        "mix": mix,
    }

PUBMED_SUMMARY = "https://eutils.ncbi.nlm.nih.gov/entrez/eutils/esummary.fcgi"

async def pubmed_summary(pmid: str) -> Dict[str, Any]:
    params = {"db": "pubmed", "id": pmid, "retmode": "json"}
    async with httpx.AsyncClient(timeout=30) as client:
        r = await client.get(PUBMED_SUMMARY, params=params)
        r.raise_for_status()
        data = r.json() or {}
        result = data.get("result", {})
        return result.get(str(pmid), {}) if result else {}

def infer_pubtype_from_pubmed_summary(s: Dict[str, Any]) -> Optional[str]:
    """
    PubMed esummary often provides a list under 'pubtype' or similar keys depending on response.
    We'll try a few common keys.
    """
    for k in ("pubtype", "pubtypes", "publicationtype", "publicationtypes"):
        v = s.get(k)
        if isinstance(v, list) and v:
            # join for easy downstream matching
            return "; ".join([str(x) for x in v if x])
        if isinstance(v, str) and v.strip():
            return v.strip()
    return None

async def fetch_doc_meta(source_id: str) -> Optional[Dict[str, Any]]:
    assert POOL is not None
    async with POOL.acquire() as con:
        row = await con.fetchrow(
            """
            SELECT source_id, doi, pmid, url, title, journal, year, publication_type, organization
            FROM documents_meta WHERE source_id=$1
            """,
            source_id
        )
    return dict(row) if row else None
