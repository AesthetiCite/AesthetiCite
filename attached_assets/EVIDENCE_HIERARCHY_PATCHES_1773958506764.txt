PATCH 8 — app/main.py  (startup backfill — optional but recommended)
================================================================================

Inside the startup handler, ADD after migrations:

    # Backfill evidence metadata for all un-tagged documents
    # Run once at startup; subsequent runs are no-ops (ON CONFLICT DO UPDATE
    # WHERE evidence_type IS NULL)
    from app.engine.evidence_hierarchy import backfill_documents_meta, INSERT_META_SQL
    from app.db.session import SessionLocal
    from sqlalchemy import text

    with SessionLocal() as db:
        # Fetch un-tagged source_ids
        rows = db.execute(text("""
            SELECT DISTINCT source_id, title, year, journal
            FROM chunks
            WHERE source_id NOT IN (
                SELECT source_id FROM documents_meta
                WHERE evidence_type IS NOT NULL AND evidence_type != ''
            )
            LIMIT 10000
        """)).fetchall()

        if rows:
            chunk_dicts = [dict(r) for r in rows]
            n = backfill_documents_meta(db, chunk_dicts)
            print(f"[startup] Backfilled evidence metadata for {n} documents.")
