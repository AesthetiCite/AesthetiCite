create extension if not exists pgcrypto;

do $$
begin
  if not exists (select 1 from pg_type where typname = 'case_status') then
    create type case_status as enum ('draft', 'active', 'paused', 'closed');
  end if;

  if not exists (select 1 from pg_type where typname = 'case_mode') then
    create type case_mode as enum ('acute_complication', 'pre_procedure_safety', 'post_procedure_followup');
  end if;

  if not exists (select 1 from pg_type where typname = 'clinical_phase') then
    create type clinical_phase as enum ('intake', 'assessment', 'intervention', 'reassessment', 'disposition');
  end if;

  if not exists (select 1 from pg_type where typname = 'severity_level') then
    create type severity_level as enum ('low', 'moderate', 'high', 'critical');
  end if;

  if not exists (select 1 from pg_type where typname = 'protocol_run_status') then
    create type protocol_run_status as enum ('in_progress', 'completed', 'aborted');
  end if;

  if not exists (select 1 from pg_type where typname = 'protocol_step_status') then
    create type protocol_step_status as enum ('pending', 'ready', 'completed', 'skipped', 'blocked');
  end if;

  if not exists (select 1 from pg_type where typname = 'intervention_category') then
    create type intervention_category as enum (
      'assessment',
      'drug_administration',
      'escalation',
      'advice_given',
      'referral',
      'documentation',
      'observation'
    );
  end if;

  if not exists (select 1 from pg_type where typname = 'clinical_direction') then
    create type clinical_direction as enum ('improving', 'unchanged', 'worsening', 'resolved');
  end if;

  if not exists (select 1 from pg_type where typname = 'disposition_type') then
    create type disposition_type as enum (
      'managed_in_clinic_with_close_followup',
      'urgent_hospital_referral',
      'ophthalmology_emergency_referral',
      'ambulance_transfer',
      'resolved_and_discharged'
    );
  end if;

  if not exists (select 1 from pg_type where typname = 'review_status') then
    create type review_status as enum ('internal_draft', 'clinical_review', 'approved', 'retired');
  end if;
end $$;

create or replace function set_updated_at()
returns trigger
language plpgsql
as $$
begin
  new.updated_at = now();
  return new;
end;
$$;

create table if not exists public.clinics (
  id uuid primary key default gen_random_uuid(),
  name text not null,
  country text,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create table if not exists public.case_sessions (
  id uuid primary key default gen_random_uuid(),
  clinic_id uuid references public.clinics(id) on delete set null,
  created_by_user_id uuid not null references auth.users(id) on delete restrict,
  status case_status not null default 'draft',
  mode case_mode not null,
  chief_concern text,
  language text not null default 'en',
  country text,
  tags text[] not null default '{}',
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create table if not exists public.patient_contexts (
  id uuid primary key default gen_random_uuid(),
  case_id uuid not null unique references public.case_sessions(id) on delete cascade,
  patient_ref text,
  age int check (age is null or age >= 0),
  sex text,
  pregnancy_status text,
  weight_kg numeric(6,2),
  known_allergies jsonb not null default '[]'::jsonb,
  relevant_history jsonb not null default '[]'::jsonb,
  current_medications jsonb not null default '[]'::jsonb,
  risk_flags text[] not null default '{}',
  consent_status text,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create table if not exists public.procedure_contexts (
  id uuid primary key default gen_random_uuid(),
  case_id uuid not null unique references public.case_sessions(id) on delete cascade,
  procedure_type text not null,
  subtype text,
  treatment_date_time timestamptz,
  operator_role text,
  anatomical_areas text[] not null default '{}',
  products_used jsonb not null default '[]'::jsonb,
  technique jsonb not null default '{}'::jsonb,
  procedure_notes text,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create table if not exists public.clinical_state_snapshots (
  id uuid primary key default gen_random_uuid(),
  case_id uuid not null references public.case_sessions(id) on delete cascade,
  state_version int not null,
  phase clinical_phase not null,
  time_since_procedure_minutes int,
  symptoms jsonb not null default '[]'::jsonb,
  vital_signs jsonb not null default '{}'::jsonb,
  red_flags text[] not null default '{}',
  airway_risk boolean not null default false,
  visual_symptoms jsonb not null default '{}'::jsonb,
  photos_attached boolean not null default false,
  free_text_observations text,
  recorded_at timestamptz not null default now(),
  created_at timestamptz not null default now(),
  unique (case_id, state_version)
);

create table if not exists public.clinical_impressions (
  id uuid primary key default gen_random_uuid(),
  case_id uuid not null references public.case_sessions(id) on delete cascade,
  derived_from_state_id uuid references public.clinical_state_snapshots(id) on delete set null,
  primary_diagnosis jsonb not null,
  differentials jsonb not null default '[]'::jsonb,
  severity_level severity_level not null,
  severity_rationale text[] not null default '{}',
  vision_emergency boolean not null default false,
  anaphylaxis_probability numeric(5,4),
  needs_immediate_action boolean not null default false,
  generated_at timestamptz not null default now(),
  created_at timestamptz not null default now()
);

create table if not exists public.protocol_definitions (
  id text primary key,
  condition_code text not null,
  title text not null,
  applicable_when jsonb not null default '{}'::jsonb,
  contraindications jsonb not null default '[]'::jsonb,
  required_inputs text[] not null default '{}',
  review_status review_status not null default 'internal_draft',
  version int not null,
  step_definitions jsonb not null,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now(),
  unique (condition_code, version)
);

create table if not exists public.protocol_runs (
  id uuid primary key default gen_random_uuid(),
  case_id uuid not null references public.case_sessions(id) on delete cascade,
  protocol_id text not null references public.protocol_definitions(id) on delete restrict,
  status protocol_run_status not null default 'in_progress',
  current_step_order int,
  started_at timestamptz not null default now(),
  completed_at timestamptz,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now()
);

create table if not exists public.protocol_run_steps (
  id uuid primary key default gen_random_uuid(),
  protocol_run_id uuid not null references public.protocol_runs(id) on delete cascade,
  step_code text not null,
  step_order int not null,
  title text not null,
  status protocol_step_status not null default 'pending',
  performed_by text,
  notes text,
  dose_support jsonb not null default '{}'::jsonb,
  completed_at timestamptz,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now(),
  unique (protocol_run_id, step_order)
);

create table if not exists public.intervention_events (
  id uuid primary key default gen_random_uuid(),
  case_id uuid not null references public.case_sessions(id) on delete cascade,
  linked_run_step_id uuid references public.protocol_run_steps(id) on delete set null,
  category intervention_category not null,
  action text not null,
  details jsonb not null default '{}'::jsonb,
  performed_by text,
  immediate_response text,
  performed_at timestamptz not null default now(),
  created_at timestamptz not null default now()
);

create table if not exists public.reassessment_events (
  id uuid primary key default gen_random_uuid(),
  case_id uuid not null references public.case_sessions(id) on delete cascade,
  findings jsonb not null,
  clinical_direction clinical_direction not null,
  recommended_next_action jsonb not null default '{}'::jsonb,
  next_reassessment_due_in_minutes int,
  time_since_last_intervention_minutes int,
  performed_at timestamptz not null default now(),
  created_at timestamptz not null default now()
);

create table if not exists public.disposition_plans (
  id uuid primary key default gen_random_uuid(),
  case_id uuid not null references public.case_sessions(id) on delete cascade,
  disposition disposition_type not null,
  escalation_required boolean not null default false,
  escalation_reason text,
  followup_plan text[] not null default '{}',
  patient_advice text[] not null default '{}',
  owner text,
  decision_time timestamptz not null default now(),
  created_at timestamptz not null default now()
);

create table if not exists public.evidence_bundles (
  id uuid primary key default gen_random_uuid(),
  case_id uuid not null references public.case_sessions(id) on delete cascade,
  linked_to jsonb not null,
  claim text not null,

  evidence_strength_summary jsonb not null,
  uncertainties jsonb not null default '[]'::jsonb,
  created_at timestamptz not null default now()
);

create table if not exists public.documentation_bundles (
  id uuid primary key default gen_random_uuid(),

  documents jsonb not null,
  audit_trail_included boolean not null default true,

  generated_at timestamptz not null default now(),
  created_at timestamptz not null default now()
);

create table if not exists public.audit_events (
  id uuid primary key default gen_random_uuid(),

  actor_id uuid,

  object_type text,
  object_id text,
  change_summary text,
  timestamp timestamptz not null default now(),
  created_at timestamptz not null default now()
);

export async function createProtocolRunForCase(
  supabase: ReturnType<typeof createBrowserSupabase>,
  caseId: string,
  protocolId: string
) {
  const { data: protocol, error: protocolError } = await supabase
    .from("protocol_definitions")
    .select("*")
    .eq("id", protocolId)
    .single();

  if (protocolError) throw protocolError;

  const { data: run, error: runError } = await supabase
    .from("protocol_runs")
    .insert({
      case_id: caseId,
      protocol_id: protocolId,
      status: "in_progress",
      current_step_order: 1,
    })
    .select()
    .single();

  if (runError) throw runError;

  const steps = (protocol.step_definitions ?? []).map((s: any) => ({
    protocol_run_id: run.id,
    step_code: s.step_code,
    step_order: s.step_order,
    title: s.title,
    status: s.step_order === 1 ? "ready" : "pending",
    dose_support: {},
  }));

  const { error: stepsError } = await supabase.from("protocol_run_steps").insert(steps);
  if (stepsError) throw stepsError;

  return run as ProtocolRun;
}

from typing import Any, Dict, List

def choose_protocol(
    procedure_type: str | None,
    subtype: str | None,
    symptoms: List[Dict[str, Any]],
    visual_symptoms: Dict[str, Any] | None = None,
) -> str | None:
    symptom_names = {s.get("name") for s in symptoms if s.get("present")}

    if procedure_type == "injectable_filler":
        if {"pain", "blanching", "livedo"} & symptom_names:
            return "vascular_occlusion_v1"

    return None

def compute_alerts(
    symptoms: List[Dict[str, Any]],
    visual_symptoms: Dict[str, Any] | None = None,
    needs_immediate_action: bool = False,
) -> list[dict]:
    alerts: list[dict] = []
    symptom_names = {s.get("name") for s in symptoms if s.get("present")}

    if needs_immediate_action:
        alerts.append({
            "type": "critical",
            "message": "Immediate clinical action is indicated."
        })

    if visual_symptoms and visual_symptoms.get("present"):
        alerts.append({
            "type": "critical",
            "message": "Visual symptoms present. Escalate immediately as an ocular emergency."
        })

    if "blanching" in symptom_names or "livedo" in symptom_names:
        alerts.append({
            "type": "high",
            "message": "Signs compatible with ischemia require urgent reassessment."
        })

    return alerts

from datetime import datetime, timezone
from typing import Any, Dict, Optional

from app.models.clinical_state import CaseLiveView
from app.services.protocol_engine import compute_alerts


def _safe_first(data):
    if not data:
        return None
    return data[0]


def build_case_live_view(
    case_row: Dict[str, Any],
    patient_row: Optional[Dict[str, Any]],
    procedure_row: Optional[Dict[str, Any]],
    latest_state: Optional[Dict[str, Any]],
    latest_impression: Optional[Dict[str, Any]],
    latest_protocol_run: Optional[Dict[str, Any]],
    next_step: Optional[Dict[str, Any]],
    latest_reassessment: Optional[Dict[str, Any]],
    latest_evidence: Optional[Dict[str, Any]],
) -> CaseLiveView:
    severity = latest_impression["severity_level"] if latest_impression else None

    minutes_since_procedure = None
    if procedure_row and procedure_row.get("treatment_date_time"):
        try:
            t = datetime.fromisoformat(
                procedure_row["treatment_date_time"].replace("Z", "+00:00")
            )
            now = datetime.now(timezone.utc)
            minutes_since_procedure = int((now - t).total_seconds() // 60)
        except Exception:
            minutes_since_procedure = latest_state.get("time_since_procedure_minutes") if latest_state else None
    elif latest_state:
        minutes_since_procedure = latest_state.get("time_since_procedure_minutes")

    alerts = compute_alerts(
        symptoms=latest_state.get("symptoms", []) if latest_state else [],
        visual_symptoms=latest_state.get("visual_symptoms", {}) if latest_state else {},
        needs_immediate_action=latest_impression.get("needs_immediate_action", False) if latest_impression else False,
    )

    evidence_summary = None
    if latest_evidence:
        ess = latest_evidence.get("evidence_strength_summary", {})
        evidence_summary = {
            "aci_score": ess.get("aci_score"),
            "confidence_label": ess.get("confidence_label"),
            "top_sources": [
                item.get("citation_label")
                for item in latest_evidence.get("evidence_items", [])[:3]
                if item.get("citation_label")
            ],
        }

    return CaseLiveView(
        case_header={
            "case_id": case_row["id"],
            "chief_concern": case_row.get("chief_concern"),
            "status": case_row["status"],
            "severity": severity,
        },
        patient_summary={
            "age": patient_row.get("age") if patient_row else None,
            "risk_flags": patient_row.get("risk_flags", []) if patient_row else [],
        } if patient_row else None,
        procedure_summary={
            "procedure_type": procedure_row.get("procedure_type") if procedure_row else None,
            "area": procedure_row.get("anatomical_areas", []) if procedure_row else [],
            "product": [p.get("product_name") for p in procedure_row.get("products_used", [])] if procedure_row else [],
        } if procedure_row else None,
        current_impression={
            "primary_diagnosis": latest_impression.get("primary_diagnosis", {}).get("label") if latest_impression else None,
            "confidence": latest_impression.get("primary_diagnosis", {}).get("confidence") if latest_impression else None,
            "needs_immediate_action": latest_impression.get("needs_immediate_action") if latest_impression else None,
        } if latest_impression else None,
        current_protocol={
            "title": latest_protocol_run.get("protocol_definition", {}).get("title") if latest_protocol_run else None,
            "current_step_order": latest_protocol_run.get("current_step_order") if latest_protocol_run else None,
            "next_step": next_step.get("title") if next_step else None,
        } if latest_protocol_run else None,
        timers={
            "minutes_since_procedure": minutes_since_procedure,
            "next_reassessment_due_in_minutes": latest_reassessment.get("next_reassessment_due_in_minutes") if latest_reassessment else None,
        },
        alerts=alerts,
        evidence_summary=evidence_summary,
    )

@router.post("/{case_id}/auto-start-protocol", response_model=ProtocolRunRead)
def auto_start_protocol(case_id: str):
    sb = get_supabase()

    proc_res = sb.table("procedure_contexts").select("*").eq("case_id", case_id).single().execute()
    state_res = (
        sb.table("clinical_state_snapshots")
        .select("*")
        .eq("case_id", case_id)
        .order("state_version", desc=True)
        .limit(1)
        .execute()
    )

    procedure = proc_res.data if proc_res.data else None
    latest_state = state_res.data[0] if state_res.data else None

    if not procedure or not latest_state:
        raise HTTPException(status_code=400, detail="Procedure context and latest state are required")

    protocol_id = choose_protocol(
        procedure_type=procedure.get("procedure_type"),
        subtype=procedure.get("subtype"),
        symptoms=latest_state.get("symptoms", []),
        visual_symptoms=latest_state.get("visual_symptoms", {}),
    )

    if not protocol_id:
        raise HTTPException(status_code=400, detail="No matching protocol found")

    existing = (
        sb.table("protocol_runs")
        .select("*")
        .eq("case_id", case_id)
        .eq("status", "in_progress")
        .limit(1)
        .execute()
    )
    if existing.data:
        return existing.data[0]

    protocol_def = sb.table("protocol_definitions").select("*").eq("id", protocol_id).single().execute()
    if not protocol_def.data:
        raise HTTPException(status_code=404, detail="Protocol definition not found")

    run_res = (
        sb.table("protocol_runs")
        .insert({
            "case_id": case_id,
            "protocol_id": protocol_id,
            "status": "in_progress",
            "current_step_order": 1,
        })
        .select("*")
        .single()
        .execute()
    )
    run = run_res.data

    steps = []
    for step in protocol_def.data["step_definitions"]:
        steps.append({
            "protocol_run_id": run["id"],
            "step_code": step["step_code"],
            "step_order": step["step_order"],
            "title": step["title"],
            "status": "ready" if step["step_order"] == 1 else "pending",
            "dose_support": {},
        })

    sb.table("protocol_run_steps").insert(steps).execute()
    return run

@router.patch("/protocol-run-steps/{step_id}", response_model=ProtocolRunStepRead)
def update_protocol_run_step(step_id: str, payload: ProtocolRunStepUpdate):
    sb = get_supabase()
    body = {k: v for k, v in payload.model_dump().items() if v is not None}
    res = sb.table("protocol_run_steps").update(body).eq("id", step_id).select("*").single().execute()
    if not res.data:
        raise HTTPException(status_code=404, detail="Protocol step not found")

    step = res.data
    run_id = step["protocol_run_id"]

    run_res = sb.table("protocol_runs").select("*").eq("id", run_id).single().execute()
    if run_res.data:
        current_order = step["step_order"]
        next_step_res = (
            sb.table("protocol_run_steps")
            .select("*")
            .eq("protocol_run_id", run_id)
            .eq("step_order", current_order + 1)
            .limit(1)
            .execute()
        )

        if step["status"] == "completed" and next_step_res.data:
            next_step = next_step_res.data[0]
            if next_step["status"] == "pending":
                sb.table("protocol_run_steps").update({"status": "ready"}).eq("id", next_step["id"]).execute()
            sb.table("protocol_runs").update({"current_step_order": current_order + 1}).eq("id", run_id).execute()

    return step

@router.get("/{case_id}/live-view", response_model=CaseLiveView)
def get_case_live_view(case_id: str):
    sb = get_supabase()

    case_res = sb.table("case_sessions").select("*").eq("id", case_id).single().execute()
    if not case_res.data:
        raise HTTPException(status_code=404, detail="Case not found")

    patient_res = sb.table("patient_contexts").select("*").eq("case_id", case_id).limit(1).execute()
    procedure_res = sb.table("procedure_contexts").select("*").eq("case_id", case_id).limit(1).execute()

    state_res = (
        sb.table("clinical_state_snapshots")
        .select("*")
        .eq("case_id", case_id)
        .order("state_version", desc=True)
        .limit(1)
        .execute()
    )

    impression_res = (
        sb.table("clinical_impressions")
        .select("*")
        .eq("case_id", case_id)
        .order("generated_at", desc=True)
        .limit(1)
        .execute()
    )

    run_res = (
        sb.table("protocol_runs")
        .select("*, protocol_definition:protocol_definitions(*)")
        .eq("case_id", case_id)
        .order("started_at", desc=True)
        .limit(1)
        .execute()
    )

    reassessment_res = (
        sb.table("reassessment_events")
        .select("*")
        .eq("case_id", case_id)
        .order("performed_at", desc=True)
        .limit(1)
        .execute()
    )

    evidence_res = (
        sb.table("evidence_bundles")
        .select("*")
        .eq("case_id", case_id)
        .order("created_at", desc=True)
        .limit(1)
        .execute()
    )

    patient_row = patient_res.data[0] if patient_res.data else None
    procedure_row = procedure_res.data[0] if procedure_res.data else None
    latest_state = state_res.data[0] if state_res.data else None
    latest_impression = impression_res.data[0] if impression_res.data else None
    latest_protocol_run = run_res.data[0] if run_res.data else None
    latest_reassessment = reassessment_res.data[0] if reassessment_res.data else None
    latest_evidence = evidence_res.data[0] if evidence_res.data else None

    next_step = None
    if latest_protocol_run:
        next_order = latest_protocol_run.get("current_step_order")
        if next_order:
            step_res = (
                sb.table("protocol_run_steps")
                .select("*")
                .eq("protocol_run_id", latest_protocol_run["id"])
                .eq("step_order", next_order)
                .limit(1)
                .execute()
            )
            if step_res.data:
                next_step = step_res.data[0]

    return build_case_live_view(
        case_row=case_res.data,
        patient_row=patient_row,
        procedure_row=procedure_row,
        latest_state=latest_state,
        latest_impression=latest_impression,
        latest_protocol_run=latest_protocol_run,
        next_step=next_step,
        latest_reassessment=latest_reassessment,
        latest_evidence=latest_evidence,
    )
