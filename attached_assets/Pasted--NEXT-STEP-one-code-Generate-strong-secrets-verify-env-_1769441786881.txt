import secrets
import string

def token(n=64):
    return secrets.token_urlsafe(n)

def strong_password(n=40):
    alphabet = string.ascii_letters + string.digits + "!@#$%^&*()-_=+"
    return "".join(secrets.choice(alphabet) for _ in range(n))

jwt_secret = token(64)
admin_key = token(48)

echo "==> 3) Add a startup check: refuse to boot if secrets are weak/missing (production safety)"
cat > app/core/env_check.py <<'EOF'
from app.core.config import settings

def validate_env():
    # Basic checks; keep lenient in dev
    if settings.ENV.lower() in ("prod", "production"):
        if not settings.JWT_SECRET or settings.JWT_SECRET == "change-this-secret":
            raise RuntimeError("JWT_SECRET is missing/weak. Set it in environment variables.")
        if not settings.ADMIN_API_KEY or settings.ADMIN_API_KEY == "change-me":
            raise RuntimeError("ADMIN_API_KEY is missing/weak. Set it in environment variables.")
EOF

echo "==> 4) Wire env_check into app/main.py startup"
python - <<'PY'
import pathlib, re

p = pathlib.Path("app/main.py")
txt = p.read_text()

# Insert import
if "validate_env" not in txt:
    txt = txt.replace(
        "from app.core.startup import ensure_users_table",
        "from app.core.startup import ensure_users_table\nfrom app.core.env_check import validate_env"
    )

# Call validate_env() right after app creation (safe)
if "validate_env()" not in txt:
    txt = re.sub(r"app = FastAPI\(title=settings\.APP_NAME\)\n", "app = FastAPI(title=settings.APP_NAME)\n\n# Validate critical environment variables\nvalidate_env()\n", txt)

p.write_text(txt)
PY
