"""
ONE code to implement the necessary improvements vs OpenEvidence:

Goal: beat/meet OpenEvidence "speed feel" by removing the PLAN call entirely.
✅ Single LLM streaming call (no blocking plan phase)
✅ Still enforces "no cite => no claim" inside the prompt
✅ Streams sources immediately + deterministic skeleton instantly
✅ Then streams the final grounded answer from ONE call
✅ Keeps your pgvector retrieval as-is (fast) and uses your proxy

You only plug TWO functions:
  1) embed_query(q) -> 384-dim vector
  2) retrieve_chunks_unified(q, qvec, k) -> {"chunks":[...], "sources":[...]}

Run:
  pip install fastapi uvicorn httpx
  export AI_PROXY_URL="http://localhost:1106/modelfarm/openai/chat/completions"
  export COMPOSE_MODEL="gpt-4o-mini"
  uvicorn app:app --host 0.0.0.0 --port 8000
"""

import os, json, time, asyncio
from typing import Any, AsyncGenerator, Dict, List, Optional

import httpx
from fastapi import FastAPI, Request
from fastapi.responses import StreamingResponse
from pydantic import BaseModel

# -----------------------------
# Env
# -----------------------------
AI_PROXY_URL   = os.getenv("AI_PROXY_URL", "http://localhost:1106/modelfarm/openai/chat/completions")
COMPOSE_MODEL  = os.getenv("COMPOSE_MODEL", "gpt-4o-mini")

# Retrieval/Prompt caps
K_DEFAULT          = int(os.getenv("K_DEFAULT", "40"))
CHUNK_CHAR_CAP     = int(os.getenv("CHUNK_CHAR_CAP", "650"))   # input cap per chunk to LLM
MAX_SOURCES_TO_LLM = int(os.getenv("MAX_SOURCES_TO_LLM", "10")) # send only top N sources to model

# LLM caps
COMPOSE_MAX_TOKENS = int(os.getenv("COMPOSE_MAX_TOKENS", "1100"))
COMPOSE_TEMP       = float(os.getenv("COMPOSE_TEMPERATURE", "0.2"))

HEARTBEAT_SECONDS  = float(os.getenv("HEARTBEAT_SECONDS", "0.8"))

app = FastAPI(title="AesthetiCite Ask (Single-Call Streaming)", version="1.0")

def sse(event: str, data: Any) -> str:
    return f"event: {event}\ndata: {json.dumps(data, ensure_ascii=False)}\n\n"

    raise NotImplementedError

async def retrieve_chunks_unified(q: str, qvec: List[float], k: int) -> Dict[str, Any]:
    """
    Must return:
      {
        "chunks": [ { "text":..., "title":..., "year":..., "doi":..., "url":... }, ... ],

# -----------------------------
# Deterministic instant skeleton (0ms LLM)
# -----------------------------
def skeleton(sources: List[Dict[str, Any]]) -> str:
    top = sources[:5]
    cites = " ".join([f"[{s.get('sid','S?')}]" for s in top])
    return (
        "Clinical summary (loading grounded answer…)\n"
        "- Evidence retrieved; generating a grounded answer with inline citations.\n"
        "- Claims will only be stated when supported by provided sources.\n\n"
        f"Top sources: {cites}\n\n"
    )

# -----------------------------
# SINGLE-CALL grounded prompt
# -----------------------------
def build_single_call_prompt(question: str, chunks: List[Dict[str, Any]]) -> str:
    compact = []
    for i, c in enumerate(chunks[:MAX_SOURCES_TO_LLM]):
        compact.append({
            "sid": f"S{i+1}",
            "t": (c.get("title") or "")[:180],
            "y": c.get("year"),
            "doi": c.get("doi") or c.get("url"),
            "u": c.get("url"),
            "x": (c.get("text") or "")[:CHUNK_CHAR_CAP],
        })

    # Strong enforcement: no cite => no claim. Also forces "Unknown" if insufficient evidence.
    # This removes the separate PLAN call completely.
    return f"""
You are AesthetiCite (evidence-grounded clinical assistant).

You MUST follow:
1) Every factual claim MUST include at least one inline citation like [S1] or [S2].
2) If the sources do not support a claim, you MUST NOT state it. Say "Evidence insufficient" instead.
3) Do NOT use knowledge outside the provided sources for factual statements. (You may give cautious general guidance only if explicitly marked as "General guidance (not evidence-cited)".)
4) Prefer fewer, stronger claims over many weak ones.

Write the answer in this format:
- Clinical Summary (2–3 lines)
- Key Points (bullets; each bullet must end with citations)
- Limitations / Uncertainty (bullets; cite if source-based)
- Suggested Follow-ups (bullets)

Question: {question}

Sources (JSON):
{json.dumps(compact, ensure_ascii=False)}
""".strip()

# -----------------------------
# Proxy streaming call
# -----------------------------
async def llm_stream(prompt: str) -> AsyncGenerator[str, None]:
    payload = {
        "model": COMPOSE_MODEL,
        "messages": [
            {"role": "system", "content": "Be concise. Do not invent. Follow citation rules exactly."},
            {"role": "user", "content": prompt},
        ],
        "temperature": COMPOSE_TEMP,
        "max_tokens": COMPOSE_MAX_TOKENS,
        "stream": True,
    }

    async with httpx.AsyncClient(timeout=None) as client:
        async with client.stream("POST", AI_PROXY_URL, json=payload) as resp:
            resp.raise_for_status()
            async for line in resp.aiter_lines():
                if not line or not line.startswith("data: "):
                    continue
                chunk = line[6:].strip()
                if chunk == "[DONE]":
                    break
                try:
                    obj = json.loads(chunk)
                    delta = obj["choices"][0].get("delta", {})
                    tok = delta.get("content")
                    if tok:
                        yield tok
                except Exception:
                    continue

# -----------------------------
# API
# -----------------------------
class AskBody(BaseModel):
    q: str
    k: int = K_DEFAULT

@app.post("/ask_single/stream")
async def ask_single_stream(body: AskBody, request: Request):
    q = body.q.strip()
    if not q:
        return StreamingResponse(iter([sse("error", {"message": "Empty query"})]), media_type="text/event-stream")

    async def gen() -> AsyncGenerator[str, None]:
        t0 = time.perf_counter()

        # Retrieval
        yield sse("status", {"phase": "retrieval", "message": "Searching evidence..."})
        t_r0 = time.perf_counter()
        qvec = await embed_query(q)
        r = await retrieve_chunks_unified(q, qvec, body.k)
        chunks = r["chunks"]
        sources = r["sources"]
        t_r1 = time.perf_counter()

        # Stream sources + instant skeleton (OpenEvidence feel)
        yield sse("sources", {"sources": sources, "ms": int((t_r1 - t0) * 1000)})
        yield sse("replace", {"channel": "final", "message": "Answer (streaming)..."})
        yield sse("token", {"t": skeleton(sources), "channel": "final"})

        if await request.is_disconnected():
            return

        # Single streaming call (no plan call)
        prompt = build_single_call_prompt(q, chunks)

        yield sse("status", {"phase": "answer", "message": "Generating grounded answer..."})
        t_c0 = time.perf_counter()
        first_tok_ms: Optional[int] = None

        # Heartbeat task to keep UI alive if first token is delayed by proxy
        stop_hb = False
        async def heartbeat():
            while not stop_hb:
                yield sse("status", {"phase": "answer", "message": "Working…"})
                await asyncio.sleep(HEARTBEAT_SECONDS)

        # Stream answer
        async for tok in llm_stream(prompt):
            if first_tok_ms is None:
                first_tok_ms = int((time.perf_counter() - t_c0) * 1000)
            yield sse("token", {"t": tok, "channel": "final"})
            if await request.is_disconnected():
                return

        t1 = time.perf_counter()
        yield sse("meta", {
            "retrieval_ms": int((t_r1 - t_r0) * 1000),
            "compose_first_token_ms": first_tok_ms,
            "compose_total_ms": int((t1 - t_c0) * 1000),
            "total_ms": int((t1 - t0) * 1000),
            "model": COMPOSE_MODEL,
            "caps": {
                "k": body.k,
                "chunk_char_cap": CHUNK_CHAR_CAP,
                "max_sources_to_llm": MAX_SOURCES_TO_LLM,
                "compose_max_tokens": COMPOSE_MAX_TOKENS,
                "compose_temp": COMPOSE_TEMP,
            }
        })
        yield sse("done", {"ok": True})

    return StreamingResponse(gen(), media_type="text/event-stream")
