function nowIso() {
  return new Date().toISOString();
}

function safeJsonParse<T>(raw: string | null): T | null {
  if (!raw) return null;
  try {
    return JSON.parse(raw) as T;
  } catch {
    return null;
  }
}

function normalizeFormForApi(form: CheckFormState): PreProcedureRequest {
  const cleanedTechnique = (form.technique || "").trim();
  return {
    procedure: form.procedure.trim(),
    region: form.region.trim(),
    product_type: form.product_type.trim(),
    technique: cleanedTechnique || undefined,
    injector_experience_level: form.injector_experience_level,
    patient_factors: { ...(form.patient_factors || {}) },
  };
}

function humanizeSlug(value: string) {
  return value
    .replace(/_/g, " ")
    .replace(/\b\w/g, (c) => c.toUpperCase());
}

function clampScore(score?: number) {
  if (typeof score !== "number" || Number.isNaN(score)) return 0;
  return Math.max(0, Math.min(100, Math.round(score)));
}

function formatDateTime(value?: string) {
  if (!value) return "—";
  const date = new Date(value);
  if (Number.isNaN(date.getTime())) return value;
  return date.toLocaleString();
}

function createDownloadLink(href: string, filename?: string) {
  const a = document.createElement("a");
  a.href = href;
  if (filename) a.download = filename;
  a.rel = "noopener";
  document.body.appendChild(a);
  a.click();
  a.remove();
}

async function downloadFromResponse(path: string, filename: string) {
  createDownloadLink(path, filename);
}

function queueSummary(queue: QueuedCheck[]) {
  const done = queue.filter((i) => i.status === "done" && i.result);
  const pending = queue.filter((i) => i.status === "pending").length;
  const running = queue.filter((i) => i.status === "running").length;
  const errors = queue.filter((i) => i.status === "error").length;
  const avgRisk =
    done.length > 0
      ? Math.round(
          done.reduce((acc, item) => acc + clampScore(item.result?.safety_assessment.overall_risk_score), 0) /
            done.length
        )
      : 0;

  const decisionCounts: Record<Decision, number> = {
    go: 0,
    caution: 0,
    high_risk: 0,
  };

  done.forEach((item) => {
    const decision = item.result?.safety_assessment.decision;
    if (decision) decisionCounts[decision] += 1;
  });

  return {
    total: queue.length,
    pending,
    running,
    errors,
    done: done.length,
    avgRisk,
    decisionCounts,
  };
}

async function runWithConcurrency<T>(
  items: T[],
  worker: (item: T) => Promise<void>,
  concurrency = 3
) {
  const queue = [...items];
  const runners = Array.from({ length: Math.min(concurrency, items.length) }).map(async () => {
    while (queue.length > 0) {
      const item = queue.shift();
      if (!item) return;
      await worker(item);
    }
  });
  await Promise.all(runners);
}

async function apiPost<T>(path: string, body: unknown): Promise<T> {
  const token = getToken();
  const res = await fetch(path, {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      ...(token ? { Authorization: `Bearer ${token}` } : {}),
    },
    body: JSON.stringify(body),
  });

  if (!res.ok) {
    const err = await res.json().catch(() => ({}));
    const detail =
      (err as { detail?: string; message?: string }).detail ||
      (err as { detail?: string; message?: string }).message ||
      `Request failed: ${res.status}`;
    throw new Error(detail);
  }

  return res.json() as Promise<T>;
}

async function apiGet<T>(path: string): Promise<T> {
  const token = getToken();
  const res = await fetch(path, {
    headers: token ? { Authorization: `Bearer ${token}` } : {},
  });

  if (!res.ok) {
    let message = `Request failed: ${res.status}`;
    try {
      const err = (await res.json()) as { detail?: string; message?: string };
      message = err.detail || err.message || message;
    } catch {
      // ignore
    }
    throw new Error(message);
  }

  return res.json() as Promise<T>;
}

function ScoreRing({
  score,
  decision,
  size = 96,
}: {
  score: number;
  decision: Decision;
  size?: number;
}) {
  const safeScore = clampScore(score);
  const cfg = DECISION_CONFIG[decision];
  const r = 36;
  const circumference = 2 * Math.PI * r;
  const stroke = circumference * (safeScore / 100);
  const color =
    decision === "go" ? "#059669" : decision === "caution" ? "#d97706" : "#dc2626";

  return (
    <div className="relative inline-flex items-center justify-center" style={{ width: size, height: size }}>
      <svg width={size} height={size} viewBox="0 0 88 88">
        <circle
          cx={44}
          cy={44}
          r={r}
          fill="none"
          stroke="currentColor"
          strokeWidth={6}
          className="text-muted/30"
        />
        <circle
          cx={44}
          cy={44}
          r={r}
          fill="none"
          stroke={color}
          strokeWidth={6}
          strokeDasharray={`${stroke} ${circumference}`}
          strokeLinecap="round"
          transform="rotate(-90 44 44)"
          style={{ transition: "stroke-dasharray 0.6s ease" }}
        />
      </svg>
      <div className="absolute inset-0 flex flex-col items-center justify-center">
        <span className="text-2xl font-black tabular-nums leading-none" style={{ color }}>
          {safeScore}
        </span>
        <span className="text-[9px] uppercase tracking-wider text-muted-foreground">/100</span>
      </div>
    </div>
  );
}

function SessionQueuePanel({
  queue,
  sessionTitle,
  onSessionTitleChange,
  onRemove,
  onDuplicate,
  onRerun,
  onClear,
  onRunAll,
  onCreateReport,
  onExportSession,
  runningAll,
  creatingReport,
  exportingSession,
  sessionId,
}: {
  queue: QueuedCheck[];
  sessionTitle: string;
  onSessionTitleChange: (v: string) => void;
  onRemove: (id: string) => void;
  onDuplicate: (id: string) => void;
  onRerun: (id: string) => void;
  onClear: () => void;
  onRunAll: () => void;
  onCreateReport: () => void;
  onExportSession: () => void;
  runningAll: boolean;
  creatingReport: boolean;
  exportingSession: boolean;
  sessionId: string | null;
}) {
  const summary = useMemo(() => queueSummary(queue), [queue]);
  const allDone = queue.length > 0 && summary.pending === 0 && summary.running === 0;
  const anyDone = summary.done > 0;

  if (queue.length === 0) {
    return (
      <EmptyState
        icon="☰"
        title="No procedures in the session queue"
        body="Use Add to Session Queue from the safety check form or send a completed result directly into the queue."
      />
    );
  }

  return (
    <div className="space-y-5">
      <SectionTitle
        title="Session Safety Report"
        subtitle="Queue multiple procedures or patients, run safety checks, and export one consolidated PDF."
        action={
          <button
            type="button"
            onClick={onClear}
            className="rounded-lg border border-border px-3 py-1.5 text-xs font-medium text-muted-foreground hover:text-foreground hover:bg-muted/50"
          >
            Clear Queue
          </button>
        }
      />

      <div>
        <Label>Session Title</Label>
        <TextInput
          value={sessionTitle}
          onChange={onSessionTitleChange}
          placeholder="e.g. Friday afternoon injectables session"
        />

        <MiniStat label="Done" value={summary.done} tone="success" />
        <MiniStat label="Pending" value={summary.pending} />
        <MiniStat label="Errors" value={summary.errors} tone={summary.errors ? "danger" : "default"} />
        <MiniStat label="Avg Risk" value={`${summary.avgRisk}/100`} tone="warning" />
        <MiniStat
          label="High Risk"
          value={summary.decisionCounts.high_risk}

          </p>

            <Tag tone="danger">HIGH RISK: {summary.decisionCounts.high_risk}</Tag>
          </div>
        </div>
      )}

      <div className="space-y-3">
        {queue.map((item) => (
          <QueueItemCard
            key={item.id}
            item={item}
            onRemove={onRemove}
            onDuplicate={onDuplicate}
            onRerun={onRerun}
          />
        ))}
      </div>

      <div className="grid grid-cols-1 gap-3 sm:grid-cols-3">
        <button
          type="button"

          onClick={onCreateReport}

            <span>⊕</span>
          )}
          Create Session Report

        <button
          type="button"

          disabled={!sessionId || exportingSession}
          className="inline-flex items-center justify-center gap-2 rounded-xl bg-emerald-600 px-5 py-3 text-sm font-bold text-white hover:bg-emerald-700 disabled:opacity-40"
        >
          {exportingSession ? (
            <span className="h-4 w-4 rounded-full border-2 border-white/30 border-t-white animate-spin" />
          ) : (
            <span>↓</span>
          )}
          Export Session PDF
        </button>
      </div>

      {allDone && !sessionId ? (
        <p className="text-xs text-muted-foreground">
          All queued checks are complete. Create the session report, then export the consolidated PDF.
        </p>
      ) : null}

      {sessionId ? (
        <div className="rounded-xl border border-emerald-500/20 bg-emerald-500/5 p-3">
          <p className="text-xs font-semibold text-emerald-700 dark:text-emerald-300">

          </p>
        </div>
      ) : null}

  );

/* ──────────────────────────────────────────────────────────────────────────────

export default function SafetyWorkspacePage() {
  const hydrated = useMemo(() => {
    const persisted = safeJsonParse<PersistedWorkspaceState>(localStorage.getItem(STORAGE_KEY));
    return persisted;
  }, []);

  const [activeTab, setActiveTab] = useState<TabKey>(hydrated?.activeTab ?? "check");
  const [form, setForm] = useState<CheckFormState>(hydrated?.lastSingleForm ?? BLANK_FORM);

  const [loading, setLoading] = useState(false);
  const [result, setResult] = useState<PreProcedureResponse | null>(null);
  const [lastRequest, setLastRequest] = useState<PreProcedureRequest | null>(null);
  const [error, setError] = useState<string | null>(null);
  const [exportingSinglePdf, setExportingSinglePdf] = useState(false);

  const [queue, setQueue] = useState<QueuedCheck[]>(hydrated?.queue ?? []);
  const [runningAll, setRunningAll] = useState(false);
  const [creatingReport, setCreatingReport] = useState(false);
  const [exportingSession, setExportingSession] = useState(false);
  const [sessionId, setSessionId] = useState<string | null>(null);
  const [sessionTitle, setSessionTitle] = useState(hydrated?.sessionTitle ?? "Session Safety Report");
  const [globalMessage, setGlobalMessage] = useState<string | null>(null);

  const resultsRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    const payload: PersistedWorkspaceState = {
      queue,
      sessionTitle,
      activeTab,
      lastSingleForm: form,
    };
    localStorage.setItem(STORAGE_KEY, JSON.stringify(payload));
  }, [queue, sessionTitle, activeTab, form]);

  const runSingleCheck = useCallback(async (request: PreProcedureRequest) => {
    return apiPost<PreProcedureResponse>("/api/safety/preprocedure-check", request);
  }, []);

  const handleRunSingle = useCallback(async () => {
    const payload = normalizeFormForApi(form);
    setLoading(true);
    setError(null);
    setResult(null);
    setGlobalMessage(null);
    setLastRequest(payload);

    try {
      const res = await runSingleCheck(payload);
      setResult(res);
      setTimeout(() => {
        resultsRef.current?.scrollIntoView({ behavior: "smooth", block: "start" });
      }, 100);
    } catch (e) {
      setError(e instanceof Error ? e.message : "Safety check failed");
    } finally {
      setLoading(false);
    }
  }, [form, runSingleCheck]);

  const buildQueueItem = useCallback((state: CheckFormState): QueuedCheck => {
    return {
      id: crypto.randomUUID(),
      patient_label: state.patient_label.trim() || `Patient ${queue.length + 1}`,
      request: normalizeFormForApi(state),
      status: "pending",
      created_at: nowIso(),
    };
  }, [queue.length]);

  const handleAddToQueue = useCallback(() => {
    const item = buildQueueItem(form);
    setQueue((prev) => [...prev, item]);
    setActiveTab("session");
    setGlobalMessage(`Added ${item.patient_label} to the session queue.`);
  }, [buildQueueItem, form]);

  const handleSendCurrentResultToQueue = useCallback(() => {
    if (!lastRequest) return;
    const item: QueuedCheck = {
      id: crypto.randomUUID(),
      patient_label: form.patient_label.trim() || `Patient ${queue.length + 1}`,
      request: { ...lastRequest },
      result: result || undefined,
      status: result ? "done" : "pending",
      created_at: nowIso(),
    };

    setQueue((prev) => [...prev, item]);
    setActiveTab("session");
    setGlobalMessage(`Added ${item.patient_label} to the session queue.`);
  }, [form.patient_label, lastRequest, queue.length, result]);

  const handleRemoveFromQueue = useCallback((id: string) => {
    setQueue((prev) => prev.filter((item) => item.id !== id));
    if (sessionId) setSessionId(null);
  }, [sessionId]);

  const handleDuplicateQueueItem = useCallback((id: string) => {
    setQueue((prev) => {
      const source = prev.find((item) => item.id === id);
      if (!source) return prev;
      const copy: QueuedCheck = {
        ...source,
        id: crypto.randomUUID(),
        patient_label: `${source.patient_label} Copy`,
        status: "pending",
        result: undefined,
        error: undefined,
        created_at: nowIso(),
      };
      return [...prev, copy];
    });
    if (sessionId) setSessionId(null);
  }, [sessionId]);

  const executeQueueItem = useCallback(async (id: string) => {
    const current = queue.find((item) => item.id === id);
    if (!current) return;

    setQueue((prev) =>
      prev.map((item) =>
        item.id === id
          ? { ...item, status: "running", error: undefined }
          : item
      )
    );

    try {
      const res = await apiPost<PreProcedureResponse>("/api/safety/preprocedure-check", current.request);
      setQueue((prev) =>
        prev.map((item) =>
          item.id === id
            ? { ...item, status: "done", result: res, error: undefined }
            : item
        )
      );
    } catch (e) {
      setQueue((prev) =>
        prev.map((item) =>
          item.id === id
            ? {
                ...item,
                status: "error",
                error: e instanceof Error ? e.message : "Failed",
              }
            : item
        )
      );
    }
  }, [queue]);

  const handleRerunQueueItem = useCallback(async (id: string) => {
    setSessionId(null);
    await executeQueueItem(id);
  }, [executeQueueItem]);

  const handleRunAll = useCallback(async () => {
    const pendingItems = queue.filter((item) => item.status === "pending" || item.status === "error");
    if (pendingItems.length === 0) return;

    setRunningAll(true);
    setSessionId(null);
    setGlobalMessage(null);

    try {
      await runWithConcurrency(
        pendingItems.map((i) => i.id),
        async (id) => {
          await executeQueueItem(id);
        },
        3
      );
      setGlobalMessage("All available safety checks were processed.");
    } finally {
      setRunningAll(false);
    }
  }, [executeQueueItem, queue]);

  const handleClearQueue = useCallback(() => {
    setQueue([]);
    setSessionId(null);
    setGlobalMessage("Session queue cleared.");
  }, []);

  const handleCreateSessionReport = useCallback(async () => {
    const completed = queue.filter((item) => item.status === "done" && item.result);
    if (completed.length === 0) return;

    setCreatingReport(true);
    setGlobalMessage(null);

    try {
      const created = await apiPost<SessionReportCreateResponse>("/api/growth/session-reports", {
        title: sessionTitle.trim() || "Session Safety Report",
        report_date: new Date().toISOString().slice(0, 10),
        clinic_id: "demo_clinic",
        clinician_id: "demo_clinician",
      });

      for (const item of completed) {
        await apiPost(`/api/growth/session-reports/${created.id}/items`, {
          patient_label: item.patient_label,
          procedure: item.request.procedure,
          region: item.request.region,
          product_type: item.request.product_type,
          technique: item.request.technique,
          injector_experience_level: item.request.injector_experience_level,
          engine_response_json: item.result,
        });
      }

      setSessionId(created.id);
      setGlobalMessage("Session report created successfully.");
    } catch (e) {
      setError(e instanceof Error ? e.message : "Failed to create session report");
    } finally {
      setCreatingReport(false);
    }
  }, [queue, sessionTitle]);

  const handleExportSession = useCallback(async () => {
    if (!sessionId) return;

    setExportingSession(true);
    setGlobalMessage(null);

    try {
      const data = await apiPost<SessionExportResponse>(
        `/api/growth/session-reports/${sessionId}/export-pdf`,
        {}
      );
      await downloadFromResponse(`/exports/${data.filename}`, data.filename);
      setGlobalMessage("Session PDF exported.");
    } catch (e) {
      setError(e instanceof Error ? e.message : "Session export failed");
    } finally {
      setExportingSession(false);
    }
  }, [sessionId]);

  const handleExportSinglePdf = useCallback(async () => {
    if (!lastRequest) return;

    setExportingSinglePdf(true);
    setGlobalMessage(null);

    try {
      const data = await apiPost<SingleExportResponse>(
        "/api/safety/preprocedure-check/export-pdf",
        lastRequest
      );
      await downloadFromResponse(`/exports/${data.filename}`, data.filename);
      setGlobalMessage("Safety check PDF exported.");
    } catch (e) {
      setError(e instanceof Error ? e.message : "PDF export failed");
    } finally {
      setExportingSinglePdf(false);
    }
  }, [lastRequest]);

  const handleRefreshSessionReport = useCallback(async () => {
    if (!sessionId) return;
    try {
      await apiGet(`/api/growth/session-reports/${sessionId}`);
      setGlobalMessage("Session report is available.");
    } catch (e) {
      setError(e instanceof Error ? e.message : "Failed to load session report");
    }
  }, [sessionId]);

  const queueStats = useMemo(() => queueSummary(queue), [queue]);

  return (
    <div className="min-h-screen bg-background">
      <div className="sticky top-0 z-20 border-b border-border bg-background/90 backdrop-blur">
        <div className="mx-auto flex max-w-7xl items-center justify-between gap-4 px-4 py-3">
          <div>
            <h1 className="flex items-center gap-2 text-base font-black tracking-tight">
              <span className="text-primary">⬡</span>
              AesthetiCite
              <span className="font-medium text-muted-foreground">/ Safety Workspace</span>
            </h1>
            <p className="mt-0.5 text-[11px] text-muted-foreground">
              Clinical safety decision support for aesthetic medicine
            </p>
          </div>

          <div className="flex items-center gap-2">
            <a
              href="/ask"
              className="rounded-lg border border-border px-3 py-1.5 text-xs font-medium text-muted-foreground transition-colors hover:bg-muted/50 hover:text-foreground"
            >
              Back to Evidence Search
            </a>
          </div>
        </div>
      </div>

      <div className="mx-auto max-w-7xl px-4 py-6 lg:py-8">
        {globalMessage ? (
          <div className="mb-4 rounded-xl border border-emerald-500/20 bg-emerald-500/5 px-4 py-3 text-sm text-emerald-700 dark:text-emerald-300">
            {globalMessage}
          </div>
        ) : null}

        {error ? (
          <div className="mb-4 rounded-xl border border-red-500/20 bg-red-500/5 px-4 py-3 text-sm text-red-700 dark:text-red-300">
            {error}
          </div>
        ) : null}

        <div className="grid grid-cols-1 gap-6 xl:grid-cols-[520px_minmax(0,1fr)]">
          <div className="space-y-4">
            <div className="overflow-hidden rounded-3xl border border-border bg-card shadow-sm">
              <div className="flex border-b border-border">
                {(["check", "session"] as const).map((tab) => {
                  const active = activeTab === tab;
                  return (
                    <button
                      key={tab}
                      type="button"
                      onClick={() => setActiveTab(tab)}
                      className={`relative flex-1 px-4 py-3 text-sm font-bold transition-colors ${
                        active ? "text-foreground" : "text-muted-foreground hover:text-foreground"
                      }`}
                    >
                      {tab === "check" ? (
                        "Safety Check"
                      ) : (
                        <span className="inline-flex items-center gap-2">
                          Session Queue
                          {queue.length > 0 ? (
                            <span className="inline-flex h-5 min-w-5 items-center justify-center rounded-full bg-primary px-1.5 text-[10px] font-black text-primary-foreground">
                              {queue.length}
                            </span>
                          ) : null}
                        </span>
                      )}
                      {active ? (
                        <span className="absolute inset-x-0 bottom-0 h-0.5 rounded-full bg-primary" />
                      ) : null}
                    </button>
                  );
                })}
              </div>

              <div className="p-5">
                {activeTab === "check" ? (
                  <CheckForm
                    value={form}
                    onChange={setForm}
                    onRunSingle={handleRunSingle}
                    onAddToQueue={handleAddToQueue}
                    onReset={() => {
                      setForm(BLANK_FORM);
                      setError(null);
                      setResult(null);
                      setLastRequest(null);
                    }}
                    loading={loading}
                    onApplyPreset={setForm}
                  />
                ) : (
                  <SessionQueuePanel
                    queue={queue}
                    sessionTitle={sessionTitle}
                    onSessionTitleChange={setSessionTitle}
                    onRemove={handleRemoveFromQueue}
                    onDuplicate={handleDuplicateQueueItem}
                    onRerun={handleRerunQueueItem}
                    onClear={handleClearQueue}
                    onRunAll={handleRunAll}
                    onCreateReport={handleCreateSessionReport}
                    onExportSession={handleExportSession}
                    runningAll={runningAll}
                    creatingReport={creatingReport}
                    exportingSession={exportingSession}
                    sessionId={sessionId}
                  />
                )}
              </div>
            </div>

            <div className="grid grid-cols-2 gap-3 lg:grid-cols-4">
              <MiniStat label="Queued" value={queueStats.total} />
              <MiniStat label="Done" value={queueStats.done} tone="success" />
              <MiniStat label="Avg Risk" value={`${queueStats.avgRisk}/100`} tone="warning" />
              <MiniStat
                label="High Risk"
                value={queueStats.decisionCounts.high_risk}
                tone={queueStats.decisionCounts.high_risk > 0 ? "danger" : "default"}
              />
            </div>

            <div className="rounded-2xl border border-border bg-card p-4">
              <p className="text-[11px] font-semibold uppercase tracking-wider text-muted-foreground">
                Product framing
              </p>
              <div className="mt-3 space-y-2 text-sm text-muted-foreground">
                <p>• Pre-Procedure Safety Check</p>
                <p>• Session Safety Report</p>
                <p>• Clinical safety decision support for aesthetic injectables</p>
              </div>
            </div>

            <p className="text-center text-[10px] text-muted-foreground/50">
              AesthetiCite Safety Workspace · Clinic demo mode · Clinical decision support only
            </p>
          </div>

          <div ref={resultsRef} className="space-y-4">
            {loading ? (
              <div className="rounded-3xl border border-border bg-card p-8">
                <div className="flex flex-col items-center gap-4 text-center">
                  <div className="relative h-16 w-16">
                    <div className="absolute inset-0 rounded-full border-4 border-primary/20 animate-ping" />
                    <div className="absolute inset-0 rounded-full border-4 border-primary/40 border-t-primary animate-spin" />
                  </div>
                  <div>
                    <p className="text-sm font-bold">Running safety analysis…</p>
                    <p className="mt-1 text-xs text-muted-foreground">
                      Evaluating procedure, anatomy, patient factors, and mitigation steps.
                    </p>
                  </div>
                </div>
              </div>
            ) : null}

            {!loading && result ? (
              <ResultPanel
                result={result}
                request={lastRequest}
                exportingPdf={exportingSinglePdf}
                onExportPdf={handleExportSinglePdf}
                onAddCurrentToQueue={handleSendCurrentResultToQueue}
              />
            ) : null}

            {!loading && !result && !error ? (
              <EmptyState
                title="No result yet"
                body="Run a pre-procedure safety check to display the decision badge, risk score, top complication risks, danger zones, mitigation steps, and evidence."
              />
            ) : null}

            {sessionId ? (
              <div className="rounded-2xl border border-border bg-card p-4">
                <SectionTitle
                  title="Session report ready"
                  subtitle="Your consolidated session report has been created and is ready for export."
                  action={
                    <button
                      type="button"
                      onClick={handleRefreshSessionReport}
                      className="rounded-lg border border-border px-3 py-1.5 text-xs font-medium hover:bg-muted/50"
                    >
                      Refresh
                    </button>
                  }
                />
                <p className="mt-3 text-sm text-muted-foreground">
                  Report ID: <code className="font-mono">{sessionId}</code>
                </p>
              </div>
            ) : null}
          </div>
        </div>
      </div>
    </div>
  );
}
