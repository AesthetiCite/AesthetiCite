# -----------------------------
# Minimal JWT (HS256) + password hashing
# -----------------------------
def _b64url(data: bytes) -> str:
    return base64.urlsafe_b64encode(data).rstrip(b"=").decode("utf-8")

def _b64url_decode(s: str) -> bytes:
    pad = "=" * (-len(s) % 4)
    return base64.urlsafe_b64decode((s + pad).encode("utf-8"))

def jwt_sign(payload: Dict[str, Any], secret: str) -> str:
    header = {"alg": "HS256", "typ": "JWT"}
    h = _b64url(json.dumps(header, separators=(",", ":")).encode("utf-8"))
    p = _b64url(json.dumps(payload, separators=(",", ":")).encode("utf-8"))
    msg = f"{h}.{p}".encode("utf-8")
    sig = hmac.new(secret.encode("utf-8"), msg, hashlib.sha256).digest()
    return f"{h}.{p}.{_b64url(sig)}"

def jwt_verify(token: str, secret: str) -> Dict[str, Any]:
    try:
        h, p, s = token.split(".")
        msg = f"{h}.{p}".encode("utf-8")
        sig = _b64url_decode(s)
        exp_sig = hmac.new(secret.encode("utf-8"), msg, hashlib.sha256).digest()
        if not hmac.compare_digest(sig, exp_sig):
            raise ValueError("bad signature")
        payload = json.loads(_b64url_decode(p))
        if "exp" in payload and int(payload["exp"]) < int(time.time()):
            raise ValueError("expired")
        return payload
    except Exception:
        raise HTTPException(status_code=401, detail="Invalid token")

def hash_password(password: str) -> str:
    salt = secrets.token_bytes(16)
    dk = hashlib.pbkdf2_hmac("sha256", password.encode("utf-8"), salt, 120_000)
    return f"pbkdf2_sha256${_b64url(salt)}${_b64url(dk)}"

def verify_password(password: str, stored: str) -> bool:
    try:
        algo, salt_b64, dk_b64 = stored.split("$", 2)
        if algo != "pbkdf2_sha256":
            return False
        salt = _b64url_decode(salt_b64)
        dk = _b64url_decode(dk_b64)
        test = hashlib.pbkdf2_hmac("sha256", password.encode("utf-8"), salt, 120_000)
        return hmac.compare_digest(dk, test)
    except Exception:
        return False

async def get_user(authorization: Optional[str] = Header(None)) -> Dict[str, Any]:
    if not authorization or not authorization.lower().startswith("bearer "):
        raise HTTPException(status_code=401, detail="Missing bearer token")
    token = authorization.split(" ", 1)[1].strip()
    payload = jwt_verify(token, JWT_SECRET)
    uid = payload.get("sub")
    if not uid:
        raise HTTPException(status_code=401, detail="Invalid token payload")

    assert POOL is not None
    async with POOL.acquire() as con:
        row = await con.fetchrow("SELECT id, email FROM users WHERE id=$1", uid)
    if not row:
        raise HTTPException(status_code=401, detail="User not found")
    return {"id": row["id"], "email": row["email"]}

def infer_evidence_type(document_type: Optional[str], journal_or_org: Optional[str]) -> str:
    t = (document_type or "").lower()
    j = (journal_or_org or "").lower()
    if "guideline" in t or "consensus" in t or "recommendation" in t:
        return "Guideline/Consensus"
    if "systematic" in t or "meta" in t:
        return "Systematic Review"
    if "random" in t or "rct" in t or "trial" in t:
        return "Randomized Trial"
    if "cohort" in t or "case-control" in t or "observational" in t:
        return "Observational Study"
    if "case report" in t or "case series" in t:
        return "Case Report/Series"
    if "review" in t:
        return "Narrative Review"
    if any(x in j for x in ["nejm", "jama", "lancet", "bmj"]):
        return "Journal Article"
    return "Other"

def evidence_rank(evidence_type: str) -> int:
    order = {
        "Guideline/Consensus": 1,
        "Systematic Review": 2,
        "Randomized Trial": 3,
        "Observational Study": 4,
        "Case Report/Series": 5,
        "Narrative Review": 6,
        "Journal Article": 7,
        "Other": 8,
    }
    return order.get(evidence_type, 9)

def compute_aci_lightweight(sources: List[Dict[str, Any]]) -> Dict[str, Any]:
    typed = []
    for s in sources:
        et = infer_evidence_type(s.get("document_type"), s.get("organization_or_journal") or s.get("journal"))
        typed.append((evidence_rank(et), et))
    typed.sort(key=lambda x: x[0])
    highest = typed[0][1] if typed else "N/A"
    supporting = len(sources)
    base = 6.0
    if highest in ("Guideline/Consensus", "Systematic Review"):
        base += 2.0
    elif highest == "Randomized Trial":
        base += 1.2
    base += min(2.0, supporting / 20.0)
    conf = max(0.0, min(10.0, base))
    gaps = []
    if supporting < 5:
        gaps.append("Limited number of directly relevant sources retrieved")
    return {
        "overall_confidence_0_10": conf,
        "highest_level": highest,
        "supporting_count": supporting,
        "gaps": gaps,
    }

def build_authority_header(sources: List[Dict[str, Any]], aci: Dict[str, Any]) -> str:
    conf = aci.get("overall_confidence_0_10")
    highest = aci.get("highest_level", "N/A")
    n_support = aci.get("supporting_count", len(sources))
    gaps = aci.get("gaps", []) or []
    conf_str = f"{conf:.1f}/10" if isinstance(conf, (int, float)) else "N/A"

    mix = []
    for s in sources[:10]:
        et = infer_evidence_type(s.get("document_type"), s.get("organization_or_journal") or s.get("journal"))
        if et not in mix:
            mix.append(et)
        if len(mix) >= 3:
            break

    gaps_text = ""
    if gaps:
        gaps_text = "\n- Evidence gaps: " + "; ".join(gaps[:3])

    safety = (
        "Safety considerations (aesthetic practice):\n"
        "- Where vascular risk exists, use conservative technique and have an occlusion response protocol.\n"
        "- Document contraindications and informed consent; reassess after conservative dosing/volumes.\n"
    )

    return (
        "Evidence Strength Summary\n"
        f"- Overall confidence: {conf_str}\n"
        f"- Highest evidence level found: {highest}\n"
        f"- Supporting sources used: {n_support}\n"
        f"- Evidence mix: {', '.join(mix) if mix else 'N/A'}"
        f"{gaps_text}\n\n"
        f"{safety}\n"
        "What is well-supported vs uncertain is stated explicitly below.\n\n"
    )

def skeleton(sources: List[Dict[str, Any]]) -> str:
    top = sources[:5]
    cites = " ".join([f"[{s.get('sid','S?')}]" for s in top])
    return (
        "Clinical summary (loading grounded answer…)\n"
        "- Evidence retrieved; generating a grounded answer with inline citations.\n"
        "- Claims will only be stated when supported by provided sources.\n\n"
        f"Top sources: {cites}\n\n"
    )

DOI_RE = re.compile(r"10\.\d{4,9}/[-._;()/:A-Z0-9]+", re.I)

def normalize_doi(s: Optional[str]) -> Optional[str]:
    if not s:
        return None
    s = s.strip()
    s = s.replace("https://doi.org/", "").replace("http://doi.org/", "")
    s = s.replace("doi:", "").strip()
    m = DOI_RE.search(s)
    return m.group(0) if m else None

def doi_url(doi: str) -> str:
    return f"https://doi.org/{doi}"

def pubmed_url(pmid: str) -> str:
    return f"https://pubmed.ncbi.nlm.nih.gov/{pmid}/"

async def upsert_doc_meta(source_id: str, doi: Optional[str], pmid: Optional[str], url: Optional[str],
                          title: Optional[str], journal: Optional[str], year: Optional[int]) -> None:
    assert POOL is not None
    async with POOL.acquire() as con:
        await con.execute(
            """
            INSERT INTO documents_meta(source_id, doi, pmid, url, title, journal, year, updated_at)
            VALUES($1,$2,$3,$4,$5,$6,$7, now())
            ON CONFLICT(source_id) DO UPDATE SET
              doi=COALESCE(EXCLUDED.doi, documents_meta.doi),
              pmid=COALESCE(EXCLUDED.pmid, documents_meta.pmid),
              url=COALESCE(EXCLUDED.url, documents_meta.url),
              title=COALESCE(EXCLUDED.title, documents_meta.title),
              journal=COALESCE(EXCLUDED.journal, documents_meta.journal),
              year=COALESCE(EXCLUDED.year, documents_meta.year),
              updated_at=now()
            """,
            source_id, doi, pmid, url, title, journal, year
        )

async def fetch_doc_meta(source_id: str) -> Optional[Dict[str, Any]]:
    assert POOL is not None
    async with POOL.acquire() as con:
        row = await con.fetchrow("SELECT source_id, doi, pmid, url, title, journal, year FROM documents_meta WHERE source_id=$1", source_id)
    return dict(row) if row else None

# -----------------------------
# Ingestion: /ingest/doi (Crossref + PubMed best-effort)
# -----------------------------
CROSSREF_API = "https://api.crossref.org/works/"
PUBMED_SEARCH = "https://eutils.ncbi.nlm.nih.gov/entrez/eutils/esearch.fcgi"
PUBMED_FETCH  = "https://eutils.ncbi.nlm.nih.gov/entrez/eutils/efetch.fcgi"

async def crossref_lookup(doi: str) -> Dict[str, Any]:
    async with httpx.AsyncClient(timeout=30) as client:
        r = await client.get(CROSSREF_API + doi, headers={"User-Agent": "AesthetiCite/1.0"})
        r.raise_for_status()
        return r.json().get("message", {})

async def pubmed_find_pmid(doi: str) -> Optional[str]:
    params = {"db": "pubmed", "term": f"{doi}[AID]", "retmode": "json", "retmax": "1"}
    async with httpx.AsyncClient(timeout=30) as client:
        r = await client.get(PUBMED_SEARCH, params=params)
        if r.status_code != 200:
            return None
        ids = r.json().get("esearchresult", {}).get("idlist", [])
        return ids[0] if ids else None

class IngestDOI(BaseModel):
    source_id: str
    doi: str

@app.post("/ingest/doi")
async def ingest_doi(body: IngestDOI, user=Depends(get_user)):
    doi = normalize_doi(body.doi)
    if not doi:
        raise HTTPException(status_code=400, detail="Invalid DOI")

    cr = await crossref_lookup(doi)
    title = (cr.get("title") or [None])[0]
    journal = (cr.get("container-title") or [None])[0]
    year = None
    if cr.get("issued", {}).get("date-parts"):
        year = cr["issued"]["date-parts"][0][0]
    url = cr.get("URL") or doi_url(doi)

    pmid = await pubmed_find_pmid(doi)

    await upsert_doc_meta(
        source_id=body.source_id,
        doi=doi,
        pmid=pmid,
        url=url,
        title=title,
        journal=journal,
        year=year,
    )
    return {"ok": True, "source_id": body.source_id, "doi": doi, "pmid": pmid, "url": url}

@app.post("/auth/signup")
async def signup(body: SignupBody):
    if len(body.password) < 8:
        raise HTTPException(status_code=400, detail="Password too short (min 8 chars)")
    uid = "u_" + secrets.token_hex(10)
    pw = hash_password(body.password)

    assert POOL is not None
    async with POOL.acquire() as con:
        try:
            await con.execute("INSERT INTO users(id, email, pw_hash) VALUES($1,$2,$3)", uid, body.email.lower(), pw)
        except asyncpg.UniqueViolationError:
            raise HTTPException(status_code=409, detail="Email already exists")

    token = jwt_sign({"sub": uid, "email": body.email.lower(), "exp": int(time.time()) + 60 * 60 * 24 * 30}, JWT_SECRET)
    return {"ok": True, "token": token, "user": {"id": uid, "email": body.email.lower()}}

@app.post("/auth/login")
async def login(body: LoginBody):
    assert POOL is not None
    async with POOL.acquire() as con:
        row = await con.fetchrow("SELECT id, email, pw_hash FROM users WHERE email=$1", body.email.lower())
    if not row or not verify_password(body.password, row["pw_hash"]):
        raise HTTPException(status_code=401, detail="Invalid credentials")

    token = jwt_sign({"sub": row["id"], "email": row["email"], "exp": int(time.time()) + 60 * 60 * 24 * 30}, JWT_SECRET)
    return {"ok": True, "token": token, "user": {"id": row["id"], "email": row["email"]}}

async def fetch_recent_messages(conversation_id: str, limit: int = MEMORY_MAX_TURNS) -> List[Dict[str, Any]]:
    assert POOL is not None
    async with POOL.acquire() as con:
        rows = await con.fetch(
            "SELECT role, content, created_at FROM messages WHERE conversation_id=$1 ORDER BY created_at DESC LIMIT $2",
            conversation_id, limit
        )
    return [{"role": r["role"], "content": r["content"]} for r in reversed(rows)]

def compress_context(messages: List[Dict[str, Any]]) -> str:
    parts = []
    for m in messages[-MEMORY_MAX_TURNS:]:
        role = "User" if m["role"] == "user" else "Assistant"
        txt = (m["content"] or "").strip()
        if len(txt) > 700:
            txt = txt[:700] + "…"
        parts.append(f"{role}: {txt}")
    ctx = "\n".join(parts)
    if len(ctx) > MEMORY_MAX_CHARS:
        ctx = ctx[-MEMORY_MAX_CHARS:]
    return ctx

def build_single_call_prompt(question: str, conversation_context: str, chunks: List[Dict[str, Any]]) -> str:
    compact = []
    for c in chunks[:MAX_SOURCES_TO_LLM]:
        compact.append({
            "sid": c.get("sid"),
            "t": (c.get("title") or "")[:160],
            "y": c.get("year"),
            "id": c.get("doi") or c.get("url"),
            "x": (c.get("text") or "")[:CHUNK_CHAR_CAP],
        })

    return f"""
You are AesthetiCite (evidence-grounded clinical assistant for aesthetic medicine).

Conversation context:
{conversation_context}

Hard rules:
1) EVERY factual claim MUST include inline citations like [S1][S3]. You may cite multiple sources per claim.
2) If sources do not support a claim, DO NOT state it. Write "Evidence insufficient".
3) Do NOT use outside knowledge for factual claims.
4) Prefer fewer, stronger, cited points over many weak points.
5) Make uncertainty explicit.

Write exactly in this structure:
Clinical Summary (2–3 lines)

Key Evidence-Based Points
- Bullet (end with citations like [S1][S3])

Safety Considerations (if applicable)

Limitations / Uncertainty

Suggested Follow-up Questions
- Question 1
- Question 2
- Question 3

User question: {question}

Sources (JSON):
{json.dumps(compact, ensure_ascii=False)}
""".strip()

async def llm_stream(prompt: str) -> AsyncGenerator[str, None]:
    payload = {
        "model": COMPOSE_MODEL,
        "messages": [
            {"role": "system", "content": "Be concise. Follow citation rules exactly. No invented facts."},
            {"role": "user", "content": prompt},
        ],
        "temperature": COMPOSE_TEMP,
        "max_tokens": COMPOSE_MAX_TOKENS,
        "stream": True,
    }
    async with httpx.AsyncClient(timeout=None) as client:
        async with client.stream("POST", AI_PROXY_URL, json=payload) as resp:
            resp.raise_for_status()
            async for line in resp.aiter_lines():
                if not line or not line.startswith("data: "):
                    continue
                chunk = line[6:].strip()
                if chunk == "[DONE]":
                    break
                try:
                    obj = json.loads(chunk)
                    delta = obj["choices"][0].get("delta", {})
                    tok = delta.get("content")
                    if tok:
                        yield tok
                except Exception:
                    continue

    conversation_id: str
    k: int = K_DEFAULT

def _ensure_source_url(s: Dict[str, Any]) -> Dict[str, Any]:
    # Provide a robust clickable URL for the frontend.
    url = s.get("url")
    doi = normalize_doi(s.get("doi") or "")
    pmid = (s.get("pmid") or "").strip() if s.get("pmid") else None

    if not url:
        if doi:
            url = doi_url(doi)
        elif pmid:
            url = pubmed_url(pmid)

    s2 = dict(s)
    if doi:
        s2["doi_norm"] = doi
        s2["doi_url"] = doi_url(doi)
    if pmid:
        s2["pubmed_url"] = pubmed_url(pmid)
    s2["url"] = url
    return s2

@app.post("/ask/stream")
async def ask_stream(body: AskBody, request: Request, user=Depends(get_user)):
    q = body.q.strip()
    if not q:
        return StreamingResponse(iter([sse("error", {"message": "Empty query"})]),
                                 media_type="text/event-stream")

    cid = body.conversation_id.strip()
    if not cid:
        raise HTTPException(status_code=400, detail="conversation_id required")

    # Ensure this conversation belongs to this user
    assert POOL is not None
    async with POOL.acquire() as con:
        own = await con.fetchval("SELECT 1 FROM conversations WHERE id=$1 AND user_id=$2", cid, user["id"])
        if not own:
            raise HTTPException(status_code=404, detail="Conversation not found")

    async def gen() -> AsyncGenerator[str, None]:
        t0 = time.perf_counter()
        yield sse("status", {"phase": "retrieval", "message": "Searching evidence..."})

        # Store user message
        await add_message(cid, "user", q)

        # Context
        recent = await fetch_recent_messages(cid, limit=MEMORY_MAX_TURNS)
        ctx = compress_context(recent)

        # Retrieval
        t_r0 = time.perf_counter()
        qvec = await embed_query(q)
        r = await retrieve_chunks_unified(q, qvec, body.k)
        chunks = r["chunks"]
        sources = r["sources"]
        t_r1 = time.perf_counter()

        # Enrich sources with evidence tags + robust URLs.
        enriched = []
        for s in sources:
            et = infer_evidence_type(s.get("document_type"), s.get("organization_or_journal") or s.get("journal"))
            s2 = dict(s)
            s2["evidence_type"] = et
            s2["evidence_rank"] = evidence_rank(et)

            # If url missing but we have a stable source_id, try meta table
            if not s2.get("url") and s2.get("source_id"):
                meta = await fetch_doc_meta(s2["source_id"])
                if meta:
                    s2["doi"] = s2.get("doi") or meta.get("doi")
                    s2["pmid"] = s2.get("pmid") or meta.get("pmid")
                    s2["url"] = s2.get("url") or meta.get("url")

            s2 = _ensure_source_url(s2)
            enriched.append(s2)

        # Authority header + instant skeleton
        aci = compute_aci_lightweight(enriched)
        header = build_authority_header(enriched, aci)

        yield sse("sources", {"sources": enriched, "ms": int((t_r1 - t0) * 1000)})
        yield sse("replace", {"channel": "final", "message": "Answer (streaming)…"})
        yield sse("token", {"t": header, "channel": "final"})
        yield sse("token", {"t": skeleton(enriched), "channel": "final"})

        if await request.is_disconnected():
            return

        # Single-call LLM stream (includes contextual follow-ups)
        prompt = build_single_call_prompt(q, ctx, chunks)

        yield sse("status", {"phase": "answer", "message": "Generating grounded answer..."})
        t_c0 = time.perf_counter()
        first_tok_ms: Optional[int] = None
        buf: List[str] = []

        async for tok in llm_stream(prompt):
            if first_tok_ms is None:
                first_tok_ms = int((time.perf_counter() - t_c0) * 1000)
            buf.append(tok)
            yield sse("token", {"t": tok, "channel": "final"})
            if await request.is_disconnected():
                return

        t1 = time.perf_counter()
        answer_text = "".join(buf).strip()
        await add_message(cid, "assistant", answer_text)

        yield sse("meta", {
            "retrieval_ms": int((t_r1 - t_r0) * 1000),
            "compose_first_token_ms": first_tok_ms,
            "compose_total_ms": int((t1 - t_c0) * 1000),
            "total_ms": int((t1 - t0) * 1000),
            "model": COMPOSE_MODEL,
            "caps": {
                "k": body.k,
                "chunk_char_cap": CHUNK_CHAR_CAP,
                "max_sources_to_llm": MAX_SOURCES_TO_LLM,
                "compose_max_tokens": COMPOSE_MAX_TOKENS,
                "compose_temp": COMPOSE_TEMP,
                "memory_max_turns": MEMORY_MAX_TURNS,
                "memory_max_chars": MEMORY_MAX_CHARS,
            }
        })
        yield sse("done", {"ok": True})

    return StreamingResponse(gen(), media_type="text/event-stream")
