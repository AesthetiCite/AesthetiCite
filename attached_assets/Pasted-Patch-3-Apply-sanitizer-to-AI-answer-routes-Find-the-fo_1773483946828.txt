Patch 4 — Sanitize the SSE citation event in both stream routes
The SSE streams emit a citations event mid-stream. That's where fake citation objects can leak to the client. In both /api/search and /api/v2/stream, add an interception layer that rewrites the citations SSE event before forwarding.
Find this pattern (appears twice, in /api/search and /api/v2/stream):
typescriptwhile (true) {
  const { done, value } = await reader.read();
  if (done) break;
  const chunk = decoder.decode(value, { stream: true });
  res.write(chunk);   // ← currently passes through unchecked
}
Replace both with:
typescriptlet sseBuffer = "";

while (true) {
  const { done, value } = await reader.read();
  if (done) break;

  sseBuffer += decoder.decode(value, { stream: true });
  const lines = sseBuffer.split("\n");
  sseBuffer = lines.pop() || "";

  for (const line of lines) {
    if (!line.startsWith("data: ")) {
      res.write(line + "\n");
      continue;
    }
    try {
      const parsed = JSON.parse(line.slice(6));
      // Intercept citation events — rebuild from evidence only
      if (
        parsed?.type === "citations" ||
        parsed?.type === "meta" ||
        (parsed?.citations && !parsed?.type)
      ) {
        const safe = sanitizeGroundedResponse(parsed);
        res.write(`data: ${JSON.stringify(safe)}\n`);
      } else {
        res.write(line + "\n");
      }
    } catch {
      res.write(line + "\n");
    }
  }
}
