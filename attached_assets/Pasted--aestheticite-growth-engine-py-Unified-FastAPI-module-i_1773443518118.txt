def now_utc_iso() -> str:
    return datetime.now(timezone.utc).isoformat()

def new_id() -> str:
    return str(uuid.uuid4())

def normalize(text: str) -> str:
    return " ".join(text.lower().strip().split())

def db() -> sqlite3.Connection:
    conn = sqlite3.connect(DB_PATH)
    conn.row_factory = sqlite3.Row
    return conn

class PDFWriter:
    def __init__(self, path: str) -> None:
        self.path = path
        self.c = canvas.Canvas(path, pagesize=A4)
        self.width, self.height = A4
        self.left = 18 * mm
        self.right = self.width - 18 * mm
        self.top = self.height - 18 * mm
        self.bottom = 18 * mm
        self.y = self.top

    def new_page(self) -> None:
        self.c.showPage()
        self.y = self.top

    def ensure_space(self, needed: float) -> None:
        if self.y - needed < self.bottom:
            self.new_page()

    def line(self, text: str, font: str = "Helvetica", size: int = 10, leading: int = 14) -> None:
        self.ensure_space(leading)
        self.c.setFont(font, size)
        self.c.drawString(self.left, self.y, text[:180])
        self.y -= leading

    def wrapped(self, text: str, font: str = "Helvetica", size: int = 10, leading: int = 14, bullet: Optional[str] = None) -> None:
        max_width = self.right - self.left
        self.c.setFont(font, size)
        prefix = f"{bullet} " if bullet else ""
        indent_width = stringWidth(prefix, font, size) if bullet else 0
        usable_width = max_width - indent_width

        words = text.split()
        current = ""
        first = True

        def flush(line_text: str, is_first: bool) -> None:
            self.ensure_space(leading)
            self.c.setFont(font, size)
            if bullet:
                if is_first:
                    self.c.drawString(self.left, self.y, prefix + line_text)
                else:
                    self.c.drawString(self.left + indent_width, self.y, line_text)
            else:
                self.c.drawString(self.left, self.y, line_text)
            self.y -= leading

        for word in words:
            candidate = word if not current else f"{current} {word}"
            if stringWidth(candidate, font, size) <= usable_width:
                current = candidate
            else:
                flush(current, first)
                first = False
                current = word
        if current:
            flush(current, first)

    def section(self, title: str) -> None:
        self.y -= 4
        self.line(title, font="Helvetica-Bold", size=12, leading=16)

    def save(self) -> None:
        self.c.save()

class HNSWRetriever:
    """
    This is a placeholder adapter.

    In production:
    - store embeddings in pgvector
    - create HNSW index
    - run ANN search using cosine distance
    """
    def search(self, query: str, top_k: int) -> List[SearchResultItem]:
        conn = db()
        cur = conn.cursor()
        cur.execute("SELECT * FROM knowledge_chunks")
        rows = cur.fetchall()
        conn.close()

        terms = set(normalize(query).split())
        scored = []
        for row in rows:
            content = normalize(row["content"])
            title = normalize(row["title"])
            score = sum(1 for t in terms if t in content or t in title)
            if score > 0:
                snippet = row["content"][:240]
                scored.append(SearchResultItem(
                    chunk_id=row["id"],
                    title=row["title"],
                    snippet=snippet,
                    source_type=row["source_type"],
                    source_ref=row["source_ref"],
                    score=float(score),
                ))
        scored.sort(key=lambda x: x.score, reverse=True)
        return scored[:top_k]

def hash_api_key(raw: str) -> str:
    return hashlib.sha256(raw.encode("utf-8")).hexdigest()

def get_api_key_record(x_api_key: Optional[str] = Header(default=None)) -> sqlite3.Row:
    if not x_api_key:
        raise HTTPException(status_code=401, detail="Missing X-API-Key header.")
    conn = db()
    cur = conn.cursor()
    cur.execute("SELECT * FROM api_keys WHERE key_hash = ? AND is_active = 1", (hash_api_key(x_api_key),))
    row = cur.fetchone()
    conn.close()
    if not row:
        raise HTTPException(status_code=401, detail="Invalid API key.")
    return row

@app.post("/session-reports/{report_id}/export-pdf")
def export_session_report_pdf(report_id: str) -> Dict[str, Any]:
    payload = get_session_report(report_id)
    report = payload["report"]
    items = payload["items"]

    path = os.path.join(EXPORT_DIR, f"session_report_{report_id}.pdf")
    pdf = PDFWriter(path)
    pdf.line("AesthetiCite Session Safety Report", font="Helvetica-Bold", size=16, leading=20)
    pdf.line(f"Title: {report['title']}", font="Helvetica-Bold")
    pdf.line(f"Report ID: {report['id']}")
    pdf.line(f"Date: {report.get('report_date') or ''}")
    if report.get("notes"):
        pdf.section("Notes")
        pdf.wrapped(report["notes"])

    for idx, item in enumerate(items, start=1):
        eng = item["engine_response_json"]
        pdf.section(f"Patient / Procedure {idx}")
        pdf.wrapped(f"Patient label: {item.get('patient_label') or 'N/A'}", bullet="•")
        pdf.wrapped(f"Procedure: {item['procedure']}", bullet="•")
        pdf.wrapped(f"Region: {item['region']}", bullet="•")
        pdf.wrapped(f"Product: {item['product_type']}", bullet="•")
        if item.get("technique"):
            pdf.wrapped(f"Technique: {item['technique']}", bullet="•")
        if item.get("injector_experience_level"):
            pdf.wrapped(f"Injector experience: {item['injector_experience_level']}", bullet="•")

        if isinstance(eng, dict):
            assessment = eng.get("safety_assessment", {})
            pdf.wrapped(
                f"Decision: {assessment.get('decision', 'N/A')} | "
                f"Risk score: {assessment.get('overall_risk_score', 'N/A')}",
                bullet="•"
            )
            for risk in eng.get("top_risks", [])[:5]:
                pdf.wrapped(
                    f"Risk: {risk.get('complication')} ({risk.get('risk_score')}/100)",
                    bullet="•"
                )
            for step in eng.get("mitigation_steps", [])[:8]:
                pdf.wrapped(f"Mitigation: {step}", bullet="•")

    pdf.save()
    return {"report_id": report_id, "pdf_path": path, "filename": os.path.basename(path)}

        rec_id, payload.clinic_id, payload.clinician_id, payload.query_text,
        payload.answer_type, payload.aci_score, payload.response_time_ms,
        payload.evidence_level, now_utc_iso()

@app.get("/dashboard/{clinic_id}", response_model=DashboardResponse)
def clinic_dashboard(clinic_id: str) -> DashboardResponse:
    conn = db()
    cur = conn.cursor()
    cur.execute("SELECT * FROM query_logs WHERE clinic_id = ?", (clinic_id,))
    rows = cur.fetchall()
    conn.close()

    total = len(rows)
    aci_vals = [float(r["aci_score"]) for r in rows if r["aci_score"] is not None]
    rt_vals = [float(r["response_time_ms"]) for r in rows if r["response_time_ms"] is not None]

    top_questions = []
    q_counter = Counter(r["query_text"] for r in rows)
    for q, count in q_counter.most_common(10):
        top_questions.append({"query": q, "count": count})

    evidence_counter = Counter(r["evidence_level"] for r in rows if r["evidence_level"])
    answer_counter = Counter(r["answer_type"] for r in rows if r["answer_type"])

    return DashboardResponse(
        total_queries=total,
        average_aci_score=round(statistics.mean(aci_vals), 2) if aci_vals else None,
        average_response_time_ms=round(statistics.mean(rt_vals), 2) if rt_vals else None,
        top_questions=top_questions,
        evidence_level_distribution=dict(evidence_counter),
        answer_type_distribution=dict(answer_counter),
    )

def simplify_for_patient(title: str, source_text: str) -> str:
    clean = source_text.replace("\n", " ").strip()
    clean = " ".join(clean.split())
    short = textwrap.shorten(clean, width=1200, placeholder="...")
    simplified = (
        f"{title}\n\n"
        "Patient summary:\n"
        "This information explains the treatment or clinical point in simpler language.\n\n"
        f"{short}\n\n"
        "What this means for you:\n"
        "- Your clinician is using this information to improve safety and decision-making.\n"
        "- The exact treatment decision still depends on your own medical situation.\n"
        "- Ask your clinician if you want the risks, benefits, and alternatives explained for your case."
    )
    return simplified

@app.post("/paper-alerts/{subscription_id}/run", response_model=PaperAlertDigestResponse)
def run_paper_alert(subscription_id: str) -> PaperAlertDigestResponse:
    conn = db()
    cur = conn.cursor()
    cur.execute("SELECT * FROM paper_alert_subscriptions WHERE id = ?", (subscription_id,))
    sub = cur.fetchone()
    if not sub:
        conn.close()
        raise HTTPException(status_code=404, detail="Subscription not found.")

    papers = pubmed_adapter.search_new_papers(sub["topic"], sub["last_checked_utc"])
    new_items = []
    for p in papers:
        item_id = new_id()
        cur.execute("""
            INSERT INTO paper_alert_items (id, subscription_id, paper_title, paper_abstract, source_url, published_date, created_at_utc)
            VALUES (?, ?, ?, ?, ?, ?, ?)
        """, (
            item_id,
            subscription_id,
            p["title"],
            p.get("abstract"),
            p.get("url"),
            p.get("published_date"),
            now_utc_iso()
        ))
        new_items.append(p)

    cur.execute("UPDATE paper_alert_subscriptions SET last_checked_utc = ? WHERE id = ?", (now_utc_iso(), subscription_id))
    conn.commit()
    conn.close()

    return PaperAlertDigestResponse(subscription_id=subscription_id, topic=sub["topic"], new_items=new_items)
