INJECTABLE_TRIGGERS = [
    "filler", "hyaluronic", "ha", "hyaluronidase", "botulinum", "botox", "dysport", "xeomin", "bocouture",
    "glabella", "tear trough", "nasolabial", "temple", "nose", "forehead", "cannula", "needle", "vascular occlusion",

DEVICE_TRIGGERS = [

    "nd:yag", "1064", "532", "picosecond", "microneedling"

REG_TRIGGERS = ["fda", "ce mark", "ce-mark", "approved", "off-label", "label", "indication"]
HIGH_RISK_ZONE = ["glabella", "nose", "nasal", "tear trough", "forehead", "blindness", "vision loss"]

EVIDENCE_WEIGHTS = {
    "guideline": 1.00,
    "meta-analysis": 0.95,
    "systematic-review": 0.90,
    "RCT": 0.85,
    "cohort": 0.70,
    "case-control": 0.65,
    "review": 0.55,
    "case-series": 0.45,
    "case-report": 0.35,
    "expert-opinion": 0.25,
    None: 0.40,
}

def classify_query(query: str) -> Dict[str, Any]:
    q = query.lower()
    injectable = any(t in q for t in INJECTABLE_TRIGGERS)
    device = any(t in q for t in DEVICE_TRIGGERS)
    needs_reg = any(t in q for t in REG_TRIGGERS) or injectable or device
    risk = "high" if any(t in q for t in HIGH_RISK_ZONE) else ("medium" if injectable else "low")
    return {"is_injectable": injectable, "is_device": device, "needs_regulatory": needs_reg, "risk_level": risk}

_WORD_RE = re.compile(r"\w+", flags=re.UNICODE)

def _fold_accents(s: str) -> str:
    # sécurité -> securite
    return "".join(ch for ch in unicodedata.normalize("NFD", s) if unicodedata.category(ch) != "Mn")

def _light_stem(tok: str) -> str:
    # conservative; only ASCII tokens
    if not tok or not tok.isascii():
        return tok
    for suf in ("ments","ment","ations","ation","iques","ique","eurs","eur","ées","ée","es","s"):
        if tok.endswith(suf) and len(tok) > len(suf) + 3:
            return tok[:-len(suf)]
    return tok

@lru_cache(maxsize=80_000)
def _tokens(text: str) -> Tuple[str, ...]:
    if not text:
        return tuple()
    t = text.lower()
    raw = _WORD_RE.findall(t)
    toks = [w for w in raw if len(w) >= 2]

    # folded tokens
    folded = _fold_accents(" ".join(toks))
    ftoks = [w for w in _WORD_RE.findall(folded) if len(w) >= 2]

    out: List[str] = []
    for w in toks:
        out.append(w); out.append(_light_stem(w))
    for w in ftoks:
        out.append(w); out.append(_light_stem(w))

    # de-dup
    seen = set()
    uniq = []
    for w in out:
        if w and w not in seen:
            seen.add(w)
            uniq.append(w)
    return tuple(uniq)

@lru_cache(maxsize=80_000)
def _token_set(text: str) -> frozenset:
    return frozenset(_tokens(text))

def support_score(claim: str, evidence_text: str) -> float:
    """
    0..1 overlap score robust to accents and minor morphology.
    Optimized for speed (cached tokens).
    """
    if not claim or not evidence_text:
        return 0.0
    cset = _token_set(claim)
    eset = _token_set(evidence_text)
    if not cset or not eset:
        return 0.0

    inter = len(cset & eset)
    if inter == 0:
        return 0.0

    claim_cov = inter / max(1, len(cset))
    jacc = inter / max(1, len(cset | eset))

    # tiny phrase bonus
    bonus = 0.0
    fc = _fold_accents(claim.lower())
    fe = _fold_accents(evidence_text.lower())
    cw = _WORD_RE.findall(fc)
    if len(cw) >= 4:
        phrase = " ".join(cw[:3])
        if phrase and phrase in fe:
            bonus = 0.04

    return float(max(0.0, min(1.0, 0.72*claim_cov + 0.28*jacc + bonus)))

# 3) ACI scoring (and strong fallback)
# =========================

def _year_modifier(year: Optional[int], now_year: int = 2026) -> float:
    if not year:
        return 0.90
    age = max(0, now_year - year)
    if age <= 2: return 1.00
    if age <= 5: return 0.95
    if age <= 10: return 0.90
    return 0.85

def _consensus_modifier(n_sources: int) -> float:
    if n_sources >= 4: return 1.00
    if n_sources == 3: return 0.95
    if n_sources == 2: return 0.90
    if n_sources == 1: return 0.80
    return 0.60

def compute_aci_from_evidence(items: List[EvidenceItem], risk_level: str) -> float:
    if not items:
        return 0.0
    base = min(sum(i.weight for i in items), 4.0)
    consensus = _consensus_modifier(len({i.chunk.source for i in items if i.chunk and i.chunk.source}))
    risk_penalty = {"low": 0.0, "medium": 0.6, "high": 1.2}.get(risk_level, 0.6)
    score = 10.0 * (1.0 - math.exp(-base)) * consensus
    score = max(0.0, min(10.0, score - risk_penalty))
    return round(score, 1)

def compute_aci_from_references(refs: List[Dict[str, Any]], risk_level: str) -> float:
    if not refs:
        return 0.0
    weights = []
    sources = set()
    for r in refs:
        et = r.get("evidence_type")
        year = r.get("year")
        w = EVIDENCE_WEIGHTS.get(et, EVIDENCE_WEIGHTS[None]) * _year_modifier(year)
        weights.append(w)
        if r.get("source"):
            sources.add(r["source"])

    base = min(sum(weights), 4.0)
    consensus = _consensus_modifier(len(sources))
    risk_penalty = {"low": 0.0, "medium": 0.6, "high": 1.2}.get(risk_level, 0.6)
    score = 10.0 * (1.0 - math.exp(-base)) * consensus
    score = max(0.0, min(10.0, score - risk_penalty))
    return round(score, 1)

def _hash(s: str) -> str:
    return hashlib.sha256(s.encode("utf-8", errors="ignore")).hexdigest()[:24]

@lru_cache(maxsize=10_000)
def cached_retrieve(query: str, k: int, filters_json: str) -> Tuple[Chunk, ...]:
    filters = json.loads(filters_json)
    chunks = hybrid_retrieve(query, k=k, filters=filters) or []
    # ensure tags list exists
    fixed = []
    for c in chunks:
        if c.tags is None:
            c.tags = []
        fixed.append(c)
    return tuple(fixed)

def evidence_boost(c: Chunk, meta: Dict[str, Any]) -> float:
    s = c.score or 1.0
    et = c.evidence_type
    s *= (0.95 + 0.25 * EVIDENCE_WEIGHTS.get(et, 0.40))  # evidence-type preference

    tags = set(c.tags or [])
    if meta["is_injectable"] and "injectables" in tags: s *= 1.25
    if meta["is_injectable"] and "complications" in tags: s *= 1.25
    if meta["is_device"] and "devices" in tags: s *= 1.20
    if meta["needs_regulatory"] and "regulatory" in tags: s *= 1.10

    # recency helps (slightly)
    s *= _year_modifier(c.year)
    return s

def retrieve_vertical(plan: Dict[str, Any], meta: Dict[str, Any], cap: int = 28) -> List[Chunk]:
    filters = {
        "domain": "aesthetic_medicine",
        "prefer": ["guideline", "meta-analysis", "systematic-review", "RCT"],
        "needs_regulatory": meta["needs_regulatory"],
    }
    filters_json = json.dumps(filters, ensure_ascii=False, sort_keys=True)

    # Parallelism reducer: do fewer queries; keep best ones
    queries = (plan.get("search_queries") or [])[:6]
    if not queries:
        queries = [plan.get("query", "")]

    pool: List[Chunk] = []
    seen = set()
    for q in queries:
        for c in cached_retrieve(q, k=10, filters_json=filters_json):
            key = (c.source, c.title, c.year or 0)
            if key in seen:
                continue
            seen.add(key)
            pool.append(c)

    # Rerank with aesthetic boosts
    for c in pool:
        c.score = evidence_boost(c, meta)

    pool.sort(key=lambda x: (x.score or 0.0), reverse=True)
    return pool[:cap]

def select_support_for_claim(claim: str, chunks: List[Chunk], min_score: float) -> List[Tuple[float, Chunk, str]]:
    """
    Selects top evidence excerpts for a claim using token overlap scoring.
    """
    scored: List[Tuple[float, Chunk, str]] = []
    for ch in chunks:
        s = support_score(claim, ch.text + " " + ch.title)
        if s >= min_score:
            excerpt = ch.text[:280].strip()
            scored.append((s, ch, excerpt))
    scored.sort(key=lambda x: x[0], reverse=True)
    return scored[:4]

def gate_and_build_evidence(claims: List[Claim], chunks: List[Chunk], meta: Dict[str, Any]) -> Tuple[List[Claim], Dict[str, List[EvidenceItem]]]:
    allowed: List[Claim] = []
    evidence_by_claim: Dict[str, List[EvidenceItem]] = {}

    # threshold tuning: FR benefits from slightly lower overlap due to morphology; high-risk needs stricter minimum
    base_min = 0.14 if meta["risk_level"] != "high" else 0.16

    for cl in claims:
        supports = select_support_for_claim(cl.claim, chunks, min_score=base_min)
        items: List[EvidenceItem] = []
        for s, ch, excerpt in supports:
            et = ch.evidence_type
            w = EVIDENCE_WEIGHTS.get(et, EVIDENCE_WEIGHTS[None]) * _year_modifier(ch.year)
            # make overlap matter (tiny multiplier)
            w *= (0.85 + 0.30 * min(1.0, s))
            items.append(EvidenceItem(chunk=ch, excerpt=excerpt, weight=w))

        if items:
            allowed.append(cl)
            evidence_by_claim[cl.claim] = items

    return allowed, evidence_by_claim

def build_references(evidence_by_claim: Dict[str, List[EvidenceItem]]) -> List[Dict[str, Any]]:
    refs = []
    seen = set()
    idx = 1
    for items in evidence_by_claim.values():
        for ei in items:
            ch = ei.chunk
            key = (ch.source, ch.title, ch.year or 0)
            if key in seen:
                continue
            seen.add(key)
            refs.append({
                "id": idx,
                "title": ch.title,
                "year": ch.year,
                "evidence_type": ch.evidence_type,
                "source": ch.source,
                "doi": ch.doi,
                "journal": ch.journal,
                "tags": ch.tags,
                "retrieval_score": ch.score,
            })
            idx += 1
    return refs

def draft_answer(query: str, meta: Dict[str, Any], allowed: List[Claim], evidence_by_claim: Dict[str, List[EvidenceItem]], refs: List[Dict[str, Any]], protocol: Optional[Dict[str, Any]]) -> str:
    # overall ACI from evidence
    all_items = [ei for lst in evidence_by_claim.values() for ei in lst]
    aci = compute_aci_from_evidence(all_items, meta["risk_level"])

    # strict fallback: if ACI is 0 but refs exist, compute from refs
    if aci == 0.0 and refs:
        aci = compute_aci_from_references(refs, meta["risk_level"])

    # Build pack for writer: only include allowed claims and their excerpts
    pack_lines = []
    for cl in allowed:
        eis = evidence_by_claim.get(cl.claim, [])
        # map each excerpt to a reference id by chunk identity (best effort)
        for ei in eis:
            # find ref id
            rid = None
            for r in refs:
                if r["source"] == ei.chunk.source and r["title"] == ei.chunk.title and (r.get("year") or 0) == (ei.chunk.year or 0):
                    rid = r["id"]
                    break
            pack_lines.append(f"CLAIM: {cl.claim}\nEXCERPT[{rid}]: {ei.excerpt}\nRISK: {cl.risk_level}\n")

    protocol_txt = ""
    if protocol:
        protocol_txt = (
            f"PROTOCOL_BLOCK_TITLE: {protocol['title']}\n"
            f"RED_FLAGS: {', '.join(protocol['red_flags'])}\n"
            f"IMMEDIATE_ACTIONS: {', '.join(protocol['immediate_actions'])}\n"
            f"NOTE: {protocol['note']}\n"
        )

    system = (
        "You are AesthetiCite Writer (aesthetic medicine).\n"
        "Hard Rules:\n"
        "- Do NOT introduce any clinical claim that is not supported by EXCERPT[...] lines.\n"
        "- Use narrative prose (no bullets unless needed for safety clarity).\n"
        "- When using an excerpt, cite it with the bracket id like [1].\n"
        "- If evidence is thin, downgrade language (may/limited evidence).\n"
        "- Never invent numeric dosing; include math only if provided by tools.\n"
        "Structure:\n"
        "Clinical Summary (2–4 sentences)\n"
        "Evidence & Rationale (tie to citations)\n"
        "Practical Considerations (aesthetic context + safety)\n"
    )

    user = (
        f"USER QUESTION:\n{query}\n\n"
        f"OVERALL_ACI: {aci}/10\n"
        f"QUERY_META: {meta}\n\n"
        f"{protocol_txt}\n"
        "SUPPORTED_EVIDENCE_PACK:\n" + "\n".join(pack_lines) + "\n"
        "REFERENCES:\n" + "\n".join([f"[{r['id']}] {r['title']} ({r.get('year','n.d.')}) {r['source']}" for r in refs])
    )

    text = llm_text(system=system, user=user).strip()
    return f"Aesthetic Confidence Index (ACI™): {aci}/10\n\n{text}"

def deepconsult(query: str) -> Dict[str, Any]:
    t0 = time.time()
    meta = classify_query(query)

    # Step A: plan
    plan = make_plan(query)
    claims = make_claims(plan, meta)

    # Step B: retrieve (vertical + rerank + cached)
    chunks = retrieve_vertical(plan, meta, cap=28)

    # Step C: gate & evidence map (fast overlap scorer)
    allowed, evidence_by_claim = gate_and_build_evidence(claims, chunks, meta)

    # Step D: safety layer + tools
    protocol = complication_protocol_layer(query)
    tool_calls = decide_tools(meta, query)
    tool_results = [execute_tool(t["name"], t["args"]) for t in tool_calls]

    # Step E: references list from evidence_by_claim (truth source)
    refs = build_references(evidence_by_claim)

    # If nothing allowed, fail safely with guidance
    if not allowed or not refs:
        return {
            "ok": True,
            "answer": (
                "Aesthetic Confidence Index (ACI™): 0/10\n\n"
                "I couldn’t find directly supporting sources in the current corpus for this question as phrased. "
                "Try specifying: (1) product/device, (2) exact region/indication, (3) patient context (age/sex/skin type), "
                "and (4) which outcome you want (efficacy, longevity, complication management)."
            ),
            "aci": 0.0,
            "references": refs,
            "tools": tool_results,
            "debug": {
                "latency_s": round(time.time() - t0, 2),
                "n_chunks": len(chunks),
                "n_claims": len(claims),
                "n_allowed": len(allowed),
            }
        }

    answer = draft_answer(query, meta, allowed, evidence_by_claim, refs, protocol)

    # Compute final ACI value from answer header
    m = re.search(r"ACI™\):\s*([0-9.]+)/10", answer)
    aci_val = float(m.group(1)) if m else 0.0

    return {
        "ok": True,
        "answer": answer,
        "aci": aci_val,
        "references": refs,
        "tools": tool_results,
        "debug": {
            "latency_s": round(time.time() - t0, 2),
            "n_chunks": len(chunks),
            "n_claims": len(claims),
            "n_allowed": len(allowed),
            "retrieval_top_sources": [c.source for c in chunks[:5]],
        }
    }

app = FastAPI(title="AesthetiCite Competitive Core")

class QueryPayload(BaseModel):
    query: str

@app.post("/deepconsult")
def deepconsult_api(payload: QueryPayload):
    q = (payload.query or "").strip()
    if not q:
        return JSONResponse({"ok": False, "error": "Missing query"}, status_code=400)
    return JSONResponse(deepconsult(q))
