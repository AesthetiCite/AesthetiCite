INJECTABLE_TRIGGERS = [
    "filler","hyaluronic","ha","hyaluronidase","botulinum","botox","dysport","xeomin","bocouture",
    "glabella","tear trough","nasolabial","temple","nose","forehead","cannula","needle",
    "vascular occlusion","blindness","vision loss"
]
DEVICE_TRIGGERS = ["laser","ipl","rf","radiofrequency","hifu","fractional","co2","erbium","nd:yag","1064","532","picosecond","microneedling"]
REG_TRIGGERS = ["fda","ce mark","ce-mark","approved","off-label","label","indication"]
HIGH_RISK_ZONE = ["glabella","nose","nasal","tear trough","forehead","blindness","vision loss"]

def classify_query(query: str) -> Dict[str, Any]:
    q = query.lower()
    injectable = any(t in q for t in INJECTABLE_TRIGGERS)
    device = any(t in q for t in DEVICE_TRIGGERS)
    needs_reg = any(t in q for t in REG_TRIGGERS) or injectable or device
    risk = "high" if any(t in q for t in HIGH_RISK_ZONE) else ("medium" if injectable else "low")
    return {"is_injectable": injectable, "is_device": device, "needs_regulatory": needs_reg, "risk_level": risk}

def complication_protocol_layer(query: str) -> Optional[Dict[str, Any]]:
    meta = classify_query(query)
    if not meta["is_injectable"]:
        return None
    return {
        "title": "Complication Protocol Layer (Injectables)",
        "red_flags": [
            "Sudden severe pain, blanching, livedo/reticular discoloration",
            "Cool skin, delayed capillary refill",
            "Visual symptoms (blurred vision, scotoma), ptosis, ophthalmoplegia",
        ],
        "immediate_actions": [
            "Stop injection immediately; do not continue in the area.",
            "Assess perfusion and symptoms; document time of onset.",
            "Initiate your clinic protocol; urgently escalate for any ocular symptoms.",
        ],
        "note": "Risk-awareness aid only. Follow IFUs, formal training, and local emergency protocols."
    }

# Time budget + caching
# =========================

class TimeBudget:
    def __init__(self, seconds: float):
        self.t0 = time.time()
        self.seconds = seconds
    def left(self) -> float:
        return max(0.0, self.seconds - (time.time() - self.t0))
    def ok(self) -> bool:
        return self.left() > 0.0

def _hash(s: str) -> str:
    return hashlib.sha256(s.encode("utf-8", errors="ignore")).hexdigest()[:24]

@lru_cache(maxsize=10_000)
def cached_retrieve(query: str, k: int, filters_json: str) -> Tuple[Tuple]:
    filters = json.loads(filters_json)
    chunks = hybrid_retrieve(query, k=k, filters=filters) or []
    # store as tuples to make cache hashable
    out = []
    for c in chunks:
        out.append((c.title, c.source, c.year, c.evidence_type, " ".join(c.tags or []), c.text))
    return tuple(out)

def _unpack_chunks(packed: Tuple[Tuple]) -> List[Chunk]:
    out = []
    for title, source, year, evidence_type, tags_str, text in packed:
        tags = tags_str.split() if tags_str else []
        out.append(Chunk(title=title, source=source, year=year, evidence_type=evidence_type, tags=tags, text=text))
    return out


# =========================
# Parallel retrieval (BIG latency win)
# =========================

async def retrieve_parallel(queries: List[str], *, k: int, filters: Dict[str, Any], budget: TimeBudget) -> List[Chunk]:
    # Keep it small: 4–6 queries max
    queries = [q.strip() for q in queries if q and q.strip()][:6]
    if not queries:
        return []

    filters_json = json.dumps(filters, ensure_ascii=False, sort_keys=True)

    loop = asyncio.get_event_loop()

    async def one(q: str) -> List[Chunk]:
        if not budget.ok():
            return []
        # cached_retrieve is sync; run in executor
        packed = await loop.run_in_executor(None, cached_retrieve, q, k, filters_json)
        return _unpack_chunks(packed)

    tasks = [asyncio.create_task(one(q)) for q in queries]

    # Hard stop: don’t wait forever
    try:
        done, pending = await asyncio.wait(tasks, timeout=min(6.5, budget.left()))
    except Exception:
        done, pending = set(), set(tasks)

    for p in pending:
        p.cancel()

    pool: List[Chunk] = []
    seen = set()
    for d in done:
        try:
            for c in d.result() or []:
                key = (c.source, c.title, c.year or 0)
                if key in seen:
                    continue
                seen.add(key)
                pool.append(c)
        except Exception:
            pass

    # lightweight rerank: prefer evidence types + tags
    def boost(c: Chunk) -> float:
        s = c.score or 1.0
        et = (c.evidence_type or "").lower()
        if "guideline" in et or "consensus" in et: s *= 1.25
        elif "meta" in et or "systematic" in et: s *= 1.18
        elif "rct" in et or "random" in et: s *= 1.12
        tags = set((c.tags or []))
        # heuristic boosts by query type will be applied later if needed
        if "complications" in tags: s *= 1.08
        return s

    pool.sort(key=boost, reverse=True)
    return pool[:24]  # cap context size

def make_micro_queries(query: str, meta: Dict[str, Any]) -> List[str]:
    # No Planner LLM: cheap deterministic query expansion
    base = query.strip()
    qs = [base]

    if meta["is_injectable"]:
        qs += [
            f"{base} guideline consensus complication management",
            f"{base} vascular occlusion recognition immediate management",
            f"{base} ocular symptoms escalation filler complication",
        ]
    if meta["is_device"]:
        qs += [
            f"{base} Fitzpatrick risk PIH mitigation review",
            f"{base} device IFU safety considerations",
        ]
    if meta["needs_regulatory"]:
        qs += [f"{base} on-label off-label indication safety wording"]

    # de-dup
    out = []
    seen = set()
    for q in qs:
        k = q.lower()
        if k not in seen:
            seen.add(k)
            out.append(q)
    return out[:6]

def format_context(chunks: List[Chunk]) -> str:
    # Chunk IDs used for citations: [S1], [S2], ...
    lines = []
    for i, c in enumerate(chunks, 1):
        et = c.evidence_type or "unknown"
        yr = c.year or "n.d."
        lines.append(f"[S{i}] {c.title} ({yr}) {et}\nSOURCE: {c.source}\nTEXT: {c.text[:1100]}\n")
    return "\n".join(lines)

async def deepconsult_under_20s(query: str, *, budget_s: float = 18.0) -> Dict[str, Any]:
    budget = TimeBudget(budget_s)
    meta = classify_query(query)

    filters = {
        "domain": "aesthetic_medicine",
        "prefer": ["guideline", "meta-analysis", "systematic-review", "RCT"],
        "needs_regulatory": meta["needs_regulatory"],
        "tags_any": (["injectables","complications"] if meta["is_injectable"] else []) + (["devices"] if meta["is_device"] else []),
    }

    micro_queries = make_micro_queries(query, meta)

    chunks = await retrieve_parallel(micro_queries, k=10, filters=filters, budget=budget)

    protocol = complication_protocol_layer(query)

    # If retrieval fails, return safe response fast
    if not chunks:
        return {
            "ok": True,
            "answer": (
                "Aesthetic Confidence Index (ACI™): 0/10\n\n"
                "I couldn’t find supporting sources in the current corpus for this question as phrased. "
                "Please specify procedure/product/device, anatomical region, and your target outcome."
            ),
            "aci": 0.0,
            "references": [],
            "debug": {"latency_s": round(budget_s - budget.left(), 2), "n_chunks": 0}
        }

    ctx = format_context(chunks)

    # ONE LLM CALL: writer must cite [S#] for every clinical claim
    system = (
        "You are AesthetiCite Writer (aesthetic medicine, evidence-grounded).\n"
        "HARD RULES:\n"
        "- Every clinical claim MUST include at least one citation to the provided sources in the form [S#].\n"
        "- If you cannot cite a claim, do NOT state it.\n"
        "- Do NOT invent dosing numbers or device parameters.\n"
        "- Write for clinicians: concise, actionable, complication-aware.\n"
        "OUTPUT FORMAT:\n"
        "Clinical Summary (2–4 sentences)\n"
        "Evidence & Rationale (tie claims to [S#])\n"
        "Practical Considerations (aesthetic context + safety)\n"
        "References (list S# used)\n"
    )

    proto_txt = ""
    if protocol:
        proto_txt = (
            f"\nSAFETY_LAYER:\n"
            f"{protocol['title']}\n"
            f"Red flags: {', '.join(protocol['red_flags'])}\n"
            f"Immediate actions: {', '.join(protocol['immediate_actions'])}\n"
            f"Note: {protocol['note']}\n"
        )

    user = (
        f"QUESTION:\n{query}\n\n"
        f"QUERY_META: {meta}\n"
        f"{proto_txt}\n"
        f"SOURCES:\n{ctx}\n"
    )

    # enforce time: if we're late, shorten context automatically (still 1 call)
    if budget.left() < 6.0:
        # keep top 8 sources only
        short_ctx = format_context(chunks[:8])
        user = user.split("SOURCES:\n")[0] + "SOURCES:\n" + short_ctx + "\n"

    answer = llm_text(system=system, user=user).strip()

    # Build references list (UI-friendly)
    used = sorted(set(re.findall(r"\[S(\d+)\]", answer)))
    used_ids = {int(x) for x in used} if used else set()

    refs = []
    for i, c in enumerate(chunks, 1):
        if i in used_ids:
            refs.append({
                "id": f"S{i}",
                "title": c.title,
                "year": c.year,
                "evidence_type": c.evidence_type,
                "source": c.source,
            })

    # Very fast, honest ACI proxy: score by evidence strength + diversity of cited sources
    # (This keeps ACI responsive under 20s; you can replace with your full ACI later.)
    def aci_proxy(refs: List[Dict[str, Any]], risk: str) -> float:
        if not refs:
            return 0.0
        wt = 0.0
        for r in refs:
            et = (r.get("evidence_type") or "").lower()
            if "guideline" in et or "consensus" in et: wt += 1.0
            elif "meta" in et or "systematic" in et: wt += 0.95
            elif "rct" in et or "random" in et: wt += 0.85
            elif "cohort" in et: wt += 0.70
            else: wt += 0.55
        wt = min(wt, 4.0)
        diversity = len({r.get("source") for r in refs if r.get("source")})
        cons = 1.0 if diversity >= 4 else (0.95 if diversity == 3 else (0.90 if diversity == 2 else 0.80))
        penalty = 0.0 if risk == "low" else (0.6 if risk == "medium" else 1.2)
        score = 10.0 * (1.0 - (2.71828 ** (-wt))) * cons
        score = max(0.0, min(10.0, score - penalty))
        return round(score, 1)

    aci = aci_proxy(refs, meta["risk_level"])

    final = f"Aesthetic Confidence Index (ACI™): {aci}/10\n\n{answer}"

    return {
        "ok": True,
        "answer": final,
        "aci": aci,
        "references": refs,
        "debug": {
            "latency_s": round(budget_s - budget.left(), 2),
            "n_chunks": len(chunks),
            "n_refs_used": len(refs),
            "queries": micro_queries,
        }
    }


# =========================
# FastAPI
# =========================

app = FastAPI(title="AesthetiCite Latency <20s Patch")

class QueryPayload(BaseModel):
    query: str
    budget_s: Optional[float] = 18.0

@app.post("/deepconsult")
async def deepconsult_api(payload: QueryPayload):
    q = (payload.query or "").strip()
    if not q:
        return JSONResponse({"ok": False, "error": "Missing query"}, status_code=400)
    budget_s = float(payload.budget_s or 18.0)
    budget_s = max(8.0, min(25.0, budget_s))  # safe clamp
    out = await deepconsult_under_20s(q, budget_s=budget_s)
    return JSONResponse(out)
