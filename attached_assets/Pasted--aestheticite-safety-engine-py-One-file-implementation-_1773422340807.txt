def now_utc_iso() -> str:
    return datetime.now(timezone.utc).isoformat()

def normalize(text: str) -> str:
    text = text.lower().strip()
    text = re.sub(r"\s+", " ", text)
    return text

def context_blob(context: Optional[ClinicalContext]) -> str:
    if not context:
        return ""
    parts: List[str] = []
    for value in [
        context.region,
        context.procedure,
        context.product_type,
        context.free_text,
        " ".join(context.symptoms),
    ]:
        if value:
            parts.append(value)
    return normalize(" ".join(parts))

def detect_procedure_insight(query: str, context: Optional[ClinicalContext]) -> Optional[ProcedureInsight]:
    blob = normalize(f"{query} {context_blob(context)}")
    for procedure_name, data in PROCEDURE_LIBRARY.items():
        if any(alias in blob for alias in data["aliases"]):
            return ProcedureInsight(
                procedure_name=procedure_name,
                likely_plane=data["likely_plane"],
                key_danger_zones=data["danger_zones"],
                technique_notes=data["technique_notes"],
                common_products_or_classes=data["products"],
            )
    return None

def score_protocol(query: str, context: Optional[ClinicalContext], protocol: Dict[str, Any]) -> float:
    q = normalize(query)
    blob = context_blob(context)
    score = 0.0

    for kw in protocol["keywords"]:
        if kw in q:
            score += 1.0
        if kw in blob:
            score += 0.5

    if context:
        if protocol["name"].startswith("Suspected vascular occlusion"):
            if context.visual_symptoms:
                score += 3.0
            if context.capillary_refill_delayed:
                score += 2.0
            if context.skin_color_change and context.skin_color_change.lower() in {"blanching", "mottling", "dusky", "violaceous"}:
                score += 2.0
            if context.pain_score_10 is not None and context.pain_score_10 >= 7:
                score += 1.5

        if protocol["name"].startswith("Suspected anaphylaxis"):
            if context.wheeze:
                score += 2.0
            if context.hypotension:
                score += 2.5
            if context.urticaria:
                score += 1.0
            if context.tongue_or_facial_swelling:
                score += 2.0

        if protocol["name"].startswith("Suspected Tyndall"):
            if context.skin_color_change and "blue" in context.skin_color_change.lower():
                score += 2.0
            if context.pain_score_10 is not None and context.pain_score_10 <= 2:
                score += 0.5

        if protocol["name"].startswith("Suspected ptosis"):
            if context.eyelid_droop:
                score += 2.0
            if context.brow_heaviness:
                score += 1.0
            if context.toxin_recent:
                score += 2.0
            if context.diplopia:
                score += 1.5

    return score

def select_protocol(query: str, context: Optional[ClinicalContext]) -> Tuple[str, Dict[str, Any], float]:
    best_key = None
    best_protocol = None
    best_score = -1.0

    for key, protocol in PROTOCOL_LIBRARY.items():
        score = score_protocol(query, context, protocol)
        if score > best_score:
            best_key = key
            best_protocol = protocol
            best_score = score

    if not best_key or not best_protocol or best_score <= 0:
        raise HTTPException(status_code=404, detail="No matching aesthetic protocol found.")

    confidence = min(0.99, round(0.45 + best_score * 0.06, 2))
    return best_key, best_protocol, confidence

def compute_risk_assessment(protocol_key: str, protocol: Dict[str, Any], context: Optional[ClinicalContext]) -> RiskAssessment:
    base = {"low": 20, "moderate": 50, "high": 75, "critical": 92}
    severity: SeverityLevel = protocol["base_severity"]
    urgency: UrgencyLevel = protocol["base_urgency"]
    score = base[severity]

    if context:
        if protocol_key == "vascular_occlusion":
            if context.visual_symptoms:
                score += 8
                severity = "critical"
                urgency = "immediate"
            if context.capillary_refill_delayed:
                score += 5
            if context.pain_score_10 is not None and context.pain_score_10 >= 8:
                score += 4

        elif protocol_key == "anaphylaxis":
            if context.hypotension or context.wheeze or context.tongue_or_facial_swelling:
                score += 6
                severity = "critical"
                urgency = "immediate"

        elif protocol_key == "ptosis_after_toxin":
            if context.diplopia or context.visual_symptoms:
                score += 10
                severity = "high"
                urgency = "urgent"

    score = max(0, min(100, score))
    return RiskAssessment(
        risk_score=score,
        severity=severity,
        urgency=urgency,
        likely_time_critical=urgency in ("urgent", "immediate"),
        evidence_strength=protocol["evidence_strength"],
    )

def build_steps(protocol: Dict[str, Any]) -> List[ProtocolStep]:
    return [
        ProtocolStep(
            step_number=i + 1,
            action=s["action"],
            rationale=s["rationale"],
            priority=s.get("priority", "primary"),
        )
        for i, s in enumerate(protocol["steps"])
    ]

def build_doses(protocol: Dict[str, Any]) -> List[DoseGuidance]:
    return [
        DoseGuidance(
            substance=d["substance"],
            recommendation=d["recommendation"],
            notes=d["notes"],
        )
        for d in protocol["dose_guidance"]
    ]

def build_decision_response(request_id: str, protocol_key: str, protocol: Dict[str, Any], confidence: float, query: str, context: Optional[ClinicalContext]) -> DecisionResponse:
    evidence = evidence_retriever.retrieve(protocol_key, query, context)
    risk = compute_risk_assessment(protocol_key, protocol, context)
    procedure_insight = detect_procedure_insight(query, context)

    return DecisionResponse(
        request_id=request_id,
        generated_at_utc=now_utc_iso(),
        engine_version=ENGINE_VERSION,
        knowledge_revision=KNOWLEDGE_REVISION,
        matched_protocol_key=protocol_key,
        matched_protocol_name=protocol["name"],
        confidence=confidence,
        risk_assessment=risk,
        clinical_summary=protocol["summary"],
        immediate_actions=build_steps(protocol),

class PDFWriter:
    def __init__(self, path: str) -> None:
        self.path = path
        self.c = canvas.Canvas(path, pagesize=A4)
        self.width, self.height = A4
        self.left = 18 * mm
        self.right = self.width - 18 * mm
        self.top = self.height - 18 * mm
        self.bottom = 18 * mm
        self.y = self.top

    def new_page(self) -> None:
        self.c.showPage()
        self.y = self.top

    def ensure_space(self, needed: float) -> None:
        if self.y - needed < self.bottom:
            self.new_page()

    def line(self, text: str, font: str = "Helvetica", size: int = 10, leading: int = 14) -> None:
        self.ensure_space(leading)
        self.c.setFont(font, size)
        self.c.drawString(self.left, self.y, text)
        self.y -= leading

    def wrapped(self, text: str, font: str = "Helvetica", size: int = 10, leading: int = 14, bullet: Optional[str] = None) -> None:
        max_width = self.right - self.left
        self.c.setFont(font, size)

        prefix = f"{bullet} " if bullet else ""
        indent_width = stringWidth(prefix, font, size) if bullet else 0
        first_limit = max_width - indent_width
        other_limit = max_width - indent_width

        words = text.split()
        line = ""
        first = True

        def flush(out: str, first_line: bool) -> None:
            self.ensure_space(leading)
            self.c.setFont(font, size)
            if bullet:
                if first_line:
                    self.c.drawString(self.left, self.y, prefix + out)
                else:
                    self.c.drawString(self.left + indent_width, self.y, out)
            else:
                self.c.drawString(self.left, self.y, out)
            self.y -= leading

        for word in words:
            candidate = word if not line else f"{line} {word}"
            limit = first_limit if first else other_limit
            if stringWidth(candidate, font, size) <= limit:
                line = candidate
            else:
                flush(line, first)
                first = False
                line = word
        if line:
            flush(line, first)

    def section(self, title: str) -> None:
        self.y -= 4
        self.line(title, font="Helvetica-Bold", size=12, leading=16)

    def save(self) -> None:
        self.c.save()

def export_pdf(response: DecisionResponse) -> str:
    filename = f"aestheticite_{response.matched_protocol_key}_{response.request_id}.pdf"
    path = os.path.join(EXPORT_DIR, filename)
    pdf = PDFWriter(path)

    pdf.line("AesthetiCite Protocol + Evidence Export", font="Helvetica-Bold", size=16, leading=20)
    pdf.line(f"Protocol: {response.matched_protocol_name}", font="Helvetica-Bold", size=11)
    pdf.line(f"Request ID: {response.request_id}")
    pdf.line(f"Generated: {response.generated_at_utc}")
    pdf.line(f"Version: {response.engine_version} | Revision: {response.knowledge_revision}")
    pdf.line(
        f"Risk Score: {response.risk_assessment.risk_score}/100 | "
        f"Severity: {response.risk_assessment.severity} | "
        f"Urgency: {response.risk_assessment.urgency}"
    )

    pdf.section("Clinical Summary")
    pdf.wrapped(response.clinical_summary)

    if response.procedure_insight:
        pdf.section("Procedure-Specific Insight")
        pdf.wrapped(f"Procedure: {response.procedure_insight.procedure_name}", font="Helvetica-Bold")
        if response.procedure_insight.likely_plane:
            pdf.wrapped(f"Likely plane: {response.procedure_insight.likely_plane}", bullet="•")
        for dz in response.procedure_insight.key_danger_zones:
            pdf.wrapped(f"Danger zone: {dz}", bullet="•")
        for note in response.procedure_insight.technique_notes:
            pdf.wrapped(f"Technique note: {note}", bullet="•")

    pdf.section("Immediate Actions")
    for step in response.immediate_actions:
        pdf.wrapped(f"Step {step.step_number}: {step.action}", font="Helvetica-Bold", bullet="•")
        pdf.wrapped(f"Rationale: {step.rationale}", bullet="-")

    pdf.section("Dose Guidance")
    if response.dose_guidance:
        for dose in response.dose_guidance:
            pdf.wrapped(f"{dose.substance}: {dose.recommendation}", font="Helvetica-Bold", bullet="•")
            pdf.wrapped(f"Notes: {dose.notes}", bullet="-")
    else:
        pdf.wrapped("No dose guidance available.", bullet="•")

    pdf.section("Red Flags")
    for item in response.red_flags:
        pdf.wrapped(item, bullet="•")

    pdf.section("Escalation")
    for item in response.escalation:
        pdf.wrapped(item, bullet="•")

    pdf.section("Monitoring")
    for item in response.monitoring:
        pdf.wrapped(item, bullet="•")

    pdf.section("Evidence")
    if response.evidence:
        for ev in response.evidence:
            pdf.wrapped(f"[{ev.source_id}] {ev.title}", font="Helvetica-Bold", bullet="•")
            pdf.wrapped(ev.note, bullet="-")
            if ev.citation_text:
                pdf.wrapped(f'Citation: "{ev.citation_text}"', bullet="-")
    else:
        pdf.wrapped("No evidence available.", bullet="•")

    pdf.section("Disclaimer")
    pdf.wrapped(response.disclaimer)

    pdf.save()
    return path

@router.post("/decision", response_model=DecisionResponse)
def decision(payload: DecisionRequest) -> DecisionResponse:
    request_id = str(uuid.uuid4())
    protocol_key, protocol, confidence = select_protocol(payload.query, payload.context)
    return build_decision_response(request_id, protocol_key, protocol, confidence, payload.query, payload.context)

@router.post("/export-pdf", response_model=ExportPDFResponse)
def export_protocol_pdf(payload: DecisionRequest) -> ExportPDFResponse:
    request_id = str(uuid.uuid4())
    protocol_key, protocol, confidence = select_protocol(payload.query, payload.context)
    response = build_decision_response(request_id, protocol_key, protocol, confidence, payload.query, payload.context)
    pdf_path = export_pdf(response)
    return ExportPDFResponse(
        request_id=request_id,
        filename=os.path.basename(pdf_path),
        pdf_path=pdf_path,
    )

@router.post("/log-case", response_model=LogCaseResponse)
def log_case(payload: LogCaseRequest) -> LogCaseResponse:
    if payload.protocol_key not in PROTOCOL_LIBRARY:
        raise HTTPException(status_code=400, detail="Unknown protocol_key.")

    case = LoggedCase(
        case_id=str(uuid.uuid4()),
        logged_at_utc=now_utc_iso(),
        clinic_id=payload.clinic_id,
        clinician_id=payload.clinician_id,
        protocol_key=payload.protocol_key,
        region=payload.region,
        procedure=payload.procedure,
        product_type=payload.product_type,
        symptoms=payload.symptoms,
        outcome=payload.outcome,
    )
    CASE_STORE.append(case)
    return LogCaseResponse(status="ok", case_id=case.case_id)

@router.get("/stats", response_model=DatasetStatsResponse)
def stats() -> DatasetStatsResponse:
    by_protocol = Counter(case.protocol_key for case in CASE_STORE if case.protocol_key)
    by_region = Counter(case.region for case in CASE_STORE if case.region)
    by_procedure = Counter(case.procedure for case in CASE_STORE if case.procedure)

    return DatasetStatsResponse(
        total_cases=len(CASE_STORE),
        by_protocol=dict(by_protocol),
        by_region=dict(by_region),
        by_procedure=dict(by_procedure),
    )
