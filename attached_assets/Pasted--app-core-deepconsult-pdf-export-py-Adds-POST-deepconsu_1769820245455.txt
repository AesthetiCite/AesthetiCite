# -----------------------------
# 3) New endpoint: from-query → DeepConsult → PDF
# -----------------------------
@router.post("/deepconsult/pdf/from-query")
async def deepconsult_pdf_from_query(req: DeepConsultPDFQueryRequest):
    # 1) Run DeepConsult
    result = deepconsult_run(req.question, req.filters, req.max_sources)

    if result.get("refused"):
        raise HTTPException(
            status_code=400,
            detail=f"DeepConsult refused; cannot export PDF. Reason: {result.get('reason','insufficient evidence')}"
        )

    # 2) Optionally attach disagreement
    if req.include_disagreement:
        # If your deepconsult_run already adds disagreement, keep it.
        if not result.get("disagreement"):
            chunks = retrieve_adapter(req.question, req.filters, min(req.max_sources, 40))
            # If your disagreement_detector returns pydantic, convert to dict; else keep as dict
            dis = disagreement_detector(req.question, chunks)
            # Normalize to dict
            if hasattr(dis, "model_dump"):
                dis = dis.model_dump()
            result["disagreement"] = dis

    # 3) Build PDF (calls your existing function in this module)
    try:
        pdf_bytes = build_pdf_bytes_from_deepconsult(result)
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"PDF export failed: {e}")

    # 4) Stream PDF back
    return StreamingResponse(
        io.BytesIO(pdf_bytes),
        media_type="application/pdf",
        headers={"Content-Disposition": "inline; filename=deepconsult_report.pdf"}
    )
