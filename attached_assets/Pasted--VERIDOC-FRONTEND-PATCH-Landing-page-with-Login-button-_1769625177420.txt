const API_BASE = process.env.NEXT_PUBLIC_API_BASE || "http://localhost:8000";

export function getToken() {
  if (typeof window === "undefined") return null;
  return localStorage.getItem("veridoc_token");
}
export function setToken(token) {
  if (typeof window === "undefined") return;
  localStorage.setItem("veridoc_token", token);
}
export function clearToken() {
  if (typeof window === "undefined") return;
  localStorage.removeItem("veridoc_token");
}

async function readJson(res) {
  const text = await res.text();
  try { return JSON.parse(text); } catch { return { raw: text }; }
}

export async function requestAccess(full_name, email, practitioner_id) {
  const res = await fetch(`${API_BASE}/auth/request-access`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ full_name, email, practitioner_id })
  });
  const data = await readJson(res);
  if (!res.ok) throw new Error(data?.detail || data?.raw || "Request failed");
  return data;
}

export async function setPassword(token, password) {
  const res = await fetch(`${API_BASE}/auth/set-password`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ token, password })
  });
  const data = await readJson(res);
  if (!res.ok) throw new Error(data?.detail || data?.raw || "Set password failed");
  return data;
}

export async function login(email, password) {
  const res = await fetch(`${API_BASE}/auth/login`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ email, password })
  });
  const data = await readJson(res);
  if (!res.ok) throw new Error(data?.detail || data?.raw || "Login failed");
  return data.access_token;
}

export async function me(token) {
  const res = await fetch(`${API_BASE}/auth/me`, {
    headers: { Authorization: `Bearer ${token}` }
  });
  const data = await readJson(res);
  if (!res.ok) throw new Error(data?.detail || data?.raw || "Auth check failed");
  return data;
}

export async function ask(token, question, domain="medicine", mode="clinic") {
  const res = await fetch(`${API_BASE}/ask`, {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      Authorization: `Bearer ${token}`
    },
    body: JSON.stringify({ question, domain, mode })
  });
  const data = await readJson(res);
  if (!res.ok) throw new Error(data?.detail || data?.raw || "Ask failed");
  return data;
}

import { useState } from "react";
import Layout from "../components/Layout";
import Link from "next/link";
import { requestAccess } from "../lib/api";

export default function RequestAccessPage() {
  const [fullName, setFullName] = useState("");
  const [email, setEmail] = useState("");
  const [practitionerId, setPractitionerId] = useState("");
  const [busy, setBusy] = useState(false);
  const [msg, setMsg] = useState("");
  const [err, setErr] = useState("");

  async function submit(e) {
    e.preventDefault();
    setErr(""); setMsg(""); setBusy(true);
    try {
      const r = await requestAccess(fullName, email, practitionerId);
      setMsg(r.message || "Check your email to set your password.");
    } catch (e2) {
      setErr(e2.message || "Request failed");
    } finally {
      setBusy(false);
    }
  }

  return (
    <Layout hideNav>
      <div className="h1">Get free access</div>
      <p className="p">Enter your details. We’ll email you a secure link to set your password.</p>

      <div className="card">
        <form onSubmit={submit}>
          <label className="label">Full name</label>
          <input value={fullName} onChange={(e)=>setFullName(e.target.value)} placeholder="Dr. Firstname Lastname" />

          <div style={{ height: 10 }} />

          <label className="label">Professional email</label>
          <input value={email} onChange={(e)=>setEmail(e.target.value)} placeholder="doctor@clinic.com" />

          <div style={{ height: 10 }} />

          <label className="label">Practitioner number</label>
          <input value={practitionerId} onChange={(e)=>setPractitionerId(e.target.value)} placeholder="License / Order number" />

          <div style={{ height: 12 }} />
          <button className="btn btnPrimary" disabled={busy}>
            {busy ? "Sending email…" : "Send password link"}
          </button>
        </form>

        {msg ? <div style={{ marginTop: 12 }} className="success">{msg}</div> : null}
        {err ? <div style={{ marginTop: 12 }} className="error">{err}</div> : null}

        <div className="hr" />
        <Link href="/login"><span className="badge">Already have a password? Log in</span></Link>
      </div>
    </Layout>
  );
}

import { useRouter } from "next/router";
import { useEffect, useState } from "react";
import Layout from "../components/Layout";
import { setPassword } from "../lib/api";
import Link from "next/link";

export default function SetPasswordPage() {
  const router = useRouter();
  const [token, setToken] = useState("");
  const [pw, setPw] = useState("");
  const [busy, setBusy] = useState(false);
  const [msg, setMsg] = useState("");
  const [err, setErr] = useState("");

  useEffect(() => {
    if (router.query.token) setToken(String(router.query.token));
  }, [router.query.token]);

  async function submit(e) {
    e.preventDefault();
    setErr(""); setMsg(""); setBusy(true);
    try {
      const r = await setPassword(token, pw);
      setMsg(r.message || "Password set. You can now log in.");
    } catch (e2) {
      setErr(e2.message || "Failed");
    } finally {
      setBusy(false);
    }
  }

  return (
    <Layout hideNav>
      <div className="h1">Set your password</div>
      <p className="p">Use the secure link sent to your email. This link expires after a short time.</p>

      <div className="card">
        <form onSubmit={submit}>
          <label className="label">New password</label>
          <input type="password" value={pw} onChange={(e)=>setPw(e.target.value)} placeholder="Minimum 8 characters" />
          <div style={{ height: 12 }} />
          <button className="btn btnPrimary" disabled={busy || !token}>
            {busy ? "Saving…" : "Set password"}
          </button>
        </form>

        {msg ? <div style={{ marginTop: 12 }} className="success">{msg}</div> : null}
        {err ? <div style={{ marginTop: 12 }} className="error">{err}</div> : null}

        <div className="hr" />
        <Link href="/login"><span className="badge">Go to login</span></Link>
      </div>
    </Layout>
  );
}
