from dataclasses import dataclass
from typing import Optional

HIGH_RISK_KEYWORDS = [
    "blindness", "vision loss", "stroke", "necrosis", "ischemia",
    "nécrose", "ischémie", "cécité", "perte de vision",
    "vascular occlusion", "occlusion vasculaire",
]

@dataclass
class SafetyResult:
    allowed: bool
    refusal_reason: Optional[str] = None
    escalation_note: Optional[str] = None

def safety_screen(question: str) -> SafetyResult:
    q = question.strip().lower()

    if len(q) < 5:
        return SafetyResult(False, "Question too short to assess safely.")

    escalation = None
    if any(k in q for k in HIGH_RISK_KEYWORDS):
        escalation = (
            "High-risk complication context detected. "
            "If any red flags are present (e.g., visual symptoms, rapidly worsening pain, expanding ischemia), "
            "initiate urgent escalation per local emergency pathways. "
            "This system provides evidence summaries and does not replace clinical judgment."
        )

    return SafetyResult(True, escalation_note=escalation)
EOF

from __future__ import annotations
from functools import lru_cache
from sentence_transformers import SentenceTransformer
import numpy as np

MODEL_NAME = "sentence-transformers/all-mpnet-base-v2"

@lru_cache(maxsize=1)
def _model() -> SentenceTransformer:
    return SentenceTransformer(MODEL_NAME)

def embed_text(text: str) -> list[float]:
    vec = _model().encode([text], normalize_embeddings=True, show_progress_bar=False)[0]
    return [float(x) for x in np.asarray(vec, dtype=np.float32).tolist()]
EOF

from __future__ import annotations
from typing import List, Tuple
from app.core.config import settings

def chunk_text(text: str) -> List[Tuple[str, str]]:
    """
    Returns list of (page_or_section, chunk_text).
    This basic chunker labels chunks sequentially; you can upgrade later to use real headings/pages.
    """
    clean = " ".join(text.split())
    if not clean:
        return []

    size = settings.CHUNK_SIZE_CHARS
    overlap = settings.CHUNK_OVERLAP_CHARS
    chunks = []
    i = 0
    idx = 0
    n = len(clean)
    while i < n:
        j = min(i + size, n)
        chunk = clean[i:j].strip()
        if chunk:
            chunks.append((f"chunk_{idx}", chunk))
            idx += 1
        if j == n:
            break
        i = max(0, j - overlap)
    return chunks
EOF

from __future__ import annotations
import pdfplumber

def extract_text_from_pdf(pdf_path: str) -> str:
    out = []
    with pdfplumber.open(pdf_path) as pdf:
        for page in pdf.pages:
            txt = page.extract_text() or ""
            if txt.strip():
                out.append(txt)
    return "\n\n".join(out)
EOF

from __future__ import annotations
from typing import List, Dict, Optional
from sqlalchemy import text
from sqlalchemy.orm import Session
from app.rag.embedder import embed_text

SQL = """
SELECT
  d.source_id,
  d.title,
  d.year,
  d.organization_or_journal,
  d.document_type,
  d.domain,
  c.page_or_section,
  COALESCE(c.evidence_level, d.document_type) AS evidence_level,
  c.text,
  (c.embedding <=> :qvec) AS distance
FROM chunks c
JOIN documents d ON d.id = c.document_id
WHERE d.status = 'active'
  AND c.embedding IS NOT NULL
  AND (:domain IS NULL OR d.domain = :domain)
ORDER BY c.embedding <=> :qvec
LIMIT :k;
"""

def retrieve_db(db: Session, question: str, domain: Optional[str] = None, k: int = 6) -> List[Dict]:
    qvec = embed_text(question)
    rows = db.execute(text(SQL), {"qvec": qvec, "domain": domain, "k": int(k)}).mappings().all()
    results: List[Dict] = []
    for r in rows:
        results.append({
            "source_id": r["source_id"],
            "title": r["title"],
            "year": r["year"],
            "organization_or_journal": r["organization_or_journal"],
            "page_or_section": r["page_or_section"],
            "evidence_level": r["evidence_level"],
            "text": r["text"],
            "distance": float(r["distance"]) if r["distance"] is not None else None,
        })
    return results
EOF

from typing import List, Dict
from app.schemas.ask import Citation

def to_citations(chunks: List[Dict]) -> List[Citation]:
    citations: List[Citation] = []
    seen = set()
    for c in chunks:
        key = (c.get("source_id"), c.get("page_or_section"))
        if key in seen:
            continue
        seen.add(key)
        citations.append(Citation(
            source_id=c.get("source_id", ""),
            title=c.get("title", "Unknown source"),
            year=c.get("year"),
            organization_or_journal=c.get("organization_or_journal"),
            page_or_section=c.get("page_or_section"),
            evidence_level=c.get("evidence_level"),
        ))
    return citations

from __future__ import annotations
from typing import List, Dict, Optional
from app.core.config import settings

def build_answer(
    question: str,
    mode: str,
    domain: str,
    retrieved: List[Dict],
    escalation_note: Optional[str] = None,
) -> str:
    """
    Evidence-bounded answer template.
    Day-to-day: keep conservative. Never invent protocols; reflect only what's retrieved.
    """
    sources_lines = []
    for r in retrieved:
        sources_lines.append(
            f"- {r.get('title')} ({r.get('year', 'n.d.')}) — {r.get('evidence_level','source')} — {r.get('page_or_section','')}".strip()
        )
    sources_block = "\n".join(sources_lines) if sources_lines else "- None"

    # Keep content conservative by design:
    summary = (
        "Clinical summary\n"
        f"- Domain: {domain}\n"
        "- This answer is generated strictly from retrieved evidence; if evidence is insufficient, the system refuses.\n"
    )

    actions = (
        "Recommended actions (evidence-bounded)\n"
        "- Follow the diagnostic and management principles described in the retrieved sources and your local governance.\n"
        "- If the scenario involves a time-critical complication, prioritize urgent escalation pathways noted in authoritative guidance.\n"
    )

    if mode == "deep_dive":
        actions += "- Deep-dive mode: review the cited sources directly and compare consensus statements versus reviews when applicable.\n"

    vigilance = (
        "Red flags / escalation\n"
        "- If severe or rapidly evolving symptoms are present, escalate urgently per local emergency protocols.\n"
    )
    if escalation_note:
        vigilance += f"- Note: {escalation_note}\n"

    evidence = (
        "Evidence level\n"
        "- Evidence may be based on guidelines/consensus/reviews; strength varies by topic and publication recency.\n"
    )

    limits = (
        "Limits and uncertainty\n"
        "- Veridoc does not replace clinical judgment and does not provide step-by-step emergency dosing unless explicitly present in retrieved sources.\n"
        "- If retrieved evidence is incomplete/outdated, the correct action is to refine sources or consult local protocols.\n"
    )

    sources = "Sources\n" + sources_block

    answer = "\n".join([summary, actions, vigilance, evidence, sources, limits])
    return answer[: settings.MAX_CONTEXT_CHARS]
EOF

from __future__ import annotations
import os
import uuid
from fastapi import APIRouter, Depends, UploadFile, File, Form
from sqlalchemy.orm import Session
from sqlalchemy import text

from app.core.config import settings
from app.core.admin_auth import require_admin
from app.db.session import get_db
from app.rag.embedder import embed_text
from ingestion.pdf_extract import extract_text_from_pdf
from ingestion.chunker import chunk_text
from app.schemas.admin import IngestMetadata

router = APIRouter(prefix="/admin", tags=["admin"])

@router.post("/ingest_pdf")
def ingest_pdf(
    # Admin auth
    _: bool = Depends(require_admin),
    # Metadata fields (sent as form fields)
    metadata_json: str = Form(...),
    pdf: UploadFile = File(...),
    db: Session = Depends(get_db),
):
    """
    Upload a PDF + metadata JSON string (IngestMetadata schema).
    Stores the PDF, extracts text, chunks, embeds, inserts into documents/chunks.
    """
    # Parse metadata
    meta = IngestMetadata.model_validate_json(metadata_json)

    # Save PDF
    os.makedirs(settings.UPLOAD_DIR, exist_ok=True)
    file_id = str(uuid.uuid4())
    safe_name = pdf.filename.replace("/", "_").replace("\\", "_")
    saved_path = os.path.join(settings.UPLOAD_DIR, f"{file_id}_{safe_name}")

    with open(saved_path, "wb") as f:
        f.write(pdf.file.read())

    # Extract + chunk
    full_text = extract_text_from_pdf(saved_path)
    chunks = chunk_text(full_text)

    if not chunks:
        return {"ok": False, "reason": "No extractable text found in PDF.", "file_path": saved_path}

    # Upsert document
    doc_row = db.execute(text("""
      INSERT INTO documents (source_id, title, authors, organization_or_journal, year, document_type, domain, version, status, url, file_path)
      VALUES (:source_id, :title, :authors, :org, :year, :dtype, :domain, :version, 'active', :url, :file_path)
      ON CONFLICT (source_id) DO UPDATE SET
        title=EXCLUDED.title,
        authors=EXCLUDED.authors,
        organization_or_journal=EXCLUDED.organization_or_journal,
        year=EXCLUDED.year,
        document_type=EXCLUDED.document_type,
        domain=EXCLUDED.domain,
        version=EXCLUDED.version,
        status='active',
        url=EXCLUDED.url,
        file_path=EXCLUDED.file_path,
        updated_at=now()
      RETURNING id;
    """), {
        "source_id": meta.source_id,
        "title": meta.title,
        "authors": meta.authors,
        "org": meta.organization_or_journal,
        "year": meta.year,
        "dtype": meta.document_type,
        "domain": meta.domain,
        "version": meta.version,
        "url": meta.url,
        "file_path": saved_path,
    }).scalar_one()

    # Replace chunks for idempotency
    db.execute(text("DELETE FROM chunks WHERE document_id = :doc_id;"), {"doc_id": doc_row})

    # Insert embedded chunks
    for idx, (pos, txt) in enumerate(chunks):
        vec = embed_text(txt)
        db.execute(text("""
          INSERT INTO chunks (document_id, chunk_index, text, page_or_section, evidence_level, embedding)
          VALUES (:doc_id, :idx, :text, :pos, NULL, :emb);
        """), {
            "doc_id": doc_row,
            "idx": idx,
            "text": txt,
            "pos": pos,
            "emb": vec,
        })

    db.execute(text("ANALYZE;"))
    db.commit()

    return {
        "ok": True,
        "source_id": meta.source_id,
        "stored_pdf": saved_path,
        "chunks_inserted": len(chunks),
        "domain": meta.domain,
        "document_type": meta.document_type,
    }
EOF

from __future__ import annotations
import time
import uuid
from fastapi import APIRouter, Depends, Request
from sqlalchemy.orm import Session
from sqlalchemy import text

from app.schemas.ask import AskRequest, AskResponse
from app.core.config import settings
from app.core.safety import safety_screen
from app.db.session import get_db
from app.rag.retriever import retrieve_db
from app.rag.citations import to_citations
from app.rag.answer import build_answer

router = APIRouter(prefix="", tags=["ask"])

@router.post("/ask", response_model=AskResponse)
def ask(payload: AskRequest, request: Request, db: Session = Depends(get_db)) -> AskResponse:
    request_id = str(uuid.uuid4())
    t0 = time.perf_counter()

    safety = safety_screen(payload.question)
    if not safety.allowed:
        latency_ms = int((time.perf_counter() - t0) * 1000)
        # log query
        db.execute(text("""
          INSERT INTO queries (id, user_id, question, domain, mode, latency_ms, citations_count, refusal, refusal_reason)
          VALUES (:id, :user_id, :q, :domain, :mode, :lat, 0, true, :reason);
        """), {
            "id": request_id,
            "user_id": request.client.host if request.client else None,
            "q": payload.question,
            "domain": payload.domain,
            "mode": payload.mode,
            "lat": latency_ms,
            "reason": safety.refusal_reason or "Request refused by safety policy.",
        })
        db.commit()
        return AskResponse(
            answer="",
            citations=[],
            refusal=True,
            refusal_reason=safety.refusal_reason or "Request refused by safety policy.",
            request_id=request_id,
            latency_ms=latency_ms,
        )

    retrieved = retrieve_db(db=db, question=payload.question, domain=payload.domain, k=6)
    citations = to_citations(retrieved)

    if len(citations) < settings.MIN_CITATIONS_REQUIRED:
        latency_ms = int((time.perf_counter() - t0) * 1000)
        db.execute(text("""
          INSERT INTO queries (id, user_id, question, domain, mode, latency_ms, citations_count, refusal, refusal_reason)
          VALUES (:id, :user_id, :q, :domain, :mode, :lat, 0, true, :reason);
        """), {
            "id": request_id,
            "user_id": request.client.host if request.client else None,
            "q": payload.question,
            "domain": payload.domain,
            "mode": payload.mode,
            "lat": latency_ms,
            "reason": "Insufficient evidence retrieved to answer with citations.",
        })
        db.commit()
        return AskResponse(
            answer="",
            citations=[],
            refusal=True,
            refusal_reason="Insufficient evidence retrieved to answer with citations.",
            request_id=request_id,
            latency_ms=latency_ms,
        )

    answer = build_answer(
        question=payload.question,
        mode=payload.mode,
        domain=payload.domain,
        retrieved=retrieved,
        escalation_note=safety.escalation_note,
    )

    latency_ms = int((time.perf_counter() - t0) * 1000)

    # log query + answer snapshot
    db.execute(text("""
      INSERT INTO queries (id, user_id, question, domain, mode, latency_ms, citations_count, refusal, refusal_reason)
      VALUES (:id, :user_id, :q, :domain, :mode, :lat, :cc, false, NULL);
    """), {
        "id": request_id,
        "user_id": request.client.host if request.client else None,
        "q": payload.question,
        "domain": payload.domain,
        "mode": payload.mode,
        "lat": latency_ms,
        "cc": len(citations),
    })
    db.execute(text("""
      INSERT INTO answers (query_id, answer_text, citations_json)
      VALUES (:qid, :ans, :cjson);
    """), {
        "qid": request_id,
        "ans": answer,
        "cjson": [c.model_dump() for c in citations],
    })
    db.commit()

    return AskResponse(
        answer=answer,
        citations=citations,
        refusal=False,
        request_id=request_id,
        latency_ms=latency_ms,
    )
EOF
