# aestheticite_complication_engine.py
#
# One-file production-ready starter for the AesthetiCite Clinical Complication Protocol Engine
#
# Features included:
# - POST /api/complications/protocol         -> structured complication protocol JSON
# - POST /api/complications/print-view       -> HTML print-ready emergency protocol
# - POST /api/complications/feedback         -> clinician feedback capture
# - protocol library with multiple complications
# - risk scoring and urgency classification
# - symptom/context-aware matching
# - retrieval adapter hook for RAG / citation system
# - audit logging to JSONL file
# - revision/version metadata
#
# How to use:
# 1) save as aestheticite_complication_engine.py
# 2) in your FastAPI app:
#       from aestheticite_complication_engine import router as complication_router
#       app.include_router(complication_router)
#
# Optional:
# - Replace DummyEvidenceRetriever with your real citation retrieval layer
# - Expand PROTOCOL_LIBRARY with your validated master protocols

SeverityLevel = Literal["low", "moderate", "high", "critical"]
UrgencyLevel = Literal["routine", "same_day", "urgent", "immediate"]
EvidenceStrength = Literal["limited", "moderate", "strong"]


class ClinicalContext(BaseModel):
    region: Optional[str] = None
    product_type: Optional[str] = None
    symptoms: List[str] = Field(default_factory=list)
    time_since_injection_minutes: Optional[int] = None
    visual_symptoms: Optional[bool] = None
    skin_color_change: Optional[str] = None
    pain_score_10: Optional[int] = Field(default=None, ge=0, le=10)
    tenderness: Optional[bool] = None
    warmth: Optional[bool] = None
    erythema: Optional[bool] = None
    fluctuance: Optional[bool] = None
    drainage: Optional[bool] = None
    fever: Optional[bool] = None
    capillary_refill_delayed: Optional[bool] = None
    filler_confirmed_ha: Optional[bool] = None
    free_text: Optional[str] = None


class ProtocolRequest(BaseModel):
    query: str = Field(..., min_length=3)
    context: Optional[ClinicalContext] = None
    clinician_id: Optional[str] = None
    clinic_id: Optional[str] = None
    mode: Literal["decision_support", "emergency", "teaching"] = "decision_support"


class EvidenceItem(BaseModel):
    source_id: str
    title: str
    note: str
    citation_text: Optional[str] = None
    url: Optional[str] = None
    source_type: Optional[str] = None
    relevance_score: Optional[float] = None


class ProtocolStep(BaseModel):
    step_number: int
    action: str
    rationale: str
    priority: Literal["primary", "secondary"] = "primary"


class DoseGuidance(BaseModel):
    substance: str
    recommendation: str
    notes: str


class RiskAssessment(BaseModel):
    risk_score: int = Field(..., ge=0, le=100)
    severity: SeverityLevel
    urgency: UrgencyLevel
    likely_time_critical: bool
    evidence_strength: EvidenceStrength


class ProtocolResponse(BaseModel):
    request_id: str
    engine_version: str
    protocol_revision: str
    generated_at_utc: str
    matched_protocol_key: str
    matched_protocol_name: str
    confidence: float
    risk_assessment: RiskAssessment
    clinical_summary: str
    immediate_actions: List[ProtocolStep]
    dose_guidance: List[DoseGuidance]
    red_flags: List[str]
    escalation: List[str]
    monitoring: List[str]
    limitations: List[str]
    follow_up_questions: List[str]
    evidence: List[EvidenceItem]

def now_utc_iso() -> str:
    return datetime.now(timezone.utc).isoformat()


def normalize(text: str) -> str:
    text = text.lower().strip()
    text = re.sub(r"\s+", " ", text)
    return text


def tokenize(text: str) -> List[str]:
    return re.findall(r"[a-z0-9]+", normalize(text))


def safe_write_jsonl(path: str, record: Dict[str, Any]) -> None:
    with open(path, "a", encoding="utf-8") as f:
        f.write(json.dumps(record, ensure_ascii=False) + "\n")


def html_list(items: List[str]) -> str:
    if not items:
        return "<ul><li>None</li></ul>"
    return "<ul>" + "".join(f"<li>{html.escape(item)}</li>" for item in items) + "</ul>"


def html_steps(steps: List[ProtocolStep]) -> str:
    if not steps:
        return "<ol><li>No steps available</li></ol>"
    return (
        "<ol>"
        + "".join(
            f"<li><strong>{html.escape(step.action)}</strong><br><span>{html.escape(step.rationale)}</span></li>"
            for step in steps
        )
        + "</ol>"
    )


def html_doses(doses: List[DoseGuidance]) -> str:
    if not doses:
        return "<ul><li>No dose guidance available</li></ul>"
    return (
        "<ul>"
        + "".join(
            f"<li><strong>{html.escape(d.substance)}:</strong> {html.escape(d.recommendation)}"
            f"<br><span>{html.escape(d.notes)}</span></li>"
            for d in doses
        )
        + "</ul>"
    )


def html_evidence(evidence: List[EvidenceItem]) -> str:
    if not evidence:
        return "<ul><li>No evidence retrieved</li></ul>"
    rows = []
    for item in evidence:
        citation_text = html.escape(item.citation_text or "")
        title = html.escape(item.title)
        note = html.escape(item.note)
        source_id = html.escape(item.source_id)
        rows.append(
            f"<li><strong>[{source_id}] {title}</strong><br>{note}"
            + (f"<br><em>{citation_text}</em>" if citation_text else "")
            + "</li>"
        )
    return "<ul>" + "".join(rows) + "</ul>"

def score_protocol(query: str, context: Optional[ClinicalContext], protocol: Dict[str, Any]) -> float:
    score = 0.0
    q = normalize(query)

    for kw in protocol["keywords"]:
        if kw in q:
            score += 1.0

    if context:
        symptom_blob = normalize(" ".join(context.symptoms))
        free_blob = normalize(context.free_text or "")
        merged = f"{symptom_blob} {free_blob}".strip()

        for kw in protocol["keywords"]:
            if kw in merged:
                score += 0.5

        if context.visual_symptoms and "vascular occlusion" in protocol["name"].lower():
            score += 3.0

        if context.capillary_refill_delayed and "vascular occlusion" in protocol["name"].lower():
            score += 2.0

        if context.filler_confirmed_ha:
            if "hyaluronic acid filler" in protocol["name"].lower() or "ha filler" in protocol["name"].lower():
                score += 1.25

        if context.product_type and "hyaluronic" in context.product_type.lower():
            if "hyaluronic acid filler" in protocol["name"].lower() or "ha filler" in protocol["name"].lower():
                score += 1.0

        if context.skin_color_change:
            scc = context.skin_color_change.lower()
            if scc in ("blanching", "mottling", "dusky", "violaceous") and "vascular occlusion" in protocol["name"].lower():
                score += 2.0
            if ("blue" in scc or "bluish" in scc or "blue-gray" in scc) and "tyndall" in protocol["name"].lower():
                score += 2.0

        if any([context.tenderness, context.warmth, context.erythema, context.fluctuance, context.drainage, context.fever]):
            if "infection" in protocol["name"].lower():
                score += 2.0
            if "nodules" in protocol["name"].lower():
                score += 1.0

        if context.time_since_injection_minutes is not None:
            if context.time_since_injection_minutes <= 60 and "vascular occlusion" in protocol["name"].lower():
                score += 1.5

        if context.pain_score_10 is not None and context.pain_score_10 >= 7 and "vascular occlusion" in protocol["name"].lower():
            score += 1.5

    return score


def select_protocol(query: str, context: Optional[ClinicalContext]) -> Tuple[str, Dict[str, Any], float]:
    best_key: Optional[str] = None
    best_protocol: Optional[Dict[str, Any]] = None
    best_score = -1.0

    for key, protocol in PROTOCOL_LIBRARY.items():
        score = score_protocol(query, context, protocol)
        if score > best_score:
            best_key = key
            best_protocol = protocol
            best_score = score

    if best_key is None or best_protocol is None or best_score <= 0:
        raise HTTPException(
            status_code=404,
            detail="No matching complication protocol found. Expand the protocol library or improve matching."
        )

    confidence = min(0.99, round(0.45 + best_score * 0.06, 2))
    return best_key, best_protocol, confidence

def compute_risk_assessment(protocol_key: str, protocol: Dict[str, Any], context: Optional[ClinicalContext]) -> RiskAssessment:
    base_map = {
        "low": 20,
        "moderate": 50,
        "high": 75,
        "critical": 92,
    }
    urgency_map = {
        "routine": "routine",
        "same_day": "same_day",
        "urgent": "urgent",
        "immediate": "immediate",
    }

    severity = protocol["base_severity"]
    urgency = urgency_map[protocol["base_urgency"]]
    score = base_map[severity]

    if context:
        if context.visual_symptoms:
            score += 8
            severity = "critical"
            urgency = "immediate"

        if context.capillary_refill_delayed:
            score += 5
            if protocol_key == "vascular_occlusion_ha_filler":
                severity = "critical"
                urgency = "immediate"

        if context.pain_score_10 is not None:
            if context.pain_score_10 >= 8:
                score += 4
            elif context.pain_score_10 >= 5:
                score += 2

        if any([context.fever, context.fluctuance, context.drainage]):
            if protocol_key == "infection_or_biofilm_after_filler":
                score += 6
                severity = "high"
                urgency = "urgent"

        if any([context.erythema, context.warmth, context.tenderness]):
            if protocol_key in ("infection_or_biofilm_after_filler", "filler_nodules_inflammatory_or_noninflammatory"):
                score += 3

    score = max(0, min(100, score))
    likely_time_critical = urgency in ("urgent", "immediate")

    return RiskAssessment(
        risk_score=score,
        severity=severity,
        urgency=urgency,
        likely_time_critical=likely_time_critical,
        evidence_strength=protocol["evidence_strength"],
    )

def build_steps(protocol: Dict[str, Any]) -> List[ProtocolStep]:
    return [
        ProtocolStep(
            step_number=i + 1,
            action=s["action"],
            rationale=s["rationale"],
            priority=s.get("priority", "primary"),
        )
        for i, s in enumerate(protocol["steps"])
    ]


def build_doses(protocol: Dict[str, Any]) -> List[DoseGuidance]:
    return [
        DoseGuidance(
            substance=d["substance"],
            recommendation=d["recommendation"],
            notes=d["notes"],
        )
        for d in protocol["dose_guidance"]
    ]


def build_response(
    request_id: str,
    protocol_key: str,
    protocol: Dict[str, Any],
    confidence: float,
    context: Optional[ClinicalContext],
    query: str,
) -> ProtocolResponse:
    evidence = evidence_retriever.retrieve(protocol_key=protocol_key, query=query, context=context)
    risk_assessment = compute_risk_assessment(protocol_key, protocol, context)

    red_flags = list(protocol["red_flags"])
    escalation = list(protocol["escalation"])
    monitoring = list(protocol["monitoring"])
    follow_up = list(protocol["follow_up_questions"])

    if context and context.visual_symptoms:
        visual_flag = "Visual disturbance, blurred vision, diplopia, vision loss"
        if visual_flag not in red_flags:
            red_flags.insert(0, visual_flag)

        ocular_escalation = "Immediate ophthalmology / emergency referral for any visual symptom"
        if ocular_escalation not in escalation:
            escalation.insert(0, ocular_escalation)

    return ProtocolResponse(
        request_id=request_id,
        engine_version=ENGINE_VERSION,
        protocol_revision=PROTOCOL_REVISION,
        generated_at_utc=now_utc_iso(),
        matched_protocol_key=protocol_key,
        matched_protocol_name=protocol["name"],
        confidence=confidence,
        risk_assessment=risk_assessment,
        clinical_summary=protocol["summary"],
        immediate_actions=build_steps(protocol),
        dose_guidance=build_doses(protocol),
        red_flags=red_flags,
        escalation=escalation,
        monitoring=monitoring,
        limitations=list(protocol["limitations"]),
        follow_up_questions=follow_up,
        evidence=evidence,
        disclaimer=(
            "This output is clinical decision support and not a substitute for clinician judgment. "
            "Escalate immediately for visual symptoms, progressive ischemia, suspected infection, "
            "systemic illness, or diagnostic uncertainty."
        ),
    )

def log_protocol_request(payload: ProtocolRequest, response: ProtocolResponse) -> None:
    record = {
        "event_type": "protocol_generated",
        "logged_at_utc": now_utc_iso(),
        "request_id": response.request_id,
        "clinician_id": payload.clinician_id,
        "clinic_id": payload.clinic_id,
        "mode": payload.mode,
        "query": payload.query,
        "context": payload.context.model_dump() if payload.context else None,
        "matched_protocol_key": response.matched_protocol_key,
        "matched_protocol_name": response.matched_protocol_name,
        "confidence": response.confidence,
        "risk_score": response.risk_assessment.risk_score,
        "severity": response.risk_assessment.severity,
        "urgency": response.risk_assessment.urgency,
        "engine_version": response.engine_version,
        "protocol_revision": response.protocol_revision,
    }
    safe_write_jsonl(AUDIT_LOG_PATH, record)


def log_feedback(payload: FeedbackRequest) -> None:
    record = {
        "event_type": "protocol_feedback",
        "logged_at_utc": now_utc_iso(),
        "request_id": payload.request_id,
        "clinician_id": payload.clinician_id,
        "clinic_id": payload.clinic_id,
        "rating": payload.rating,
        "was_useful": payload.was_useful,
        "comment": payload.comment,
        "selected_protocol_key": payload.selected_protocol_key,
    }
    safe_write_jsonl(FEEDBACK_LOG_PATH, record)

@router.post("/protocol", response_model=ProtocolResponse)
def generate_complication_protocol(payload: ProtocolRequest) -> ProtocolResponse:
    request_id = str(uuid.uuid4())
    protocol_key, protocol, confidence = select_protocol(payload.query, payload.context)
    response = build_response(
        request_id=request_id,
        protocol_key=protocol_key,
        protocol=protocol,
        confidence=confidence,
        context=payload.context,
        query=payload.query,
    )
    log_protocol_request(payload, response)
    return response


@router.post("/print-view", response_model=PrintViewResponse)
def generate_print_view(payload: ProtocolRequest) -> PrintViewResponse:
    request_id = str(uuid.uuid4())
    protocol_key, protocol, confidence = select_protocol(payload.query, payload.context)
    response = build_response(
        request_id=request_id,
        protocol_key=protocol_key,
        protocol=protocol,
        confidence=confidence,
        context=payload.context,
        query=payload.query,
    )
    log_protocol_request(payload, response)
    return PrintViewResponse(
        request_id=request_id,
        html=build_print_html(response),
    )


@router.post("/feedback", response_model=FeedbackResponse)
def submit_feedback(payload: FeedbackRequest) -> FeedbackResponse:
    log_feedback(payload)
    return FeedbackResponse(status="ok", request_id=payload.request_id)

def run_complication_engine(query: str, context: Optional[Dict[str, Any]] = None) -> Dict[str, Any]:
    ctx = ClinicalContext(**context) if context else None
    request_id = str(uuid.uuid4())
    protocol_key, protocol, confidence = select_protocol(query, ctx)
    response = build_response(
        request_id=request_id,
        protocol_key=protocol_key,
        protocol=protocol,
        confidence=confidence,
        context=ctx,
        query=query,
    )
    return response.model_dump()
