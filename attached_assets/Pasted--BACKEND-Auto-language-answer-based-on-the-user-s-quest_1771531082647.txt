def detect_lang_from_text(q: str) -> str:
    """
    Heuristic language detection (no external libs, no extra LLM calls).
    Returns: 'fr' | 'en' | 'es' | 'ar'
    """
    s = (q or "").strip()
    if not s:
        return "en"

    # Arabic script detection
    for ch in s:
        if "\u0600" <= ch <= "\u06FF" or "\u0750" <= ch <= "\u077F" or "\u08A0" <= ch <= "\u08FF":
            return "ar"

    sl = s.lower()

    # French markers
    fr_markers = [
        " le ", " la ", " les ", " des ", " du ", " de ", " une ", " un ",
        "pour ", "avec ", "sans ", "chez ", "dans ", "sur ", "est ", "sont ",
        "comment ", "pourquoi ", "quand ", "quel ", "quelle ", "quelles ", "quels ",
        "prise en charge", "recommandation", "contre-indication", "effets indésirables",
        "injection", "toxine botulinique", "acide hyaluronique"
    ]
    fr_hits = sum(1 for m in fr_markers if m in sl)

    # Spanish markers
    es_markers = [
        " el ", " la ", " los ", " las ", " de ", " del ", " una ", " un ",
        "para ", "con ", "sin ", "cómo ", "por qué ", "cuándo ", "qué ",
        "manejo", "recomendación", "contraindicación", "efectos adversos",
        "inyección", "toxina botulínica", "ácido hialurónico"
    ]
    es_hits = sum(1 for m in es_markers if m in sl)

    if fr_hits >= 2 and fr_hits >= es_hits:
        return "fr"
    if es_hits >= 2 and es_hits > fr_hits:
        return "es"
    return "en"

def language_label(lang: str) -> str:
    return {"en": "English", "fr": "French", "es": "Spanish", "ar": "Arabic"}.get(lang, "English")
