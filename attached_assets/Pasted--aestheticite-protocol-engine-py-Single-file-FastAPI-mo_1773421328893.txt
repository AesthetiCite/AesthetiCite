# aestheticite_protocol_engine.py
#
# Single-file FastAPI module for:
# 1) Structured emergency protocols for:
#    - vascular occlusion
#    - anaphylaxis
#    - Tyndall effect
#    - ptosis
# 2) Live evidence hook
# 3) One-click "Protocol + Evidence + PDF export"
#
# Requirements:
#   pip install fastapi pydantic reportlab
#
# How to use:
#   from aestheticite_protocol_engine import router as protocol_router
#   app.include_router(protocol_router)
#
# Endpoints:
#   POST /api/protocols/run         -> JSON protocol + evidence
#   POST /api/protocols/export-pdf  -> generates PDF and returns file path + download URL hint
#
# Notes:
# - Replace DummyEvidenceRetriever with your real RAG / citation layer.
# - The PDF export is intentionally simple, clinical, and print-friendly.
# - This is decision support, not a substitute for clinician judgment.

class ClinicalContext(BaseModel):
    region: Optional[str] = None
    product_type: Optional[str] = None
    symptoms: List[str] = Field(default_factory=list)
    time_since_injection_minutes: Optional[int] = None
    visual_symptoms: Optional[bool] = None
    skin_color_change: Optional[str] = None
    pain_score_10: Optional[int] = Field(default=None, ge=0, le=10)

    tenderness: Optional[bool] = None
    warmth: Optional[bool] = None
    erythema: Optional[bool] = None
    capillary_refill_delayed: Optional[bool] = None

    wheeze: Optional[bool] = None
    hypotension: Optional[bool] = None
    facial_or_tongue_swelling: Optional[bool] = None
    generalized_urticaria: Optional[bool] = None

    eyelid_droop: Optional[bool] = None
    brow_heaviness: Optional[bool] = None
    diplopia: Optional[bool] = None

    filler_confirmed_ha: Optional[bool] = None
    toxin_recent: Optional[bool] = None

    free_text: Optional[str] = None


class ProtocolRequest(BaseModel):
    query: str = Field(..., min_length=3)
    context: Optional[ClinicalContext] = None
    clinician_id: Optional[str] = None
    clinic_id: Optional[str] = None
    mode: Literal["decision_support", "emergency", "teaching"] = "decision_support"


class EvidenceItem(BaseModel):
    source_id: str
    title: str
    note: str
    citation_text: Optional[str] = None
    source_type: Optional[str] = None
    relevance_score: Optional[float] = None


class ProtocolStep(BaseModel):
    step_number: int
    action: str
    rationale: str
    priority: Literal["primary", "secondary"] = "primary"


class DoseGuidance(BaseModel):
    substance: str
    recommendation: str
    notes: str


class RiskAssessment(BaseModel):
    risk_score: int = Field(..., ge=0, le=100)
    severity: SeverityLevel
    urgency: UrgencyLevel
    likely_time_critical: bool
    evidence_strength: EvidenceStrength


class ProtocolResponse(BaseModel):
    request_id: str
    engine_version: str
    protocol_revision: str
    generated_at_utc: str

    matched_protocol_key: str
    matched_protocol_name: str
    confidence: float

    risk_assessment: RiskAssessment
    clinical_summary: str
    immediate_actions: List[ProtocolStep]
    dose_guidance: List[DoseGuidance]
    red_flags: List[str]
    escalation: List[str]
    monitoring: List[str]
    limitations: List[str]
    follow_up_questions: List[str]
    evidence: List[EvidenceItem]
    disclaimer: str


class ExportPDFResponse(BaseModel):
    request_id: str
    pdf_path: str
    filename: str


# ---------------------------------------------------------------------
# Utility
# ---------------------------------------------------------------------

def now_utc_iso() -> str:
    return datetime.now(timezone.utc).isoformat()


def normalize(text: str) -> str:
    text = text.lower().strip()
    text = re.sub(r"\s+", " ", text)
    return text


def joined_context_text(context: Optional[ClinicalContext]) -> str:
    if not context:
        return ""
    parts: List[str] = []
    if context.region:
        parts.append(context.region)
    if context.product_type:
        parts.append(context.product_type)
    if context.symptoms:
        parts.extend(context.symptoms)
    if context.free_text:
        parts.append(context.free_text)
    return normalize(" ".join(parts))

# ---------------------------------------------------------------------
# Matching
# ---------------------------------------------------------------------

def score_protocol(query: str, context: Optional[ClinicalContext], protocol: Dict[str, Any]) -> float:
    score = 0.0
    q = normalize(query)
    ctext = joined_context_text(context)

    for kw in protocol["keywords"]:
        if kw in q:
            score += 1.0
        if kw in ctext:
            score += 0.5

    if context:
        if context.visual_symptoms and "vascular occlusion" in protocol["name"].lower():
            score += 3.0
        if context.capillary_refill_delayed and "vascular occlusion" in protocol["name"].lower():
            score += 2.0
        if context.skin_color_change:
            scc = context.skin_color_change.lower()
            if scc in ("blanching", "mottling", "dusky", "violaceous") and "vascular occlusion" in protocol["name"].lower():
                score += 2.0
            if ("blue" in scc or "bluish" in scc or "blue-gray" in scc) and "tyndall" in protocol["name"].lower():
                score += 2.0

        if context.wheeze or context.hypotension or context.facial_or_tongue_swelling or context.generalized_urticaria:
            if "anaphylaxis" in protocol["name"].lower():
                score += 3.0

        if context.eyelid_droop or context.brow_heaviness or context.diplopia or context.toxin_recent:
            if "ptosis" in protocol["name"].lower():
                score += 2.5

        if context.product_type and "hyaluronic" in context.product_type.lower():
            if "hyaluronic acid filler" in protocol["name"].lower():
                score += 1.0

    return score


def select_protocol(query: str, context: Optional[ClinicalContext]) -> Tuple[str, Dict[str, Any], float]:
    best_key: Optional[str] = None
    best_protocol: Optional[Dict[str, Any]] = None
    best_score = -1.0

    for key, protocol in PROTOCOL_LIBRARY.items():
        score = score_protocol(query, context, protocol)
        if score > best_score:
            best_score = score
            best_key = key
            best_protocol = protocol

    if not best_key or not best_protocol or best_score <= 0:
        raise HTTPException(status_code=404, detail="No matching protocol found.")

    confidence = min(0.99, round(0.45 + best_score * 0.06, 2))
    return best_key, best_protocol, confidence


# ---------------------------------------------------------------------
# Risk assessment
# ---------------------------------------------------------------------

def compute_risk_assessment(protocol_key: str, protocol: Dict[str, Any], context: Optional[ClinicalContext]) -> RiskAssessment:
    base_map = {"low": 20, "moderate": 50, "high": 75, "critical": 92}
    severity: SeverityLevel = protocol["base_severity"]
    urgency: UrgencyLevel = protocol["base_urgency"]
    score = base_map[severity]

    if context:
        if protocol_key == "vascular_occlusion":
            if context.visual_symptoms:
                score += 8
                severity = "critical"
                urgency = "immediate"
            if context.capillary_refill_delayed:
                score += 5
            if context.pain_score_10 is not None and context.pain_score_10 >= 8:
                score += 4

        if protocol_key == "anaphylaxis":
            if context.hypotension or context.wheeze or context.facial_or_tongue_swelling:
                score += 6
                severity = "critical"
                urgency = "immediate"

        if protocol_key == "ptosis_after_toxin":
            if context.diplopia or context.visual_symptoms:
                score += 10
                urgency = "urgent"
                severity = "high"

    score = max(0, min(100, score))
    return RiskAssessment(
        risk_score=score,
        severity=severity,
        urgency=urgency,
        likely_time_critical=urgency in ("urgent", "immediate"),
        evidence_strength=protocol["evidence_strength"],
    )


# ---------------------------------------------------------------------
# Builders
# ---------------------------------------------------------------------

def build_steps(protocol: Dict[str, Any]) -> List[ProtocolStep]:
    return [
        ProtocolStep(
            step_number=i + 1,
            action=s["action"],
            rationale=s["rationale"],
            priority=s.get("priority", "primary"),
        )
        for i, s in enumerate(protocol["steps"])
    ]


def build_doses(protocol: Dict[str, Any]) -> List[DoseGuidance]:
    return [
        DoseGuidance(
            substance=d["substance"],
            recommendation=d["recommendation"],
            notes=d["notes"],
        )
        for d in protocol["dose_guidance"]
    ]


def build_response(
    request_id: str,
    protocol_key: str,
    protocol: Dict[str, Any],
    confidence: float,
    query: str,
    context: Optional[ClinicalContext],
) -> ProtocolResponse:
    evidence = evidence_retriever.retrieve(protocol_key, query, context)
    risk = compute_risk_assessment(protocol_key, protocol, context)

    return ProtocolResponse(
        request_id=request_id,
        engine_version=ENGINE_VERSION,
        protocol_revision=PROTOCOL_REVISION,
        generated_at_utc=now_utc_iso(),
        matched_protocol_key=protocol_key,
        matched_protocol_name=protocol["name"],
        confidence=confidence,
        risk_assessment=risk,
        clinical_summary=protocol["summary"],
        immediate_actions=build_steps(protocol),
        dose_guidance=build_doses(protocol),
        red_flags=list(protocol["red_flags"]),
        escalation=list(protocol["escalation"]),
        monitoring=list(protocol["monitoring"]),
        limitations=list(protocol["limitations"]),
        follow_up_questions=list(protocol["follow_up_questions"]),
        evidence=evidence,
        disclaimer=(
            "This output is clinical decision support and not a substitute for clinician judgment. "
            "Escalate immediately for visual symptoms, airway compromise, shock, progressive ischemia, "
            "or diagnostic uncertainty."
        ),
    )

class PDFBuilder:
    def __init__(self, path: str) -> None:
        self.path = path
        self.c = canvas.Canvas(path, pagesize=A4)
        self.width, self.height = A4
        self.left = 18 * mm
        self.right = self.width - 18 * mm
        self.top = self.height - 18 * mm
        self.bottom = 18 * mm
        self.y = self.top

    def _new_page(self) -> None:
        self.c.showPage()
        self.y = self.top

    def _ensure_space(self, needed: float) -> None:
        if self.y - needed < self.bottom:
            self._new_page()

    def line(self, text: str, font: str = "Helvetica", size: int = 10, leading: int = 14) -> None:
        self._ensure_space(leading)
        self.c.setFont(font, size)
        self.c.drawString(self.left, self.y, text)
        self.y -= leading

    def wrapped_text(
        self,
        text: str,
        font: str = "Helvetica",
        size: int = 10,
        leading: int = 14,
        bullet: Optional[str] = None,
    ) -> None:
        max_width = self.right - self.left
        self.c.setFont(font, size)

        prefix = f"{bullet} " if bullet else ""
        available_first = max_width - stringWidth(prefix, font, size)
        words = text.split()
        current = ""
        first_line = True

        def flush_line(line: str, is_first: bool) -> None:
            self._ensure_space(leading)
            self.c.setFont(font, size)
            if is_first and bullet:
                self.c.drawString(self.left, self.y, prefix + line)
            else:
                indent = stringWidth(prefix, font, size) if bullet else 0
                self.c.drawString(self.left + indent, self.y, line)
            self.y -= leading

        for word in words:
            test = word if not current else f"{current} {word}"
            width_limit = available_first if first_line else max_width - (stringWidth(prefix, font, size) if bullet else 0)
            if stringWidth(test, font, size) <= width_limit:
                current = test
            else:
                flush_line(current, first_line)
                first_line = False
                current = word

        if current:
            flush_line(current, first_line)

    def section_title(self, title: str) -> None:
        self.y -= 4
        self.line(title, font="Helvetica-Bold", size=12, leading=16)

    def save(self) -> None:
        self.c.save()


def export_protocol_pdf(response: ProtocolResponse) -> str:
    filename = f"protocol_{response.matched_protocol_key}_{response.request_id}.pdf"
    path = os.path.join(EXPORT_DIR, filename)
    pdf = PDFBuilder(path)

    pdf.line("AesthetiCite Clinical Protocol Export", font="Helvetica-Bold", size=16, leading=20)
    pdf.line(f"Protocol: {response.matched_protocol_name}", font="Helvetica-Bold", size=11)
    pdf.line(f"Request ID: {response.request_id}")
    pdf.line(f"Generated: {response.generated_at_utc}")
    pdf.line(f"Engine Version: {response.engine_version} | Revision: {response.protocol_revision}")
    pdf.line(
        f"Risk Score: {response.risk_assessment.risk_score}/100 | "
        f"Severity: {response.risk_assessment.severity} | "
        f"Urgency: {response.risk_assessment.urgency}"
    )

    pdf.section_title("Clinical Summary")
    pdf.wrapped_text(response.clinical_summary)

    pdf.section_title("Immediate Actions")
    for step in response.immediate_actions:
        pdf.wrapped_text(f"Step {step.step_number}: {step.action}", font="Helvetica-Bold", bullet="•")
        pdf.wrapped_text(f"Rationale: {step.rationale}", bullet="-")

    pdf.section_title("Dose Guidance")
    if response.dose_guidance:
        for dose in response.dose_guidance:
            pdf.wrapped_text(f"{dose.substance}: {dose.recommendation}", font="Helvetica-Bold", bullet="•")
            pdf.wrapped_text(f"Notes: {dose.notes}", bullet="-")
    else:
        pdf.wrapped_text("No dose guidance available.", bullet="•")

    pdf.section_title("Red Flags")
    for item in response.red_flags:
        pdf.wrapped_text(item, bullet="•")

    pdf.section_title("Escalation")
    for item in response.escalation:
        pdf.wrapped_text(item, bullet="•")

    pdf.section_title("Monitoring")
    for item in response.monitoring:
        pdf.wrapped_text(item, bullet="•")

    pdf.section_title("Follow-up Questions")
    for item in response.follow_up_questions:
        pdf.wrapped_text(item, bullet="•")

    pdf.section_title("Limitations")
    for item in response.limitations:
        pdf.wrapped_text(item, bullet="•")

    pdf.section_title("Evidence")
    if response.evidence:
        for e in response.evidence:
            pdf.wrapped_text(f"[{e.source_id}] {e.title}", font="Helvetica-Bold", bullet="•")
            pdf.wrapped_text(e.note, bullet="-")
            if e.citation_text:
                pdf.wrapped_text(f'Citation: "{e.citation_text}"', bullet="-")
    else:
        pdf.wrapped_text("No evidence retrieved.", bullet="•")

    pdf.section_title("Disclaimer")
    pdf.wrapped_text(response.disclaimer)

    pdf.save()
    return path

@router.post("/run", response_model=ProtocolResponse)
def run_protocol(payload: ProtocolRequest) -> ProtocolResponse:
    request_id = str(uuid.uuid4())
    protocol_key, protocol, confidence = select_protocol(payload.query, payload.context)
    return build_response(
        request_id=request_id,
        protocol_key=protocol_key,
        protocol=protocol,
        confidence=confidence,
        query=payload.query,
        context=payload.context,
    )


@router.post("/export-pdf", response_model=ExportPDFResponse)
def run_protocol_and_export_pdf(payload: ProtocolRequest) -> ExportPDFResponse:
    request_id = str(uuid.uuid4())
    protocol_key, protocol, confidence = select_protocol(payload.query, payload.context)
    response = build_response(
        request_id=request_id,
        protocol_key=protocol_key,
        protocol=protocol,
        confidence=confidence,
        query=payload.query,
        context=payload.context,
    )
    pdf_path = export_protocol_pdf(response)
    return ExportPDFResponse(
        request_id=request_id,
        pdf_path=pdf_path,
        filename=os.path.basename(pdf_path),
    )
