import os
from typing import List
from openai import OpenAI

# Environment:
#   export OPENAI_API_KEY="..."
# Optional:
#   export OPENAI_MODEL="gpt-4.1-mini"
#   export OPENAI_EMBED_MODEL="text-embedding-3-small"
#   export OPENAI_EMBED_DIM="1536"  # MUST match EMBED_DIM in your main patch

OPENAI_MODEL = os.getenv("OPENAI_MODEL", "gpt-4.1-mini")
OPENAI_EMBED_MODEL = os.getenv("OPENAI_EMBED_MODEL", "text-embedding-3-small")
OPENAI_EMBED_DIM = int(os.getenv("OPENAI_EMBED_DIM", os.getenv("EMBED_DIM", "1536")))

_client = OpenAI(api_key=os.environ.get("OPENAI_API_KEY"))

def llm_text(system: str, user: str, temperature: float = 0.2) -> str:
    """
    Uses OpenAI Responses API.
    Returns plain text.
    """
    # Responses API: input can be a string; for structured roles, pass a list of messages.
    # We keep it simple and deterministic for clinical output.
    resp = _client.responses.create(
        model=OPENAI_MODEL,
        input=[
            {"role": "system", "content": system},
            {"role": "user", "content": user},
        ],
        temperature=temperature,
    )

    # The SDK returns output text via output_text in most recent versions.
    # Fall back gracefully if the shape changes.
    if hasattr(resp, "output_text") and resp.output_text:
        return resp.output_text

    # Fallback: try to reconstruct text from output items
    try:
        parts = []
        for item in (resp.output or []):
            for c in (getattr(item, "content", None) or []):
                t = getattr(c, "text", None)
                if t:
                    parts.append(t)
        return "\n".join(parts).strip()
    except Exception:
        return str(resp)

def embed(text: str) -> List[float]:
    """
    Uses OpenAI Embeddings API.
    IMPORTANT: Your main patch must set EMBED_DIM to match the embedding model output:
      - text-embedding-3-small -> 1536 dims
      - text-embedding-3-large -> 3072 dims
    """
    # Keep queries short-ish for speed; your retrieval chunks should be embedded at ingestion time.
    text = (text or "").strip()
    if not text:
        return [0.0] * OPENAI_EMBED_DIM

    emb = _client.embeddings.create(
        model=OPENAI_EMBED_MODEL,
        input=text,
        # If you want to force smaller vectors with text-embedding-3-*,
        # you can pass dimensions=... (supported on these models).
        # dimensions=OPENAI_EMBED_DIM,
    )

    vec = emb.data[0].embedding
    if len(vec) != OPENAI_EMBED_DIM:
        raise RuntimeError(
            f"Embedding dim mismatch: got {len(vec)} but OPENAI_EMBED_DIM={OPENAI_EMBED_DIM}. "
            f"Fix EMBED_DIM/OPENAI_EMBED_DIM or switch embedding model."
        )
    return vec
